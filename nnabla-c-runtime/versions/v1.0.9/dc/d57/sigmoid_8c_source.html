34 rt_function_error_t allocate_sigmoid_local_context(rt_function_t *f) {

36  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;

47  f->local_context = (void *)c;

74 rt_function_error_t exec_sigmoid(rt_function_t *f) {
75  sigmoid_local_context_t *c = (sigmoid_local_context_t *)(f->local_context);
76  float *x = (float *)(c->input->data);
77  float *y = (float *)(c->output->data);
78 
79  int i; // Iterator
80  for (i = 0; i < c->output_size; i++) {
81  y[i] = 1.0f / (1.0f + expf(-x[i]));
82  }
83  return RT_FUNCTION_ERROR_NOERROR;
84 }

86 rt_function_error_t exec_sigmoid_generic(rt_function_t *f) {
87  sigmoid_local_context_t *c = (sigmoid_local_context_t *)(f->local_context);
88 
89  int i; // Iterator
90  for (i = 0; i < c->output_size; i++) {
91  float val_x = c->get_input(c->input, i);
92  float val_y = 1.0f / (1.0f + expf(-val_x));
93  c->set_output(c->output, i, val_y);
94  }
95  return RT_FUNCTION_ERROR_NOERROR;
