38 rt_function_error_t allocate_prelu_local_context(rt_function_t *f) {
39  if (f->num_of_inputs != 2) {
40  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
41  }
42  if (f->num_of_outputs != 1) {
43  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
44  }
45 
46  prelu_private_t *p = rt_malloc_func(sizeof(prelu_private_t));
47  if (p == 0) {
48  return RT_FUNCTION_ERROR_MALLOC;
49  }
50  ((prelu_local_context_t *)(f->local_context))->data = (void *)p;
51  p->input = f->inputs[0];
52  p->get_input = select_getter(p->input);
53  p->input_size = calc_shape_size(f->inputs[0]->shape);
54  p->weight = f->inputs[1];
55  p->get_weight = select_getter(p->weight);
56  p->weight_size = calc_shape_size(f->inputs[1]->shape);
57  p->output = f->outputs[0];
58  p->set_output = select_setter(p->output);
59  p->in_shape = clone_list(f->inputs[0]->shape);
60  p->in_stride = calc_contiguous_strides(f->inputs[0]->shape);
61  if (p->input->type == NN_DATA_TYPE_FLOAT &&
62  p->weight->type == NN_DATA_TYPE_FLOAT &&
63  p->output->type == NN_DATA_TYPE_FLOAT) {
64 #ifdef CONFIG_PRELU_FLOAT32
65  f->exec_func = exec_prelu;
66 #endif /* CONFIG_PRELU_FLOAT32 */
67  } else {
68 #ifdef CONFIG_PRELU_GENERIC
69  f->exec_func = exec_prelu_generic;
70 #endif /* CONFIG_PRELU_GENERIC */
71  }
72  return RT_FUNCTION_ERROR_NOERROR;
73 }

84 #ifdef CONFIG_PRELU_FLOAT32
85 rt_function_error_t exec_prelu(rt_function_t *f) {
86  prelu_local_context_t *context = (prelu_local_context_t *)(f->local_context);
87  prelu_private_t *p = (prelu_private_t *)(context->data);
88  float *x = (float *)(p->input->data);
89  float *w = (float *)(p->weight->data);
90  float *y = (float *)(p->output->data);
91  int base_shape = p->in_shape.data[context->base_axis];
92  int base_stride = p->in_stride.data[context->base_axis];
93 
94  if (p->weight_size == 1) {
95  for (int s = 0; s < p->input_size; ++s) {
96  y[s] = (x[s] >= 0) ? x[s] : x[s] * (*w);
97  }
98  } else {
99  for (int s = 0; s < p->input_size; ++s) {
100  int iw = (s / base_stride) % base_shape;
101  y[s] = (x[s] >= 0) ? x[s] : x[s] * w[iw];
102  }
103  }
104  return RT_FUNCTION_ERROR_NOERROR;
105 }
106 #endif /* CONFIG_PRELU_FLOAT32 */

108 #ifdef CONFIG_PRELU_GENERIC
109 rt_function_error_t exec_prelu_generic(rt_function_t *f) {
110  prelu_local_context_t *context = (prelu_local_context_t *)(f->local_context);
111  prelu_private_t *p = (prelu_private_t *)(context->data);
112  int base_shape = p->in_shape.data[context->base_axis];
113  int base_stride = p->in_stride.data[context->base_axis];
114 
115  if (p->weight_size == 1) {
116  float w = p->get_weight(p->weight, 0);
117  for (int s = 0; s < p->input_size; ++s) {
118  float x = p->get_input(p->input, s);
119  float y = (x >= 0) ? x : x * w;
120  p->set_output(p->output, s, y);
121  }
122  } else {
123  for (int s = 0; s < p->input_size; ++s) {
124  int iw = (s / base_stride) % base_shape;
125  float x = p->get_input(p->input, s);
126  float w = p->get_weight(p->weight, iw);
127  float y = (x >= 0) ? x : x * w;
128  p->set_output(p->output, s, y);
129  }
130  }
131  return RT_FUNCTION_ERROR_NOERROR;
132 }
