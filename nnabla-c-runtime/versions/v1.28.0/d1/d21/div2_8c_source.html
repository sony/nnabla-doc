28 rt_function_error_t allocate_div2_local_context(rt_function_t *f) {
29  if (f->num_of_inputs != 2) {
30  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
31  }
32  if (f->num_of_outputs != 1) {
33  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
34  }
35  if (f->inputs[0]->shape.size != f->inputs[1]->shape.size) {
36  return RT_FUNCTION_ERROR_INVALID_SHAPE;
37  }
38  if (f->outputs[0]->shape.size != f->inputs[0]->shape.size) {
39  return RT_FUNCTION_ERROR_INVALID_SHAPE;
40  }
41  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&
42  f->inputs[1]->type == NN_DATA_TYPE_FLOAT &&
43  f->outputs[0]->type == NN_DATA_TYPE_FLOAT) {
44 #ifdef CONFIG_DIV2_FLOAT32
45  f->exec_func = exec_div2;
46 #endif /* CONFIG_DIV2_FLOAT32 */
47  }
48 
49  else if (f->inputs[0]->type == NN_DATA_TYPE_INT16 &&
50  f->inputs[1]->type == NN_DATA_TYPE_INT16 &&
51  f->outputs[0]->type == NN_DATA_TYPE_INT16) {
52 #ifdef CONFIG_DIV2_FIXED16
53  f->exec_func = exec_div2_fixed16;
54 #endif /* CONFIG_DIV2_FIXED16 */
55  }
56 
57  else if (f->inputs[0]->type == NN_DATA_TYPE_INT8 &&
58  f->inputs[1]->type == NN_DATA_TYPE_INT8 &&
59  f->outputs[0]->type == NN_DATA_TYPE_INT8) {
60 #ifdef CONFIG_DIV2_FIXED8
61  f->exec_func = exec_div2_fixed8;
62 #endif /* CONFIG_DIV2_FIXED8 */
63  }
64 
65  else {
66 #ifdef CONFIG_DIV2_GENERIC
67  f->exec_func = exec_div2_generic;
68 #endif /* CONFIG_DIV2_GENERIC */
69  }
70  return RT_FUNCTION_ERROR_NOERROR;
71 }
