39 rt_function_error_t allocate_unpooling_local_context(rt_function_t *f) {
40  unpooling_local_context_t *context =
41  (unpooling_local_context_t *)(f->local_context);
42  unpooling_private_t *p = rt_malloc_func(sizeof(unpooling_private_t));
43  if (p == 0) {
44  return RT_FUNCTION_ERROR_MALLOC;
45  }
46  p->input_shape = clone_list(f->inputs[0]->shape);
47  p->output_shape = clone_list(f->outputs[0]->shape);
48  p->input_strides = calc_contiguous_strides(f->inputs[0]->shape);
49  p->output_strides = calc_contiguous_strides(f->outputs[0]->shape);
50  p->input = f->inputs[0];
51  p->get_input = select_getter(p->input);
52  p->output = f->outputs[0];
53  p->set_output = select_setter(p->output);
54 
55  if (context->kernel.size > p->input_shape.size) {
56  free_list(p->input_shape);
57  free_list(p->output_shape);
58  free_list(p->input_strides);
59  free_list(p->output_strides);
60  rt_free_func(p);
61  return RT_FUNCTION_ERROR_INVALID_SHAPE;
62  }
63 
64  // Calc and set output shape.
65  rt_list_t shape = allocate_list(p->input_shape.size);
66  int diff = p->input_shape.size - context->kernel.size;
67  int i;
68  for (i = 0; i < diff; i++) {
69  shape.data[i] = 1;
70  }
71  for (i = diff; i < p->input_shape.size; i++) {
72  shape.data[i] = context->kernel.data[i - diff];
73  }
74  p->kernel = clone_list(shape);
75  for (i = 0; i < p->input_shape.size; i++) {
76  p->output_shape.data[i] = p->input_shape.data[i] * p->kernel.data[i];
77  }
78  free_list(shape);
79 
80  ((unpooling_local_context_t *)(f->local_context))->data = (void *)p;
81 
82  f->exec_func = exec_unpooling;
83 
84  return RT_FUNCTION_ERROR_NOERROR;
85 }

101 static void unpooling_forward_recursive_generic(unpooling_private_t *p,
102  int x_offset, int y_offset,
103  int dim) {
104  int current_x_offset = x_offset;
105  int current_y_offset = y_offset;
106  const int x_stride = p->input_strides.data[dim];
107  const int y_stride = p->output_strides.data[dim];
108  const int kernel = p->kernel.data[dim];
109  const int size = p->output_shape.data[dim];
110 
111  if (dim == p->input_shape.size - 1) {
112  if (x_stride == 1 && kernel == 1) {
113  for (int i = 0; i < size; i++) {
114  float value = p->get_input(p->input, i + current_x_offset);
115  p->set_output(p->output, i + current_y_offset, value);
116  }
117  } else {
118  int count = 0;
119  for (int i = 0; i < size; i++) {
120  float value = p->get_input(p->input, current_x_offset);
121  p->set_output(p->output, current_y_offset, value);
122  if (++count >= kernel) {
123  count = 0;
124  current_x_offset += x_stride;
125  }
126  current_y_offset += y_stride;
127  }
128  }
129  } else {
130  int count = 0;
131  int i;
132  for (i = 0; i < size; i++) {
133  unpooling_forward_recursive_generic(p, current_x_offset, current_y_offset,
134  dim + 1);
135  if (++count >= kernel) {
136  count = 0;
137  current_x_offset += x_stride;
138  }
139  current_y_offset += y_stride;
140  }
141  }
142 }

145 #ifdef CONFIG_UNPOOLING_FLOAT32
146 static void unpooling_forward_recursive(unpooling_private_t *p, int x_offset,
147  int y_offset, int dim) {
148  int current_x_offset = x_offset;
149  int current_y_offset = y_offset;
150  const int x_stride = p->input_strides.data[dim];
151  const int y_stride = p->output_strides.data[dim];
152  const int kernel = p->kernel.data[dim];
153  const int size = p->output_shape.data[dim];
154  const float *x = (float *)(p->input->data);
155  float *y = (float *)(p->output->data);
156 
157  if (dim == p->input_shape.size - 1) {
158  const float *current_x = x + current_x_offset;
159  float *current_y = y + current_y_offset;
160  if (x_stride == 1 && kernel == 1) {
161  memcpy(current_y, current_x, sizeof(float) * size);
162  } else {
163  const float *end_y = current_y + size * y_stride;
164  int count = 0;
165  while (current_y != end_y) {
166  *current_y = *current_x;
167  if (++count >= kernel) {
168  count = 0;
169  current_x += x_stride;
170  }
171  current_y += y_stride;
172  }
173  }
174  } else {
175  int count = 0;
176  int i;
177  for (i = 0; i < size; i++) {
178  unpooling_forward_recursive(p, current_x_offset, current_y_offset,
179  dim + 1);
180  if (++count >= kernel) {
181  count = 0;
182  current_x_offset += x_stride;
183  }
184  current_y_offset += y_stride;
185  }
186  }
187 }
188 #endif /* CONFIG_UNPOOLING_FLOAT32 */

191  unpooling_private_t *p =
192  (unpooling_private_t *)(((unpooling_local_context_t *)(f->local_context))

196 #ifdef CONFIG_UNPOOLING_FLOAT32

201  unpooling_forward_recursive_generic(p, 0, 0, 0);
202 #endif /* CONFIG_UNPOOLING_GENERIC */
203  }
204  return RT_FUNCTION_ERROR_NOERROR;
