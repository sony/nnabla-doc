80  stack_private_t *p = (stack_private_t *)(c->data);
81 
82  float *y = (float *)(f->outputs[0]->data);

85  for (int j = 0; j < p->outer_size; j++) {
86  for (int k = 0; k < p->inner_size; k++) {

88  x[j * p->inner_size + k];
