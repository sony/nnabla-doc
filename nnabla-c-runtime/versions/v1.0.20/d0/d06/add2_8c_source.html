24 rt_function_error_t allocate_add2_local_context(rt_function_t *f) {
25  if (f->num_of_inputs != 2) {
26  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
27  }
28  if (f->num_of_outputs != 1) {
29  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
30  }
31  if (f->inputs[0]->shape.size != f->inputs[1]->shape.size) {
32  return RT_FUNCTION_ERROR_INVALID_SHAPE;
33  }
34  if (f->outputs[0]->shape.size != f->inputs[0]->shape.size) {
35  return RT_FUNCTION_ERROR_INVALID_SHAPE;
36  }
37 
38  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&
39  f->inputs[1]->type == NN_DATA_TYPE_FLOAT &&

45 #ifdef CONFIG_ADD2_GENERIC
