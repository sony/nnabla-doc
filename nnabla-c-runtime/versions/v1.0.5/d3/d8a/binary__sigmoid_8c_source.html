33  if (f->num_of_inputs != 1) {
34  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
35  }
36  if (f->num_of_outputs != 1) {
37  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
38  }
39  binary_sigmoid_private_context_t *p =
40  malloc(sizeof(binary_sigmoid_private_context_t));
41  if (p == 0) {
42  return RT_FUNCTION_ERROR_MALLOC;

44  f->local_context = p;
45  p->input = f->inputs[0];

67 rt_function_error_t exec_binary_sigmoid(rt_function_t *f) {
68  binary_sigmoid_private_context_t *p =
69  (binary_sigmoid_private_context_t *)(f->local_context);
70  float *input = (float *)(p->input->data);
71  float *output = (float *)(p->output->data);
72  int i; // Iterator
73  for (i = 0; i < p->output_size; i++) {
74  float x = (input[i] + 1) / 2;
75  if (x < 0) {
76  x = 0;
77  }
78  if (x > 1) {
79  x = 1;
80  }
81  output[i] = roundf(x);
82  }
83  return RT_FUNCTION_ERROR_NOERROR;
84 }

86 rt_function_error_t exec_binary_sigmoid_generic(rt_function_t *f) {
87  binary_sigmoid_private_context_t *p =
88  (binary_sigmoid_private_context_t *)(f->local_context);
89  int i; // Iterator
90  for (i = 0; i < p->output_size; i++) {
91  float x = (p->get_input(p->input, i) + 1) / 2;
92  if (x < 0) {
93  x = 0;
94  }
95  if (x > 1) {
96  x = 1;
97  }
98  p->set_output(p->output, i, roundf(x));
99  }
100  return RT_FUNCTION_ERROR_NOERROR;
