35 rt_function_error_t allocate_prelu_local_context(rt_function_t *f) {
36  if (f->num_of_inputs != 2) {
37  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
38  }
39  if (f->num_of_outputs != 1) {
40  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
41  }
42 
43  prelu_private_t *p = malloc(sizeof(prelu_private_t));
44  if (p == 0) {
45  return RT_FUNCTION_ERROR_MALLOC;
46  }
47  ((prelu_local_context_t *)(f->local_context))->data = (void *)p;
48  p->input = f->inputs[0];
49  p->get_input = select_getter(p->input);
50  p->input_size = calc_shape_size(f->inputs[0]->shape);
51  p->weight = f->inputs[1];
52  p->get_weight = select_getter(p->weight);
53  p->weight_size = calc_shape_size(f->inputs[1]->shape);
54  p->output = f->outputs[0];
55  p->set_output = select_setter(p->output);
56  p->in_shape = clone_list(f->inputs[0]->shape);
57  p->in_stride = calc_contiguous_strides(f->inputs[0]->shape);
58  if (p->input->type == NN_DATA_TYPE_FLOAT &&
59  p->weight->type == NN_DATA_TYPE_FLOAT &&
60  p->output->type == NN_DATA_TYPE_FLOAT) {
61  f->exec_func = exec_prelu;
62  } else {
63  f->exec_func = exec_prelu_generic;
64  }
65  return RT_FUNCTION_ERROR_NOERROR;
66 }

77 rt_function_error_t exec_prelu(rt_function_t *f) {
78  prelu_local_context_t *context = (prelu_local_context_t *)(f->local_context);
79  prelu_private_t *p = (prelu_private_t *)(context->data);
80  float *x = (float *)(p->input->data);
81  float *w = (float *)(p->weight->data);
82  float *y = (float *)(p->output->data);
83  int base_shape = p->in_shape.data[context->base_axis];
84  int base_stride = p->in_stride.data[context->base_axis];
85 
86  if (p->weight_size == 1) {
87  for (int s = 0; s < p->input_size; ++s) {
88  y[s] = (x[s] >= 0) ? x[s] : x[s] * (*w);
89  }
90  } else {
91  for (int s = 0; s < p->input_size; ++s) {
92  int iw = (s / base_stride) % base_shape;
93  y[s] = (x[s] >= 0) ? x[s] : x[s] * w[iw];
94  }
95  }
96  return RT_FUNCTION_ERROR_NOERROR;
97 }

99 rt_function_error_t exec_prelu_generic(rt_function_t *f) {
100  prelu_local_context_t *context = (prelu_local_context_t *)(f->local_context);
101  prelu_private_t *p = (prelu_private_t *)(context->data);
102  int base_shape = p->in_shape.data[context->base_axis];
103  int base_stride = p->in_stride.data[context->base_axis];
104 
105  if (p->weight_size == 1) {
106  float w = p->get_weight(p->weight, 0);
107  for (int s = 0; s < p->input_size; ++s) {
108  float x = p->get_input(p->input, s);
109  float y = (x >= 0) ? x : x * w;
110  p->set_output(p->output, s, y);
111  }
112  } else {
113  for (int s = 0; s < p->input_size; ++s) {
114  int iw = (s / base_stride) % base_shape;
115  float x = p->get_input(p->input, s);
116  float w = p->get_weight(p->weight, iw);
117  float y = (x >= 0) ? x : x * w;
118  p->set_output(p->output, s, y);
119  }
120  }
121  return RT_FUNCTION_ERROR_NOERROR;
122 }
