24  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;

26  if (f->num_of_outputs != 1) {

28  }
29  int input_size = calc_shape_size(f->inputs[0]->shape);
30  int output_size = calc_shape_size(f->outputs[0]->shape);
31  if (input_size != output_size) {

33  }
34  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&
35  f->outputs[0]->type == NN_DATA_TYPE_FLOAT) {

38  f->exec_func = exec_pow_scalar_generic;
39  }
40  return RT_FUNCTION_ERROR_NOERROR;
41 }
