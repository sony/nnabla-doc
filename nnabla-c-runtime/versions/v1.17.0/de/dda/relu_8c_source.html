40 rt_function_error_t allocate_relu_local_context(rt_function_t *f) {
41  if (f->num_of_inputs != 1) {
42  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
43  }
44  if (f->num_of_outputs != 1) {
45  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
46  }
47 
48  relu_private_t *p = rt_malloc_func(sizeof(relu_private_t));
49  if (p == 0) {
50  return RT_FUNCTION_ERROR_MALLOC;
51  }
52 
53  p->input = f->inputs[0];
54  p->get_input = select_getter(p->input);
55  p->input_size = calc_shape_size(f->inputs[0]->shape);
56 
57  p->output = f->outputs[0];
58  p->set_output = select_setter(p->output);
59  p->output_size = calc_shape_size(f->outputs[0]->shape);
60 
61  if (p->input_size != p->output_size) {
62  rt_free_func(p);
63  return RT_FUNCTION_ERROR_INVALID_SHAPE;
64  }
65  ((relu_local_context_t *)(f->local_context))->data = (void *)p;
66  if (p->input->type == NN_DATA_TYPE_FLOAT &&
67  p->output->type == NN_DATA_TYPE_FLOAT) {
68 #ifdef CONFIG_RELU_FLOAT32
69  f->exec_func = exec_relu;
70 #endif /* CONFIG_RELU_FLOAT32 */
71  } else if (p->input->type == NN_DATA_TYPE_INT8 &&
72  p->output->type == NN_DATA_TYPE_INT8) {
73 #ifdef CONFIG_RELU_FIXED8
74  f->exec_func = exec_relu_fixed8;
75 #endif /* CONFIG_RELU_FIXED8 */
76  } else if (p->input->type == NN_DATA_TYPE_INT16 &&
77  p->output->type == NN_DATA_TYPE_INT16) {
78 #ifdef CONFIG_RELU_FIXED16
79  f->exec_func = exec_relu_fixed16;
80 #endif /* CONFIG_RELU_FIXED16 */
81  } else {
82 #ifdef CONFIG_RELU_GENERIC
83  f->exec_func = exec_relu_generic;
84 #endif /* CONFIG_RELU_GENERIC */
85  }
86  return RT_FUNCTION_ERROR_NOERROR;
87 }

95 rt_function_error_t exec_relu(rt_function_t *f) {
96  relu_local_context_t *context = (relu_local_context_t *)(f->local_context);
97  relu_private_t *p = (relu_private_t *)(context->data);
98  float *x = (float *)(p->input->data);
99  float *y = (float *)(p->output->data);
100 
101  int i; // Iterator
102  for (i = 0; i < p->output_size; i++) {
103  y[i] = (x[i] > 0.0f) ? x[i] : 0.0f;
104  }
105  return RT_FUNCTION_ERROR_NOERROR;
106 }
