29  rt_variable_getter get_input2 = select_getter(input2);

32  int x_in1 = (x_out % x_size_in1);
33  int x_in2 = (x_out % x_size_in2);
34  int z_out = y_out * x_size_out + x_out;
35  int z_in1 = y_in1 * x_size_in1 + x_in1;
36  int z_in2 = y_in2 * x_size_in2 + x_in2;

38  calc_dim_arithmetic_generic(output, input1, input2, dim_index + 1, z_out,
39  z_in1, z_in2, calc_func);
40  } else {
41  float data_in1 = get_input1(input1, z_in1);
42  float data_in2 = get_input2(input2, z_in2);
43  set_output(output, z_out, calc_func(data_in1, data_in2));
