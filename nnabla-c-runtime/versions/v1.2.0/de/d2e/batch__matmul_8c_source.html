47 rt_function_error_t allocate_batch_matmul_local_context(rt_function_t *f) {
48  if (f->num_of_inputs != 2) {
49  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
50  }
51  if (f->num_of_outputs != 1) {
52  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
53  }
54 
55  batch_matmul_local_context_t *context =
56  (batch_matmul_local_context_t *)(f->local_context);
57  batch_matmul_private_t *p = rt_malloc_func(sizeof(batch_matmul_private_t));
58  if (p == 0) {
59  return RT_FUNCTION_ERROR_MALLOC;
60  }
61  ((batch_matmul_local_context_t *)(f->local_context))->data = (void *)p;
62  p->row_a = f->inputs[0]->shape.data[f->inputs[0]->shape.size - 2];
63  p->col_a = f->inputs[0]->shape.data[f->inputs[0]->shape.size - 1];
64  p->row_b = f->inputs[1]->shape.data[f->inputs[1]->shape.size - 2];
65  p->col_b = f->inputs[1]->shape.data[f->inputs[1]->shape.size - 1];
66  p->row_y = context->transpose_a ? p->col_a : p->row_a;
67  p->col_y = context->transpose_b ? p->row_b : p->col_b;
68  p->offset_a = p->row_a * p->col_a;
69  p->offset_b = p->row_b * p->col_b;
70  p->offset_y = p->row_y * p->col_y;
71 
72  p->input_a = f->inputs[0];
73  p->get_input_a = select_getter(p->input_a);
74  p->input_b = f->inputs[1];
75  p->get_input_b = select_getter(p->input_b);
76 
77  p->samples = 1;
78  for (int i = 0; i < f->inputs[0]->shape.size - 2; ++i) {
79  p->samples *= f->inputs[0]->shape.data[i];
80  }
81 
82  int samples_b = 1;
83  for (int j = 0; j < f->inputs[1]->shape.size - 2; ++j) {
84  samples_b *= f->inputs[1]->shape.data[j];
85  }
86 
87  p->output = f->outputs[0];
88  p->get_output = select_getter(p->output);
89  p->set_output = select_setter(p->output);
90  p->output_size = calc_shape_size(f->outputs[0]->shape);
91 
92  if (p->samples != samples_b) {
93  return RT_FUNCTION_ERROR_INVALID_SHAPE;
94  }
95  if (p->input_a->type == NN_DATA_TYPE_FLOAT &&
96  p->input_b->type == NN_DATA_TYPE_FLOAT &&
97  p->output->type == NN_DATA_TYPE_FLOAT) {
98 #ifdef CONFIG_BATCHMATMUL_FLOAT32
99  f->exec_func = exec_batch_matmul;
100 #endif /* CONFIG_BATCHMATMUL_FLOAT32 */
101  } else {
102 #ifdef CONFIG_BATCHMATMUL_GENERIC
103  f->exec_func = exec_batch_matmul_generic;
104 #endif /* CONFIG_BATCHMATMUL_GENERIC */
105  }
106  return RT_FUNCTION_ERROR_NOERROR;
107 }

114 static int get_next(int i, int m, int n) { return (i % n) * m + i / n; }

116 static int get_pre(int i, int m, int n) { return (i % m) * n + i / m; }

118 static void move_data(float *mtx, int i, int m, int n) {
119  float temp = mtx[i];
120  int cur = i;
121  int pre = get_pre(cur, m, n);
122  while (pre != i) {
123  mtx[cur] = mtx[pre];
124  cur = pre;
125  pre = get_pre(cur, m, n);
126  }
127  mtx[cur] = temp;
128 }

130 static void transpose(float *mtx, int m, int n) {
131  for (int i = 0; i < m * n; i++) {
132  int next = get_next(i, m, n);
133  while (next > i)
134  next = get_next(next, m, n);
135  if (next == i)
136  move_data(mtx, i, m, n);
137  }
138 }

141 rt_function_error_t exec_batch_matmul(rt_function_t *f) {
142  batch_matmul_local_context_t *context =
143  (batch_matmul_local_context_t *)(f->local_context);
144  batch_matmul_private_t *p = (batch_matmul_private_t *)(context->data);
145  float *input_a = (float *)(p->input_a->data);
146  float *input_b = (float *)(p->input_b->data);
147  float *output = (float *)(p->output->data);
148 
149  int i;
150  if (context->transpose_a) {
151  for (i = 0; i < p->samples; i++) {
152  transpose(input_a + p->offset_a * i, p->row_a, p->col_a);
153  }
154  p->row_a = p->row_a ^ p->col_a;
155  p->col_a = p->row_a ^ p->col_a;
156  p->row_a = p->row_a ^ p->col_a;
157  }
158  if (context->transpose_b) {
159  for (i = 0; i < p->samples; i++) {
160  transpose(input_b + p->offset_b * i, p->row_b, p->col_b);
161  }
162  p->row_b = p->row_b ^ p->col_b;
163  p->col_b = p->row_b ^ p->col_b;
164  p->row_b = p->row_b ^ p->col_b;
165  }
166  for (i = 0; i < p->samples; i++) {
167  float *mtx_y = output + p->offset_y * i;
168  float *mtx_a = input_a + p->offset_a * i;
169  float *mtx_b = input_b + p->offset_b * i;
170  int row_a = p->row_a;
171  int col_b = p->col_b;
172  int col_a = p->col_a;
173  for (int j = 0; j < row_a; j++) {
174  for (int k = 0; k < col_b; k++) {
175  for (int l = 0; l < col_a; l++) {
176  float a = *(mtx_a + col_a * j + l);
177  float b = *(mtx_b + col_b * l + k);
178  *(mtx_y + col_b * j + k) += a * b;
179  }
180  }
181  }
182  }
183 
184  return RT_FUNCTION_ERROR_NOERROR;
185 }

189 rt_function_error_t exec_batch_matmul_generic(rt_function_t *f) {
190  batch_matmul_local_context_t *context =
191  (batch_matmul_local_context_t *)(f->local_context);
192  batch_matmul_private_t *p = (batch_matmul_private_t *)(context->data);
193  float *input_a = (float *)(p->input_a->data);
194  float *input_b = (float *)(p->input_b->data);
195 
196  int i;
197  if (context->transpose_a) {
198  for (i = 0; i < p->samples; i++) {
199  transpose(input_a + p->offset_a * i, p->row_a, p->col_a);
200  }
201  p->row_a = p->row_a ^ p->col_a;
202  p->col_a = p->row_a ^ p->col_a;
203  p->row_a = p->row_a ^ p->col_a;
204  }
205  if (context->transpose_b) {
206  for (i = 0; i < p->samples; i++) {
207  transpose(input_b + p->offset_b * i, p->row_b, p->col_b);
208  }
209  p->row_b = p->row_b ^ p->col_b;
210  p->col_b = p->row_b ^ p->col_b;
211  p->row_b = p->row_b ^ p->col_b;
212  }
213  for (i = 0; i < p->samples; i++) {
214  int row_a = p->row_a;
215  int col_b = p->col_b;
216  int col_a = p->col_a;
217  for (int j = 0; j < row_a; j++) {
218  for (int k = 0; k < col_b; k++) {
219  for (int l = 0; l < col_a; l++) {
220  float a = p->get_input_a(p->input_a, p->offset_y * i + col_a * j + l);
221  float b = p->get_input_b(p->input_b, p->offset_b * i + col_b * l + k);
222  float y = p->get_output(p->output, p->offset_y * i + col_b * j + k);
223  y += a * b;
224  p->set_output(p->output, p->offset_y * i + col_b * j + k, y);
225  }
226  }
227  }
228  }
229 
230  return RT_FUNCTION_ERROR_NOERROR;
231 }
