32 ///
33 /// But here is some modification to use callback more flexible.
34 ///
35 /// To use callback, you must set 'function implement flag' in NNB file.
36 /// You can get template setting file with following command.
37 /// @code{.sh}
38 /// $ cp build/test/nnabla/succeed/Affine_000.nntxt .
39 /// $ nnabla_cli nnb_template Affine_000.nntxt Affine_000.yaml
40 /// @endcode
41 ///
42 /// Then you should be got following 'Affine_000.yaml'.
43 /// @code{.yaml}
44 /// functions:
45 /// Affine:
46 /// implement: 0
47 /// variables:
48 /// x: FLOAT32
49 /// weight: FLOAT32
50 /// bias: FLOAT32
51 /// y0: FLOAT32
52 /// @endcode
53 ///
54 /// Edit 'Affine_000.yaml' and save as 'settings.yaml'
55 /// @code{.yaml}
56 /// functions:
57 /// Affine:
58 /// implement: 1
59 /// @endcode
60 ///
61 /// NOTE: If you DON'T want to use callback, set implements to 100.
62 ///
63 /// And convert nntxt to nnb again with settings.yaml.
64 /// @code{.sh}
65 /// $ nnabla_cli convert -s settings.yaml
66 /// nnabla_cli convert -s settings.yaml Affine_000.nntxt Affine_000.nnb
67 /// @endcode

149 /// @brief Example callback for allocate function local context.
150 /// It just does followings.
151 /// - check func->impl == 1
152 /// - register @ref cb_exec as function executor.
153 /// - register @ref cb_free as function local context de-allocator.
154 ///
155 /// See also @ref rt_initialize_context
156 static rt_return_value_t cb_alloc(nn_network_t *net, void *function_context) {
157  WHOAMI("%s", __func__);
158  rt_function_context_t *func = (rt_function_context_t *)function_context;
159  if ((int)func->info->impl != 1) {
160  return RT_RET_FUNCTION_DONT_MATCH;
161  }
162 
163  func->func.exec_func = cb_exec;
164  func->func.free_local_context_func = cb_free;
165  return RT_RET_FUNCTION_MATCH;
166 }

168 /// @brief Simple example to use user defined functions.
169 int main(int argc, char *argv[]) {
170  WHOAMI("Callback test.");
171  rt_context_pointer context = 0;

173  rt_return_value_t ret = rt_allocate_context(&context);
174  assert(ret == RT_RET_NOERROR);
175 
176  ret = rt_add_callback(context, NN_FUNCTION_AFFINE, cb_alloc);
177  assert(ret == RT_RET_NOERROR);
178 
179  nn_network_t *net = (nn_network_t *)affine_nnb;
180  ret = rt_initialize_context(context, net);
181  assert(ret == RT_RET_NOERROR);
182 
183  assert(rt_num_of_input(context) == 1);
184  assert(rt_input_size(context, 0) == sizeof(input));
185  memcpy(rt_input_buffer(context, 0), input, sizeof(input));
186 
187  ret = rt_forward(context);
188  assert(ret == RT_RET_NOERROR);
189 
190  ret = rt_free_context(&context);
191  assert(ret == RT_RET_NOERROR);
