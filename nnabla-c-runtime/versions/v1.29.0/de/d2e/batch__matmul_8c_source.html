48 rt_function_error_t allocate_batch_matmul_local_context(rt_function_t *f) {
49  if (f->num_of_inputs != 2) {
50  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
51  }
52  if (f->num_of_outputs != 1) {
53  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
54  }
55 
56  batch_matmul_local_context_t *context =
57  (batch_matmul_local_context_t *)(f->local_context);
58  batch_matmul_private_t *p = rt_malloc_func(sizeof(batch_matmul_private_t));
59  if (p == 0) {
60  return RT_FUNCTION_ERROR_MALLOC;
61  }
62  ((batch_matmul_local_context_t *)(f->local_context))->data = (void *)p;
63  p->row_a = f->inputs[0]->shape.data[f->inputs[0]->shape.size - 2];
64  p->col_a = f->inputs[0]->shape.data[f->inputs[0]->shape.size - 1];
65  p->row_b = f->inputs[1]->shape.data[f->inputs[1]->shape.size - 2];
66  p->col_b = f->inputs[1]->shape.data[f->inputs[1]->shape.size - 1];
67  p->row_y = context->transpose_a ? p->col_a : p->row_a;
68  p->col_y = context->transpose_b ? p->row_b : p->col_b;
69  p->offset_a = p->row_a * p->col_a;
70  p->offset_b = p->row_b * p->col_b;
71  p->offset_y = p->row_y * p->col_y;
72 
73  p->input_a = f->inputs[0];
74  p->get_input_a = select_getter(p->input_a);
75  p->input_b = f->inputs[1];
76  p->get_input_b = select_getter(p->input_b);
77 
78  p->samples = 1;
79  for (int i = 0; i < f->inputs[0]->shape.size - 2; ++i) {
80  p->samples *= f->inputs[0]->shape.data[i];
81  }
82 
83  int samples_b = 1;
84  for (int j = 0; j < f->inputs[1]->shape.size - 2; ++j) {
85  samples_b *= f->inputs[1]->shape.data[j];
86  }
87 
88  p->output = f->outputs[0];
89  p->get_output = select_getter(p->output);
90  p->set_output = select_setter(p->output);
91  p->output_size = calc_shape_size(f->outputs[0]->shape);
92 
93  if (p->samples != samples_b) {
94  return RT_FUNCTION_ERROR_INVALID_SHAPE;
95  }
96  if (p->input_a->type == NN_DATA_TYPE_FLOAT &&
97  p->input_b->type == NN_DATA_TYPE_FLOAT &&
98  p->output->type == NN_DATA_TYPE_FLOAT) {
99 #ifdef CONFIG_BATCHMATMUL_FLOAT32
100  f->exec_func = exec_batch_matmul;
101 #endif /* CONFIG_BATCHMATMUL_FLOAT32 */
102  } else {
103 #ifdef CONFIG_BATCHMATMUL_GENERIC
104  f->exec_func = exec_batch_matmul_generic;
105 #endif /* CONFIG_BATCHMATMUL_GENERIC */
106  }
107  return RT_FUNCTION_ERROR_NOERROR;
108 }

115 static int get_next(int i, int m, int n) { return (i % n) * m + i / n; }

117 static int get_pre(int i, int m, int n) { return (i % m) * n + i / m; }

119 static void move_data(float *mtx, int i, int m, int n) {
120  float temp = mtx[i];
121  int cur = i;
122  int pre = get_pre(cur, m, n);
123  while (pre != i) {
124  mtx[cur] = mtx[pre];
125  cur = pre;
126  pre = get_pre(cur, m, n);
127  }
128  mtx[cur] = temp;
129 }

131 static void transpose(float *mtx, int m, int n) {
132  for (int i = 0; i < m * n; i++) {
133  int next = get_next(i, m, n);
134  while (next > i)
135  next = get_next(next, m, n);
136  if (next == i)
137  move_data(mtx, i, m, n);
138  }
139 }

142 rt_function_error_t exec_batch_matmul(rt_function_t *f) {
143  batch_matmul_local_context_t *context =
144  (batch_matmul_local_context_t *)(f->local_context);
145  batch_matmul_private_t *p = (batch_matmul_private_t *)(context->data);
146  float *input_a = (float *)(p->input_a->data);
147  float *input_b = (float *)(p->input_b->data);
148  float *output = (float *)(p->output->data);
149 
150  int i;
151  if (context->transpose_a) {
152  for (i = 0; i < p->samples; i++) {
153  transpose(input_a + p->offset_a * i, p->row_a, p->col_a);
154  }
155  p->row_a = p->row_a ^ p->col_a;
156  p->col_a = p->row_a ^ p->col_a;
157  p->row_a = p->row_a ^ p->col_a;
158  }
159  if (context->transpose_b) {
160  for (i = 0; i < p->samples; i++) {
161  transpose(input_b + p->offset_b * i, p->row_b, p->col_b);
162  }
163  p->row_b = p->row_b ^ p->col_b;
164  p->col_b = p->row_b ^ p->col_b;
165  p->row_b = p->row_b ^ p->col_b;
166  }
167  for (i = 0; i < p->samples; i++) {
168  float *mtx_y = output + p->offset_y * i;
169  float *mtx_a = input_a + p->offset_a * i;
170  float *mtx_b = input_b + p->offset_b * i;
171  int row_a = p->row_a;
172  int col_b = p->col_b;
173  int col_a = p->col_a;
174  for (int j = 0; j < row_a; j++) {
175  for (int k = 0; k < col_b; k++) {
176  for (int l = 0; l < col_a; l++) {
177  float a = *(mtx_a + col_a * j + l);
178  float b = *(mtx_b + col_b * l + k);
179  *(mtx_y + col_b * j + k) += a * b;
180  }
181  }
182  }
183  }
184 
185  return RT_FUNCTION_ERROR_NOERROR;
186 }

191  batch_matmul_local_context_t *context =

195  float *input_b = (float *)(p->input_b->data);
196 
197  int i;
198  if (context->transpose_a) {
199  for (i = 0; i < p->samples; i++) {
200  transpose(input_a + p->offset_a * i, p->row_a, p->col_a);
201  }
202  p->row_a = p->row_a ^ p->col_a;
203  p->col_a = p->row_a ^ p->col_a;
204  p->row_a = p->row_a ^ p->col_a;
205  }
206  if (context->transpose_b) {
207  for (i = 0; i < p->samples; i++) {
208  transpose(input_b + p->offset_b * i, p->row_b, p->col_b);
209  }
210  p->row_b = p->row_b ^ p->col_b;
211  p->col_b = p->row_b ^ p->col_b;
212  p->row_b = p->row_b ^ p->col_b;
213  }
214  for (i = 0; i < p->samples; i++) {
215  int row_a = p->row_a;
216  int col_b = p->col_b;
217  int col_a = p->col_a;
218  for (int j = 0; j < row_a; j++) {
219  for (int k = 0; k < col_b; k++) {
220  for (int l = 0; l < col_a; l++) {
221  float a = p->get_input_a(p->input_a, p->offset_y * i + col_a * j + l);
222  float b = p->get_input_b(p->input_b, p->offset_b * i + col_b * l + k);
223  float y = p->get_output(p->output, p->offset_y * i + col_b * j + k);
224  y += a * b;
225  p->set_output(p->output, p->offset_y * i + col_b * j + k, y);
226  }
227  }
228  }
229  }
230 
231  return RT_FUNCTION_ERROR_NOERROR;
232 }
