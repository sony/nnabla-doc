38 allocate_batch_normalization_local_context(rt_function_t *f) {
39  batch_normalization_local_context_t *context =
40  (batch_normalization_local_context_t *)(f->local_context);
41  batch_normalization_private_t *p =
42  rt_malloc_func(sizeof(batch_normalization_private_t));
43  if (p == 0) {
44  return RT_FUNCTION_ERROR_MALLOC;
45  }
46  // Check axes
47  if (context->axes.size != 1) {
48  rt_free_func(p);
49  return RT_FUNCTION_ERROR_INVALID_SHAPE;
50  }
51  // Check and parse shapes
52  rt_list_t input_shape = clone_list(f->inputs[0]->shape);
53  const int axis = context->axes.data[0];
54  const int size = calc_shape_size(input_shape);
55  const int size_axis = axis <= 0 ? size
56  : shape_product_of(f->inputs[0], axis,
57  f->inputs[0]->shape.size);
58  p->batch_size = size / size_axis;
59  p->specified_axis_size = input_shape.data[context->axes.data[0]];
60  p->output_size = size / p->batch_size / p->specified_axis_size;
61  p->multiplication_axis_output = p->specified_axis_size * p->output_size;
62  p->multiplication_batch_axis = p->batch_size * p->output_size;
63  if (p->batch_size * p->specified_axis_size * p->output_size != size) {
64  rt_free_func(p);
65  return RT_FUNCTION_ERROR_INVALID_SHAPE;
66  }
67 
68  p->batch_mean.shape = clone_list(f->inputs[1]->shape);
69  p->batch_var.shape = clone_list(f->inputs[2]->shape);
70  p->batch_mean.data =
71  rt_malloc_func(sizeof(float) * calc_shape_size(p->batch_mean.shape));
72  p->batch_var.data =
73  rt_malloc_func(sizeof(float) * calc_shape_size(p->batch_var.shape));
74  free_list(input_shape);
75  ((batch_normalization_local_context_t *)(f->local_context))->data = (void *)p;
76 
77 #ifdef CONFIG_BATCHNORMALIZATION_FLOAT32
78  f->exec_func = exec_batch_normalization;
79 #endif /* CONFIG_BATCHNORMALIZATION_FLOAT32 */
80 #ifdef CONFIG_BATCHNORMALIZATION_GENERIC
81  for (int i = 0; i < f->num_of_inputs; i++) {
82  if (f->inputs[i]->type != NN_DATA_TYPE_FLOAT) {

88  f->exec_func = exec_batch_normalization_generic;
89  }

91  return RT_FUNCTION_ERROR_NOERROR;
92 }

108 static void forward_impl_batch(rt_function_t *f,
109  batch_normalization_local_context_t *context,
110  batch_normalization_private_t *p) {
111  const float *x = (float *)(f->inputs[0]->data);
112  const float *beta = (float *)(f->inputs[1]->data);
113  const float *gamma = (float *)(f->inputs[2]->data);
114  float *y = (float *)(f->outputs[0]->data);
115  float *m = (float *)p->batch_mean.data; // batch mean
116  float *v = (float *)p->batch_var.data; // batch var
117  float *rm = (float *)(f->inputs[3]->data); // running mean
118  float *rv = (float *)(f->inputs[4]->data); // running var
119  const int specified_axis_size = p->specified_axis_size;
120  const int output_size = p->output_size;
121  const int multiplication_axis_output = p->multiplication_axis_output;
122  const int multiplication_batch_axis = p->multiplication_batch_axis;
123 
124  if (f->num_of_outputs == 3) {
125  m = (float *)(f->outputs[1]->data);
126  v = (float *)(f->outputs[2]->data);
127  }
128 
129  int i1;
130  for (i1 = 0; i1 < specified_axis_size; i1++) {
131  m[i1] = 0;
132  v[i1] = 0;
133  int i02;
134  for (i02 = 0; i02 < multiplication_batch_axis; i02++) {
135  const int i0 = i02 / output_size;
136  const int i2 = i02 % output_size;
137  const int i = i0 * multiplication_axis_output + i1 * output_size + i2;
138  const float value = x[i];
139  m[i1] += value;
140  v[i1] += value * value;
141  }
142  m[i1] /= multiplication_batch_axis;
143  v[i1] = v[i1] / multiplication_batch_axis - m[i1] * m[i1];
144 
145  // Moving mean and var
146  rm[i1] = context->decay_rate * rm[i1] + (1 - context->decay_rate) * m[i1];
147  rv[i1] = context->decay_rate * rv[i1] + (1 - context->decay_rate) * v[i1] *
148  multiplication_batch_axis /
149  (multiplication_batch_axis - 1);
150 
151  const float stdvar = sqrtf(v[i1] + context->eps);
152  // Subtract mean and divide by std, and apply beta and gamma.
153  for (i02 = 0; i02 < multiplication_batch_axis; i02++) {
154  const int i0 = i02 / output_size;
155  const int i2 = i02 % output_size;
156  const int i = i0 * multiplication_axis_output + i1 * output_size + i2;
157  y[i] = (x[i] - m[i1]) * gamma[i1] / stdvar + beta[i1];
158  }
159  }
160 }

208 forward_impl_batch_generic(rt_function_t *f,
209  batch_normalization_local_context_t *context,
210  batch_normalization_private_t *p) {
211  rt_variable_t *input_x = f->inputs[0];
212  rt_variable_getter get_x = select_getter(input_x);
213  rt_variable_t *input_beta = f->inputs[1];
214  rt_variable_getter get_beta = select_getter(input_beta);
215  rt_variable_t *input_gamma = f->inputs[2];
216  rt_variable_getter get_gamma = select_getter(input_gamma);
217  rt_variable_t *input_rm = f->inputs[3]; // running mean
218  rt_variable_getter get_rm = select_getter(input_rm);
219  rt_variable_setter set_rm = select_setter(input_rm);
220  rt_variable_t *input_rv = f->inputs[4]; // running var
221  rt_variable_getter get_rv = select_getter(input_rv);
222  rt_variable_setter set_rv = select_setter(input_rv);
223  rt_variable_t *output = f->outputs[0];
224  rt_variable_setter set_output = select_setter(output);
225  float *m = (float *)p->batch_mean.data; // batch mean
226  float *v = (float *)p->batch_var.data; // batch var
227  const int specified_axis_size = p->specified_axis_size;
228  const int output_size = p->output_size;
229  const int multiplication_axis_output = p->multiplication_axis_output;
230  const int multiplication_batch_axis = p->multiplication_batch_axis;
231 
232  if (f->num_of_outputs == 3) {
233  m = (float *)(f->outputs[1]->data);
234  v = (float *)(f->outputs[2]->data);
235  }
236 
237  int i1;
238  for (i1 = 0; i1 < specified_axis_size; i1++) {
239  m[i1] = 0;
240  v[i1] = 0;
241  int i02;
242  for (i02 = 0; i02 < multiplication_batch_axis; i02++) {
243  const int i0 = i02 / output_size;
244  const int i2 = i02 % output_size;
245  const int i = i0 * multiplication_axis_output + i1 * output_size + i2;
246  const float value = get_x(input_x, i);
247  m[i1] += value;
248  v[i1] += value * value;
249  }
250  m[i1] /= multiplication_batch_axis;
251  v[i1] = v[i1] / multiplication_batch_axis - m[i1] * m[i1];
252 
253  // Moving mean and var
254  float rm = get_rm(input_rm, i1);
255  float rv = get_rv(input_rv, i1);
256  rm = context->decay_rate * rm + (1 - context->decay_rate) * m[i1];
257  rv = context->decay_rate * rv + (1 - context->decay_rate) * v[i1] *
258  multiplication_batch_axis /
259  (multiplication_batch_axis - 1);
260  set_rm(input_rm, i1, rm);
261  set_rv(input_rv, i1, rv);
262 
263  const float stdvar = sqrtf(v[i1] + context->eps);
264  // Subtract mean and divide by std, and apply beta and gamma.
265  for (i02 = 0; i02 < multiplication_batch_axis; i02++) {
266  const int i0 = i02 / output_size;
267  const int i2 = i02 % output_size;
268  const int i = i0 * multiplication_axis_output + i1 * output_size + i2;
269  float x = get_x(input_x, i);
270  float beta = get_beta(input_beta, i1);
271  float gamma = get_gamma(input_gamma, i1);
272  float y = (x - m[i1]) * gamma / stdvar + beta;
273  set_output(output, i, y);
274  }
275  }
