27 rt_function_error_t allocate_mean_subtraction_local_context(rt_function_t *f) {
28  mean_subtraction_local_context_t *ctx =
29  (mean_subtraction_local_context_t *)f->local_context;
30  rt_list_t in_shape = f->inputs[0]->shape;
31  int s0 = 1, i;
32  for (i = ctx->base_axis; i < in_shape.size; ++i) {
33  s0 *= in_shape.data[i];
34  }
35  if (calc_shape_size(f->inputs[1]->shape) != s0) {
36  return RT_FUNCTION_ERROR_INVALID_SHAPE;

38  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&

61 rt_function_error_t exec_mean_subtraction(rt_function_t *f) {
62  mean_subtraction_local_context_t *ctx =
63  (mean_subtraction_local_context_t *)f->local_context;
64  rt_list_t in_shape = f->inputs[0]->shape;
65  float *y = (float *)(f->outputs[0]->data);
66  float *x = (float *)(f->inputs[0]->data);
67  float *rm = (float *)(f->inputs[1]->data);
68  int i, j, s0 = 1, s1;
69 
70  for (i = ctx->base_axis; i < in_shape.size; ++i) {
71  s0 *= in_shape.data[i];
72  }
73  s1 = calc_shape_size(f->inputs[0]->shape) / s0;
74  for (i = 0; i < s0; ++i) {
75  for (j = 0; j < s1; ++j) {
76  int idx = j * s0 + i;
77  y[idx] = x[idx] - rm[i];
78  }
79  }
80 
81  return RT_FUNCTION_ERROR_NOERROR;
82 }

92  rt_variable_t *input_rm = f->inputs[1];
93  rt_variable_getter get_rm = select_getter(input_rm);
94  rt_variable_t *output = f->outputs[0];
95  rt_variable_setter set_output = select_setter(output);
96  int i, j, s0 = 1, s1;
97 
98  for (i = ctx->base_axis; i < in_shape.size; ++i) {
99  s0 *= in_shape.data[i];
100  }
101  s1 = calc_shape_size(f->inputs[0]->shape) / s0;
102  for (i = 0; i < s0; ++i) {
103  for (j = 0; j < s1; ++j) {
104  int idx = j * s0 + i;
105  float x = get_x(input_x, idx);
106  float rm = get_rm(input_rm, i);
107  set_output(output, idx, x - rm);
108  }
109  }
110 
111  return RT_FUNCTION_ERROR_NOERROR;
112 }
