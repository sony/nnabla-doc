38 rt_function_error_t allocate_unpooling_local_context(rt_function_t *f) {
39  unpooling_local_context_t *context =
40  (unpooling_local_context_t *)(f->local_context);
41  unpooling_private_t *p = rt_malloc_func(sizeof(unpooling_private_t));
42  if (p == 0) {
43  return RT_FUNCTION_ERROR_MALLOC;
44  }
45  p->input_shape = clone_list(f->inputs[0]->shape);
46  p->output_shape = clone_list(f->outputs[0]->shape);
47  p->input_strides = calc_contiguous_strides(f->inputs[0]->shape);
48  p->output_strides = calc_contiguous_strides(f->outputs[0]->shape);
49  p->input = f->inputs[0];
50  p->get_input = select_getter(p->input);
51  p->output = f->outputs[0];
52  p->set_output = select_setter(p->output);
53 
54  if (context->kernel.size > p->input_shape.size) {
55  free_list(p->input_shape);
56  free_list(p->output_shape);
57  free_list(p->input_strides);
58  free_list(p->output_strides);
59  rt_free_func(p);
60  return RT_FUNCTION_ERROR_INVALID_SHAPE;
61  }
62 
63  // Calc and set output shape.
64  rt_list_t shape = allocate_list(p->input_shape.size);
65  int diff = p->input_shape.size - context->kernel.size;
66  int i;
67  for (i = 0; i < diff; i++) {
68  shape.data[i] = 1;
69  }
70  for (i = diff; i < p->input_shape.size; i++) {
71  shape.data[i] = context->kernel.data[i - diff];
72  }
73  p->kernel = clone_list(shape);
74  for (i = 0; i < p->input_shape.size; i++) {
75  p->output_shape.data[i] = p->input_shape.data[i] * p->kernel.data[i];
76  }
77  free_list(shape);
78 
79  ((unpooling_local_context_t *)(f->local_context))->data = (void *)p;
80 
81  f->exec_func = exec_unpooling;
82 
83  return RT_FUNCTION_ERROR_NOERROR;
84 }

100 static void unpooling_forward_recursive_generic(unpooling_private_t *p,
101  int x_offset, int y_offset,
102  int dim) {
103  int current_x_offset = x_offset;
104  int current_y_offset = y_offset;
105  const int x_stride = p->input_strides.data[dim];
106  const int y_stride = p->output_strides.data[dim];
107  const int kernel = p->kernel.data[dim];
108  const int size = p->output_shape.data[dim];
109 
110  if (dim == p->input_shape.size - 1) {
111  if (x_stride == 1 && kernel == 1) {
112  for (int i = 0; i < size; i++) {
113  float value = p->get_input(p->input, i + current_x_offset);
114  p->set_output(p->output, i + current_y_offset, value);
115  }
116  } else {
117  int count = 0;
118  for (int i = 0; i < size; i++) {
119  float value = p->get_input(p->input, current_x_offset);
120  p->set_output(p->output, current_y_offset, value);
121  if (++count >= kernel) {
122  count = 0;
123  current_x_offset += x_stride;
124  }
125  current_y_offset += y_stride;
126  }
127  }
128  } else {
129  int count = 0;
130  int i;
131  for (i = 0; i < size; i++) {
132  unpooling_forward_recursive_generic(p, current_x_offset, current_y_offset,
133  dim + 1);
134  if (++count >= kernel) {
135  count = 0;
136  current_x_offset += x_stride;
137  }
138  current_y_offset += y_stride;
139  }
140  }
141 }

144 #ifdef CONFIG_UNPOOLING_FLOAT32
145 static void unpooling_forward_recursive(unpooling_private_t *p, int x_offset,
146  int y_offset, int dim) {
147  int current_x_offset = x_offset;
148  int current_y_offset = y_offset;
149  const int x_stride = p->input_strides.data[dim];
150  const int y_stride = p->output_strides.data[dim];
151  const int kernel = p->kernel.data[dim];
152  const int size = p->output_shape.data[dim];
153  const float *x = (float *)(p->input->data);
154  float *y = (float *)(p->output->data);
155 
156  if (dim == p->input_shape.size - 1) {
157  const float *current_x = x + current_x_offset;
158  float *current_y = y + current_y_offset;
159  if (x_stride == 1 && kernel == 1) {
160  memcpy(current_y, current_x, sizeof(float) * size);
161  } else {
162  const float *end_y = current_y + size * y_stride;
163  int count = 0;
164  while (current_y != end_y) {
165  *current_y = *current_x;
166  if (++count >= kernel) {
167  count = 0;
168  current_x += x_stride;
169  }
170  current_y += y_stride;
171  }
172  }
173  } else {
174  int count = 0;
175  int i;
176  for (i = 0; i < size; i++) {
177  unpooling_forward_recursive(p, current_x_offset, current_y_offset,
178  dim + 1);
179  if (++count >= kernel) {
180  count = 0;
181  current_x_offset += x_stride;
182  }
183  current_y_offset += y_stride;
184  }
185  }
186 }
187 #endif /* CONFIG_UNPOOLING_FLOAT32 */

189 rt_function_error_t exec_unpooling(rt_function_t *f) {
190  unpooling_private_t *p =
191  (unpooling_private_t *)(((unpooling_local_context_t *)(f->local_context))
192  ->data);
193  if (p->input->type == NN_DATA_TYPE_FLOAT &&
194  p->output->type == NN_DATA_TYPE_FLOAT) {
195 #ifdef CONFIG_UNPOOLING_FLOAT32
196  unpooling_forward_recursive(p, 0, 0, 0);
197 #endif /* CONFIG_UNPOOLING_FLOAT32 */
198  } else {
199 #ifdef CONFIG_UNPOOLING_GENERIC
200  unpooling_forward_recursive_generic(p, 0, 0, 0);
201 #endif /* CONFIG_UNPOOLING_GENERIC */
202  }
203  return RT_FUNCTION_ERROR_NOERROR;
204 }
