20 // Common algorithm for arithmetic calculation between two vectors has same
21 // dimension.
22 void calc_dim_arithmetic_fixed8(rt_variable_t *output, rt_variable_t *input1,
23  rt_variable_t *input2, int dim_index, int y_out,
24  int y_in1, int y_in2,
25  int8_t (*calc_func)(int8_t, int8_t)) {
26  int dim_size = output->shape.size;
27  int x_size_out = output->shape.data[dim_index];
28  int x_size_in1 = input1->shape.data[dim_index];
29  int x_size_in2 = input2->shape.data[dim_index];
30  for (int x_out = 0; x_out < x_size_out; x_out++) {
31  int x_in1 = (x_out % x_size_in1);
32  int x_in2 = (x_out % x_size_in2);
33  int z_out = y_out * x_size_out + x_out;
34  int z_in1 = y_in1 * x_size_in1 + x_in1;
35  int z_in2 = y_in2 * x_size_in2 + x_in2;
36  if (dim_index + 1 < dim_size) {
37  calc_dim_arithmetic_fixed8(output, input1, input2, dim_index + 1, z_out,
38  z_in1, z_in2, calc_func);
39  } else {
40  int8_t *data_out = output->data;
41  int8_t *data_in1 = input1->data;
42  int8_t *data_in2 = input2->data;
43  data_out[z_out] = calc_func(data_in1[z_in1], data_in2[z_in2]);
44  }
45  }
46 }

136 void equalize_bits_fixed8(rt_variable_t *v1, rt_variable_t *v2) {
137  // Rescale two arrays to have the same number of range bits.
138  // Downscale the array having most precision bits.
139 
140  if (v1->fp_pos == v2->fp_pos) {
141  return;
142  }
143 
144  const rt_variable_t *to_keep = v1->fp_pos < v2->fp_pos ? v1 : v2;
145  rt_variable_t *to_rescale = to_keep == v1 ? v2 : v1;
146  const unsigned rescale_size = calc_shape_size(to_rescale->shape);
147 
148  rescale_fixed8(to_rescale->data, rescale_size, to_rescale->fp_pos,
149  to_keep->fp_pos);
150 
151  // Update fp_pos for rescaled variable
152  to_rescale->fp_pos = to_keep->fp_pos;
153 
154  return;
155 }
