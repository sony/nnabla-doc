29 rt_function_error_t allocate_add_scalar_local_context(rt_function_t *f) {
30  if (f->num_of_inputs != 1) {
31  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
32  }
33  if (f->num_of_outputs != 1) {
34  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
35  }
36  int input_size = calc_shape_size(f->inputs[0]->shape);
37  int output_size = calc_shape_size(f->outputs[0]->shape);
38  if (input_size != output_size) {
39  return RT_FUNCTION_ERROR_INVALID_SHAPE;
40  }
41  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&
42  f->outputs[0]->type == NN_DATA_TYPE_FLOAT) {
43 #ifdef CONFIG_ADDSCALAR_FLOAT32
44  f->exec_func = exec_add_scalar;
45 #endif /* CONFIG_ADDSCALAR_FLOAT32 */
46  } else if (f->inputs[0]->type == NN_DATA_TYPE_INT16 &&
47  f->outputs[0]->type == NN_DATA_TYPE_INT16) {
48 #ifdef CONFIG_ADDSCALAR_FIXED16
49  f->exec_func = exec_add_scalar_fixed16;
50 #endif /* CONFIG_ADDSCALAR_FIXED16 */
51  } else if (f->inputs[0]->type == NN_DATA_TYPE_INT8 &&
52  f->outputs[0]->type == NN_DATA_TYPE_INT8) {
53 #ifdef CONFIG_ADDSCALAR_FIXED8
54  f->exec_func = exec_add_scalar_fixed8;
55 #endif /* CONFIG_ADDSCALAR_FIXED8 */
56  } else {
57 #ifdef CONFIG_ADDSCALAR_GENERIC
58  f->exec_func = exec_add_scalar_generic;
59 #endif /* CONFIG_ADDSCALAR_GENERIC */
60  }
61  return RT_FUNCTION_ERROR_NOERROR;
62 }
