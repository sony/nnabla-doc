62 rt_function_error_t exec_mean_subtraction(rt_function_t *f) {
63  mean_subtraction_local_context_t *ctx =
64  (mean_subtraction_local_context_t *)f->local_context;
65  rt_list_t in_shape = f->inputs[0]->shape;
66  float *y = (float *)(f->outputs[0]->data);
67  float *x = (float *)(f->inputs[0]->data);
68  float *rm = (float *)(f->inputs[1]->data);
69  int i, j, s0 = 1, s1;
70 
71  for (i = ctx->base_axis; i < in_shape.size; ++i) {
72  s0 *= in_shape.data[i];
73  }
74  s1 = calc_shape_size(f->inputs[0]->shape) / s0;
75  for (i = 0; i < s0; ++i) {
76  for (j = 0; j < s1; ++j) {
77  int idx = j * s0 + i;
78  y[idx] = x[idx] - rm[i];
79  }
80  }
81 
82  return RT_FUNCTION_ERROR_NOERROR;
83 }

98 
99  for (i = ctx->base_axis; i < in_shape.size; ++i) {
100  s0 *= in_shape.data[i];

102  s1 = calc_shape_size(f->inputs[0]->shape) / s0;
103  for (i = 0; i < s0; ++i) {
104  for (j = 0; j < s1; ++j) {
105  int idx = j * s0 + i;

107  float rm = get_rm(input_rm, i);
108  set_output(output, idx, x - rm);
109  }
110  }
