23 rt_function_error_t allocate_pooling(rt_function_t *f,
24  pooling_context_t *context,
25  pooling_private_t *p) {
26  if (p == 0) {
27  return RT_FUNCTION_ERROR_MALLOC;
28  }
29  p->input_shape = clone_list(f->inputs[0]->shape);
30  p->output_shape = clone_list(f->outputs[0]->shape);
31  p->input_n_kernel_size_diff = p->input_shape.size - context->kernel.size;
32  if (context->stride.size == 0) {
33  context->stride = clone_list(context->kernel);
34  } else {
35  if (context->kernel.size != context->stride.size) {
36  return RT_FUNCTION_ERROR_INVALID_SHAPE;
37  }
38  if (context->kernel.size > p->input_shape.size) {
39  return RT_FUNCTION_ERROR_INVALID_SHAPE;
40  }
41  }
42  if (context->kernel.size != 2 && context->kernel.size != 3) {
43  return RT_FUNCTION_ERROR_INVALID_SHAPE;
44  }
45  if (context->kernel.size != context->pad.size) {
46  return RT_FUNCTION_ERROR_INVALID_SHAPE;
47  }
48 
49  // Calc and set output shape.
50  rt_list_t shape = allocate_list(context->kernel.size);
51  int i;
52  for (i = 0; i < shape.size; i++) {
53  int _w = p->input_shape.data[i + p->input_n_kernel_size_diff];
54  int _k = context->kernel.data[i];
55  int _s = context->stride.data[i];
56  int _p = context->pad.data[i];
57  shape.data[i] = (_w + _p - (context->ignore_border ? _k - _p : 1)) / _s + 1;
58  }
59  for (i = 0; i < p->input_shape.size; i++) {
60  if (i < p->input_n_kernel_size_diff) {
61  p->output_shape.data[i] = p->input_shape.data[i];
62  } else {
63  p->output_shape.data[i] = shape.data[i - p->input_n_kernel_size_diff];
64  }
65  }
66  free_list(shape);
67 
68  // Calc x_map_size and y_map_size.
69  p->input_strides = calc_contiguous_strides(f->inputs[0]->shape);
70  p->output_strides = calc_contiguous_strides(f->outputs[0]->shape);
71  p->x_map_size = (p->input_n_kernel_size_diff == 0)
72  ? calc_shape_size(f->inputs[0]->shape)
73  : p->input_strides.data[p->input_n_kernel_size_diff - 1];
74  p->y_map_size = (p->input_n_kernel_size_diff == 0)
75  ? calc_shape_size(f->outputs[0]->shape)
76  : p->output_strides.data[p->input_n_kernel_size_diff - 1];
77 
78  // Init calc_context.
79  p->calc_context.hstart = 0;
80  p->calc_context.hend = 0;
81  p->calc_context.wstart = 0;
82  p->calc_context.wend = 0;
83  p->calc_context.dstart = 0;
84  p->calc_context.dend = 0;
85  p->calc_context.pool_size = 0;
86  p->calc_context.x = f->inputs[0];
87  p->calc_context.get_x = select_getter(p->calc_context.x);
88  p->calc_context.y = f->outputs[0];
89  p->calc_context.set_y = select_setter(p->calc_context.y);
90  p->calc_context.including_pad = context->including_pad;
91 
92  return RT_FUNCTION_ERROR_NOERROR;
93 }

99  free_list(p->output_strides);

104 rt_function_error_t exec_pooling(rt_function_t *f, pooling_context_t *context,
105  pooling_private_t *p,
106  exec_pooling_func_t exec) {
107  float *y = (float *)(p->calc_context.y->data);
108  const int hx = p->input_shape.data[p->input_n_kernel_size_diff + 0];
109  const int wx = p->input_shape.data[p->input_n_kernel_size_diff + 1];
110  const int hy = p->output_shape.data[p->input_n_kernel_size_diff + 0];
111  const int wy = p->output_shape.data[p->input_n_kernel_size_diff + 1];
112  const int hkernel = context->kernel.data[0];
113  const int wkernel = context->kernel.data[1];
114  const int hstride = context->stride.data[0];
115  const int wstride = context->stride.data[1];
116  const int hpad = context->pad.data[0];
117  const int wpad = context->pad.data[1];
118  const int n_map = calc_shape_size(f->inputs[0]->shape) / p->x_map_size;
119  p->calc_context.offset_x = 0;
120  p->calc_context.offset_y = 0;
121  p->calc_context.kernel_size = context->kernel.size;
122 
123  if (context->kernel.size == 2) {
124  for (int n = 0; n < n_map; n++) {
125  for (int iy = 0; iy < hy; iy++) {
126  for (int jy = 0; jy < wy; jy++) {
127  int hstart = iy * hstride - hpad;
128  int wstart = jy * wstride - wpad;
129  int hend = (int)fminf((float)(hstart + hkernel), (float)(hx + hpad));
130  int wend = (int)fminf((float)(wstart + wkernel), (float)(wx + wpad));
131  p->calc_context.pool_size = (hend - hstart) * (wend - wstart);
132  p->calc_context.hstart = (int)fmaxf((float)hstart, 0);
133  p->calc_context.wstart = (int)fmaxf((float)wstart, 0);
134  p->calc_context.hend = (int)fminf((float)hend, (float)hx);
135  p->calc_context.wend = (int)fminf((float)wend, (float)wx);
136  p->calc_context.hstride =
137  p->input_strides.data[p->input_n_kernel_size_diff + 0];
138  int k =
139  iy * p->output_strides.data[p->input_n_kernel_size_diff + 0] + jy;
140  float val = exec(p->calc_context);
141  *(y + k + p->calc_context.offset_y) = val;
142  }
143  }
144  p->calc_context.offset_x += p->x_map_size;
145  p->calc_context.offset_y += p->y_map_size;
146  }
147  } else if (context->kernel.size == 3) {
148  const int dx = p->input_shape.data[p->input_n_kernel_size_diff + 2];
149  const int dy = p->output_shape.data[p->input_n_kernel_size_diff + 2];
150  const int dkernel = context->kernel.data[2];
151  const int dstride = context->stride.data[2];
152  const int dpad = context->pad.data[2];
153 
154  for (int n = 0; n < n_map; n++) {
155  for (int iy = 0; iy < hy; iy++) {
156  for (int jy = 0; jy < wy; jy++) {
157  for (int ky = 0; ky < dy; ky++) {
158  int hstart = iy * hstride - hpad;
159  int wstart = jy * wstride - wpad;
160  int dstart = ky * dstride - dpad;
161  int hend =
162  (int)fminf((float)(hstart + hkernel), (float)(hx + hpad));
163  int wend =
164  (int)fminf((float)(wstart + wkernel), (float)(wx + wpad));
165  int dend =
166  (int)fminf((float)(dstart + dkernel), (float)(dx + dpad));
167  p->calc_context.pool_size =
168  (hend - hstart) * (wend - wstart) * (dend - dstart);
169  p->calc_context.hstart = (int)fmaxf((float)hstart, 0);
170  p->calc_context.wstart = (int)fmaxf((float)wstart, 0);
171  p->calc_context.dstart = (int)fmaxf((float)dstart, 0);
172  p->calc_context.hend = (int)fminf((float)hend, (float)hx);
173  p->calc_context.wend = (int)fminf((float)wend, (float)wx);
174  p->calc_context.dend = (int)fminf((float)dend, (float)dx);
175  p->calc_context.hstride =
176  p->input_strides.data[p->input_n_kernel_size_diff + 0];
177  p->calc_context.wstride =
178  p->input_strides.data[p->input_n_kernel_size_diff + 1];
179  int k =
180  iy * p->output_strides.data[p->input_n_kernel_size_diff + 0] +
181  jy * p->output_strides.data[p->input_n_kernel_size_diff + 1] +
182  ky;
183  float val = exec(p->calc_context);
184  *(y + k + p->calc_context.offset_y) = val;
185  }
186  }
187  }
188  p->calc_context.offset_x += p->x_map_size;
189  p->calc_context.offset_y += p->y_map_size;
190  }
191  }
192  return RT_FUNCTION_ERROR_NOERROR;
193 }
