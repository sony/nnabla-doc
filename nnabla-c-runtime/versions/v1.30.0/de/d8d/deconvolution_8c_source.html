52 rt_function_error_t allocate_deconvolution_local_context(rt_function_t *f) {
53  deconvolution_local_context_t *c =
54  (deconvolution_local_context_t *)(f->local_context);
55 
56  if (f->num_of_inputs != 2 && f->num_of_inputs != 3) {
57  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
58  }
59 
60  if (f->num_of_outputs != 1) {
61  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
62  }
63 
64  deconvolution_private_t *p = rt_malloc_func(sizeof(deconvolution_private_t));
65  if (p == 0) {
66  return RT_FUNCTION_ERROR_MALLOC;
67  }
68  c->data = (void *)p;
69  p->input = f->inputs[0];
70  p->get_input = select_getter(p->input);
71 
72  p->weight = f->inputs[1];
73  p->get_weight = select_getter(p->weight);
74 
75  p->output = f->outputs[0];
76  p->get_output = select_getter(p->output);
77  p->set_output = select_setter(p->output);
78 
79  if (f->num_of_inputs > 2) {
80  p->bias = f->inputs[2];
81  p->get_bias = select_getter(p->bias);
82  } else {
83  p->bias = 0;
84  }
85 
86  p->base_loop_size = 1;
87  for (int i = 0; i < c->base_axis; i++) {
88  p->base_loop_size *= p->input->shape.data[i];
89  }
90  p->spatial_dims = p->input->shape.size - c->base_axis - 1;
91 
92  p->input_shape = allocate_list(p->spatial_dims);
93  p->kernel_shape = allocate_list(p->spatial_dims);
94  p->output_shape = allocate_list(p->spatial_dims);
95  p->in_position = allocate_list(p->spatial_dims);
96  p->out_position = allocate_list(p->spatial_dims);
97 
98  for (int i = 0; i < p->spatial_dims; i++) {
99  p->kernel_shape.data[i] = p->weight->shape.data[i + 2];
100  p->input_shape.data[i] = p->input->shape.data[i + c->base_axis + 1];
101  p->output_shape.data[i] = p->output->shape.data[i + c->base_axis + 1];
102  }
103 
104  if (p->input->type == NN_DATA_TYPE_FLOAT &&
105  p->output->type == NN_DATA_TYPE_FLOAT &&
106  p->weight->type == NN_DATA_TYPE_FLOAT &&
107  ((p->bias && p->bias->type == NN_DATA_TYPE_FLOAT) || !p->bias)) {
108 #ifdef CONFIG_DECONVOLUTION_FLOAT32
109  f->exec_func = exec_deconvolution;
110 #endif /* CONFIG_DECONVOLUTION_FLOAT32 */
111  } else {
112 #ifdef CONFIG_DECONVOLUTION_GENERIC
113  f->exec_func = exec_deconvolution_generic;
114 #endif /* CONFIG_DECONVOLUTION_GENERIC */
115  }
116  return RT_FUNCTION_ERROR_NOERROR;
117 }

133 rt_function_error_t exec_deconvolution(rt_function_t *f) {
134  deconvolution_local_context_t *c =
135  (deconvolution_local_context_t *)(f->local_context);
136  deconvolution_private_t *p = (deconvolution_private_t *)(c->data);
137 
138  int output_size = calc_shape_size(p->output_shape);
139  int kernel_size = calc_shape_size(p->kernel_shape);
140  int input_size = calc_shape_size(p->input_shape);
141 
142  memset(p->output->data, 0, sizeof(float) * calc_shape_size(p->output->shape));
143 
144  for (int b = 0; b < p->base_loop_size; b++) {
145  for (int g = 0; g < c->group; g++) {
146  for (int om = 0; om < p->weight->shape.data[1]; om++) {
147  int output_offset = (b * c->group * p->weight->shape.data[1] +
148  g * p->weight->shape.data[1] + om) *
149  output_size;
150  for (int im = 0; im < p->weight->shape.data[0] / c->group; im++) {
151  int input_offset = (b * p->weight->shape.data[0] +
152  g * p->weight->shape.data[0] / c->group + im) *
153  input_size;
154  int kernel_offset = (g * p->weight->shape.data[1] *
155  p->weight->shape.data[0] / c->group +
156  im * p->weight->shape.data[1] + om) *
157  kernel_size;
158  for (int o = 0; o < output_size; o++) {
159  float sum = 0.0f;
160  pos_to_shape(p->out_position, p->output_shape, o);
161  for (int k = 0; k < kernel_size; k++) {
162  pos_to_shape(p->in_position, p->kernel_shape, k);
163  uint8_t condition = 1;
164  for (int j = 0; j < p->spatial_dims; j++) {
165  int k1 =
166  c->dilation.data[j] * (p->kernel_shape.data[j] - 1) + 1;
167  p->in_position.data[j] *= c->dilation.data[j];
168  p->in_position.data[j] -= k1 - c->pad.data[j] - 1;
169  p->in_position.data[j] += p->out_position.data[j];
170  if (p->in_position.data[j] % c->stride.data[j] != 0) {
171  condition = 0;
172  break;
173  }
174  p->in_position.data[j] =
175  p->in_position.data[j] / c->stride.data[j];
176  if (p->in_position.data[j] < 0 ||
177  p->in_position.data[j] >= p->input_shape.data[j]) {
178  condition = 0;
179  break;
180  }
181  }
182  if (condition) {
183  float x = *((float *)(p->input->data) + input_offset +
184  shape_to_pos(p->input_shape, p->in_position));
185  float w = *((float *)(p->weight->data) + kernel_offset +
186  (kernel_size - k - 1));
187  sum += x * w;
188  }
189  }
190  *((float *)(p->output->data) + output_offset + o) += sum;
191  }
192  }
193  if (p->bias) {
194  for (int o = 0; o < output_size; o++) {
195  *((float *)(p->output->data) + output_offset + o) +=
196  *((float *)(p->bias->data) + g * p->weight->shape.data[1] + om);
197  }
198  }
199  }
200  }
201  }
202 
203  return RT_FUNCTION_ERROR_NOERROR;
204 }
205 #endif /* CONFIG_DECONVOLUTION_FLOAT32 */

259  p->input, input_offset +
260  shape_to_pos(p->input_shape, p->in_position));
261  float w = p->get_weight(p->weight,

274  float bias =
275  p->get_bias(p->bias, g * p->weight->shape.data[1] + om);
276  y += bias;
