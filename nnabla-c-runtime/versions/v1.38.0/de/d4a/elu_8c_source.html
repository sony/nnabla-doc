44 #ifdef CONFIG_ELU_FLOAT32
45 rt_function_error_t exec_elu(rt_function_t *f) {
46  elu_local_context_t *context = (elu_local_context_t *)(f->local_context);
47  const float *x = (float *)(f->inputs[0]->data);
48  float *y = (float *)(f->outputs[0]->data);
49  const int size = calc_shape_size(f->inputs[0]->shape);
50  int s;
51  for (s = 0; s < size; s++) {
52  y[s] = (float)(x[s] > (float)0 ? x[s]
53  : context->alpha * (expf(x[s]) - (float)1));
54  }
55  return RT_FUNCTION_ERROR_NOERROR;
56 }
57 #endif /* CONFIG_ELU_FLOAT32 */

59 #ifdef CONFIG_ELU_GENERIC
60 rt_function_error_t exec_elu_generic(rt_function_t *f) {
61  elu_local_context_t *context = (elu_local_context_t *)(f->local_context);
62  rt_variable_t *input = f->inputs[0];
63  rt_variable_getter get_input = select_getter(input);
64  rt_variable_t *output = f->outputs[0];
65  rt_variable_setter set_output = select_setter(output);
66  const int size = calc_shape_size(f->inputs[0]->shape);
67  int s;
68  for (s = 0; s < size; s++) {
69  float val_x = get_input(input, s);
70  float val_y =
71  (float)(val_x > (float)0 ? val_x
72  : context->alpha * (expf(val_x) - (float)1));
73  set_output(output, s, val_y);
74  }
75  return RT_FUNCTION_ERROR_NOERROR;
76 }
