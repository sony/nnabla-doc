30 /// @brief NNB format binary data from Affine_000.nnp
31 ///
32 /// By default just call @ref rt_add_callback to use call back function.
33 ///
34 /// But here is some modification to use callback more flexible.
35 ///
36 /// To generate the 'Affine_000.nnp' file, you can use the following Python
37 /// code:
38 /// @code{.py}
39 /// import nnabla as nn
40 /// import nnabla.parametric_functions as PF
41 /// from nnabla.utils.save import save
42 ///
43 /// x = nn.Variable((1, 4, 6), need_grad=True)
44 /// y0 = PF.affine(inp=x, n_outmaps=(2, 3), name='network1', base_axis=1)
45 /// contents = {
46 /// 'networks': [
47 /// {'name': 'network1',
48 /// 'batch_size': 1,
49 /// 'outputs': {'y0': y0},
50 /// 'names': {'x': x}}],
51 /// 'executors': [
52 /// {'name': 'inference',
53 /// 'network': 'network1',
54 /// 'data': ['x'],
55 /// 'output': ['y0']}]}
56 
57 /// save("Affine_000.nnp", contents=contents)
58 /// @endcode
59 /// Extracts the network.nntxt from it by unzipping 'Affine_000.nnp'.
60 /// @code{.sh}
61 /// $ unzip Affine_000.nnp
62 /// @endcode
63 /// Then you should have 'network.nntxt'.
64 ///
65 /// To use callback, you must set 'function implement flag' in NNB file.
66 /// You can get template setting file with following command.
67 /// @code{.sh}
68 /// $ nnabla_cli nnb_template network.nntxt Affine_000.yaml
69 /// @endcode
70 ///
71 /// Then you should be got following 'Affine_000.yaml'.
72 /// @code{.yaml}
73 /// functions:
74 /// '@network1/Affine':
75 /// implement: 0
76 /// variables:
77 /// x: FLOAT32
78 /// y0: FLOAT32
79 /// network1/affine/W: FLOAT32
80 /// network1/affine/b: FLOAT32
81 /// @endcode
82 ///
83 /// Edit 'Affine_000.yaml' and save as 'settings.yaml'
84 /// @code{.yaml}
85 /// functions:
86 /// '@network1/Affine':
87 /// implement: 1
88 /// @endcode
89 ///
90 /// NOTE: If you DON'T want to use callback, set implements to 100.
91 ///
92 /// And convert nntxt to nnb again with settings.yaml.
93 /// @code{.sh}
94 /// $ nnabla_cli convert -s settings.yaml
95 /// nnabla_cli convert -s settings.yaml network.nntxt Affine_000.nnb
96 /// @endcode
97 ///
98 /// Then we create binary data with following command.
99 /// @code{.sh}
100 /// $ xxd -i <Affine_000.nnb >Affine_000.c
101 /// @endcode

208 /// @brief Example callback for allocate function local context.
209 /// It just does followings.
210 /// - check func->impl == 1
211 /// - register @ref cb_exec as function executor.
212 /// - register @ref cb_free as function local context de-allocator.
213 ///
214 /// See also @ref rt_initialize_context
215 static rt_return_value_t cb_alloc(nn_network_t *net, void *function_context) {
216  WHOAMI("%s", __func__);
217  rt_function_context_t *func = (rt_function_context_t *)function_context;
218  if ((int)func->info->impl != 1) {
219  return RT_RET_FUNCTION_DONT_MATCH;
220  }
221 
222  func->func.exec_func = cb_exec;
223  func->func.free_local_context_func = cb_free;
224  return RT_RET_FUNCTION_MATCH;
225 }
