39 rt_function_error_t allocate_prelu_local_context(rt_function_t *f) {
40  if (f->num_of_inputs != 2) {
41  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
42  }
43  if (f->num_of_outputs != 1) {
44  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
45  }
46 
47  prelu_private_t *p = rt_malloc_func(sizeof(prelu_private_t));
48  if (p == 0) {
49  return RT_FUNCTION_ERROR_MALLOC;
50  }
51  ((prelu_local_context_t *)(f->local_context))->data = (void *)p;
52  p->input = f->inputs[0];
53  p->get_input = select_getter(p->input);
54  p->input_size = calc_shape_size(f->inputs[0]->shape);
55  p->weight = f->inputs[1];
56  p->get_weight = select_getter(p->weight);
57  p->weight_size = calc_shape_size(f->inputs[1]->shape);
58  p->output = f->outputs[0];
59  p->set_output = select_setter(p->output);
60  p->in_shape = clone_list(f->inputs[0]->shape);
61  p->in_stride = calc_contiguous_strides(f->inputs[0]->shape);
62  if (p->input->type == NN_DATA_TYPE_FLOAT &&
63  p->weight->type == NN_DATA_TYPE_FLOAT &&
64  p->output->type == NN_DATA_TYPE_FLOAT) {
65 #ifdef CONFIG_PRELU_FLOAT32
66  f->exec_func = exec_prelu;
67 #endif /* CONFIG_PRELU_FLOAT32 */
68  } else {
69 #ifdef CONFIG_PRELU_GENERIC
70  f->exec_func = exec_prelu_generic;
71 #endif /* CONFIG_PRELU_GENERIC */
72  }
73  return RT_FUNCTION_ERROR_NOERROR;
74 }

86 rt_function_error_t exec_prelu(rt_function_t *f) {
87  prelu_local_context_t *context = (prelu_local_context_t *)(f->local_context);
88  prelu_private_t *p = (prelu_private_t *)(context->data);
89  float *x = (float *)(p->input->data);
90  float *w = (float *)(p->weight->data);
91  float *y = (float *)(p->output->data);
92  int base_shape = p->in_shape.data[context->base_axis];
93  int base_stride = p->in_stride.data[context->base_axis];
94 
95  if (p->weight_size == 1) {
96  for (int s = 0; s < p->input_size; ++s) {
97  y[s] = (x[s] >= 0) ? x[s] : x[s] * (*w);
98  }
99  } else {
100  for (int s = 0; s < p->input_size; ++s) {
101  int iw = (s / base_stride) % base_shape;
102  y[s] = (x[s] >= 0) ? x[s] : x[s] * w[iw];
103  }
104  }
105  return RT_FUNCTION_ERROR_NOERROR;
106 }
107 #endif /* CONFIG_PRELU_FLOAT32 */

110 rt_function_error_t exec_prelu_generic(rt_function_t *f) {
111  prelu_local_context_t *context = (prelu_local_context_t *)(f->local_context);
112  prelu_private_t *p = (prelu_private_t *)(context->data);
113  int base_shape = p->in_shape.data[context->base_axis];
114  int base_stride = p->in_stride.data[context->base_axis];
115 
116  if (p->weight_size == 1) {
117  float w = p->get_weight(p->weight, 0);
118  for (int s = 0; s < p->input_size; ++s) {
119  float x = p->get_input(p->input, s);
120  float y = (x >= 0) ? x : x * w;
121  p->set_output(p->output, s, y);
122  }
123  } else {
124  for (int s = 0; s < p->input_size; ++s) {
125  int iw = (s / base_stride) % base_shape;
126  float x = p->get_input(p->input, s);
127  float w = p->get_weight(p->weight, iw);
128  float y = (x >= 0) ? x : x * w;
129  p->set_output(p->output, s, y);
130  }
131  }
132  return RT_FUNCTION_ERROR_NOERROR;
133 }
