74  rt_context_pointer context, nn_function_type_t type,
75  rt_return_value_t (*allocate_local_context)(nn_network_t *net, void *f)) {
76 
77  rt_context_t *c = context;
78 
79  rt_function_callback_t *callbacks;
80  callbacks = rt_malloc_func((c->num_of_callbacks + 1) *
81  sizeof(rt_function_callback_t));
82  if (callbacks == 0) {
83  return RT_RET_ERROR_ALLOCATE_CALLBACK_BUFFER;
84  }
85 
86  if (c->callbacks == 0) {
87  c->callbacks = callbacks;
88  } else {
89  memcpy(callbacks, c->callbacks,
90  c->num_of_callbacks * sizeof(rt_function_callback_t));
91  rt_free_func(c->callbacks);
92  c->callbacks = callbacks;
93  }
94 
95  (c->callbacks + c->num_of_callbacks)->type = type;
96  (c->callbacks + c->num_of_callbacks)->allocate_local_context =
97  allocate_local_context;
98 
99  c->num_of_callbacks += 1;
100  return RT_RET_NOERROR;
101 }

103 rt_return_value_t rt_initialize_context(rt_context_pointer context,
104  nn_network_t *n) {
105  rt_context_t *c = context;
106  int i, j; // Iterator
107 
108  //////////////////////////////////////////////////////////////////////////////
109  // Binary format version check
110  if (n->version < NN_BINARY_FORMAT_MINIMUM_VERSION ||
111  n->version > NN_BINARY_FORMAT_VERSION) {
112  return RT_RET_ERROR_VERSION_UNMATCH;
113  }
114 
115  //////////////////////////////////////////////////////////////////////////////
116  // API level check
117  if (n->api_level > NN_API_LEVEL) {
118  return RT_RET_ERROR_VERSION_UNMATCH;
119  }
120 
121  //////////////////////////////////////////////////////////////////////////////
122  // Buffer list
123  c->num_of_buffers = n->buffers.size;
124  c->buffers =
125  rt_malloc_func(sizeof(rt_variable_buffer_context_t) * c->num_of_buffers);
126 
127  if (c->buffers == 0) {
128  return RT_RET_ERROR_ALLOCATE_CONTEXT;
129  }
130  for (i = 0; i < c->num_of_buffers; i++) {
131  c->buffers[i].allocate_type = RT_BUFFER_ALLOCATE_TYPE_INITIAL;
132  }
133 
134  //////////////////////////////////////////////////////////////////////////////
135  // Inputs
136  rt_list_t inputs = create_rt_list_from_nn_list(n, n->inputs);
137  c->num_of_inputs = inputs.size;
138  c->input_variable_ids = rt_malloc_func(sizeof(int *) * c->num_of_inputs);
139  for (i = 0; i < c->num_of_inputs; i++) {
140  c->input_variable_ids[i] = inputs.data[i];
141  }
142 
143  //////////////////////////////////////////////////////////////////////////////
144  // Outputs
145  rt_list_t outputs = create_rt_list_from_nn_list(n, n->outputs);
146  c->num_of_outputs = outputs.size;
147  c->output_variable_ids = rt_malloc_func(sizeof(int *) * c->num_of_outputs);
148  for (i = 0; i < c->num_of_outputs; i++) {
149  c->output_variable_ids[i] = outputs.data[i];
150  }
151 
152  //////////////////////////////////////////////////////////////////////////////
153  // Allocate buffers
154  int *list = (int *)NN_GET(n, n->buffers.list);
155  for (i = 0; i < c->num_of_buffers; i++) {
156  if (c->buffers[i].allocate_type == RT_BUFFER_ALLOCATE_TYPE_INITIAL) {
157  c->buffers[i].allocate_type = RT_BUFFER_ALLOCATE_TYPE_MALLOC;
158  if (n->version == 2) {
159  c->buffers[i].buffer =
160  rt_variable_malloc_func(*(list + i) * sizeof(float));
161  memset(c->buffers[i].buffer, 0, *(list + i) * sizeof(float));
162  } else {
163  c->buffers[i].buffer = rt_variable_malloc_func(*(list + i));
164  memset(c->buffers[i].buffer, 0, *(list + i));
165  }
166  if (c->buffers[i].buffer == 0) {
167  return RT_RET_ERROR_ALLOCATE_CONTEXT;
168  }
169  }
170  }
171 
172  //////////////////////////////////////////////////////////////////////////////
173  // Variables
174  c->num_of_variables = n->variables.size;
175  c->variables = rt_malloc_func(sizeof(rt_variable_t) * c->num_of_variables);
176  if (c->variables == 0) {
177  return RT_RET_ERROR_ALLOCATE_CONTEXT;
178  }
179 
180  list = (int *)NN_GET(n, n->variables.list);
181  for (i = 0; i < c->num_of_variables; i++) {
182  nn_variable_t *var = (nn_variable_t *)(NN_GET(n, *(list + i)));
183  c->variables[i].shape = create_rt_list_from_nn_list(n, var->shape);
184  c->variables[i].type = var->type;
185  c->variables[i].fp_pos = var->fp_pos;
186 
187  if (var->type == NN_DATA_TYPE_INT8 || var->type == NN_DATA_TYPE_INT16) {
188  c->variables[i].coefficient = (1.0 / (1 << var->fp_pos));
189  } else {
190  c->variables[i].coefficient = 0;
191  }
192 
193  if (var->data_index < 0) {
194  int index = (-1 * var->data_index) - 1;
195  if (index >= c->num_of_buffers) {
196  return RT_RET_ERROR_INVALID_BUFFER_INDEX;
197  }
198  c->variables[i].data = c->buffers[index].buffer;
199  } else {
200  c->variables[i].data = NN_GET(n, var->data_index);
201  }
202  }
203 
204  //////////////////////////////////////////////////////////////////////////////
205  // Functions
206  c->num_of_functions = n->functions.size;
207  c->functions =
208  rt_malloc_func(sizeof(rt_function_context_t) * c->num_of_functions);
209  list = (int *)NN_GET(n, n->functions.list);
210  for (i = 0; i < c->num_of_functions; i++) {
211  nn_function_t *func = (nn_function_t *)(NN_GET(n, *(list + i)));
212  c->functions[i] = allocate_function_io(n, c, func);
213 
214  int callback_registered_flag = 0;
215  if (func->impl <= NN_END_OF_USER_DEFINED_FUNCTION_IMPLEMENT) {
216  for (j = 0; j < c->num_of_callbacks; j++) {
217  if ((c->callbacks + j)->type == func->type) {
218  rt_return_value_t ret =
219  (c->callbacks +
220  j)->allocate_local_context(n, (void *)(&(c->functions[i])));
221  if (ret == RT_RET_FUNCTION_MATCH) {
222  callback_registered_flag = 1;
223  break;
224  }
225  }
226  }
227  }
228  if (!callback_registered_flag) {
229  allocate_function_context(n, func, c->functions + i);
230  }
231  }
232 
233  c->network = n;
234 
235  return RT_RET_NOERROR;
236 }

238 rt_return_value_t rt_free_context(rt_context_pointer *context) {
239  rt_context_t *c = *context;
240 
241  int i; // Iterator
242 
243  // Buffers
244  for (i = 0; i < c->num_of_buffers; i++) {
245  if (c->buffers[i].allocate_type == RT_BUFFER_ALLOCATE_TYPE_MALLOC) {
246  rt_variable_free_func(c->buffers[i].buffer);
247  }
248  }
249  rt_free_func(c->buffers);
250 
251  // Variables
252  rt_free_func(c->variables);
253 
254  // Functions
255  for (i = 0; i < c->num_of_functions; i++) {
256  rt_free_func(c->functions[i].func.inputs);
257  rt_free_func(c->functions[i].func.outputs);
258 
259  c->functions[i].func.free_local_context_func(&(c->functions[i].func));
260  if (c->functions[i].func.local_context != 0) {
261  rt_free_func(c->functions[i].func.local_context);
262  c->functions[i].func.local_context = NULL;
263  }
264  }
265  rt_free_func(c->functions);
266 
267  rt_free_func(c->input_variable_ids);
268  rt_free_func(c->output_variable_ids);
269 
270  // Callback
271  if (c->callbacks) {
272  rt_free_func(c->callbacks);
273  }
274 
275  rt_free_func(*context);
276  return RT_RET_NOERROR;
277 }
