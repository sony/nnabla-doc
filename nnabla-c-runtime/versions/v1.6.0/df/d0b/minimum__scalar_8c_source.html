25 rt_function_error_t allocate_minimum_scalar_local_context(rt_function_t *f) {
26  if (f->num_of_inputs != 1) {
27  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
28  }
29  if (f->num_of_outputs != 1) {
30  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
31  }
32  int input_size = calc_shape_size(f->inputs[0]->shape);
33  int output_size = calc_shape_size(f->outputs[0]->shape);
34  if (input_size != output_size) {
35  return RT_FUNCTION_ERROR_INVALID_SHAPE;
36  }
37  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&
38  f->outputs[0]->type == NN_DATA_TYPE_FLOAT) {
39 #ifdef CONFIG_MINIMUMSCALAR_FLOAT32
40  f->exec_func = exec_minimum_scalar;
41 #endif /* CONFIG_MINIMUMSCALAR_FLOAT32 */
42  } else {
43 #ifdef CONFIG_MINIMUMSCALAR_GENERIC
44  f->exec_func = exec_minimum_scalar_generic;
45 #endif /* CONFIG_MINIMUMSCALAR_GENERIC */
46  }
47  return RT_FUNCTION_ERROR_NOERROR;
48 }
