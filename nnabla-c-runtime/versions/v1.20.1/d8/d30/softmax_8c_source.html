34 rt_function_error_t allocate_softmax_local_context(rt_function_t *f) {
35  softmax_local_context_t *context =
36  (softmax_local_context_t *)(f->local_context);
37  softmax_private_t *p = rt_malloc_func(sizeof(softmax_private_t));
38  if (p == 0) {
39  return RT_FUNCTION_ERROR_MALLOC;
40  }
41  const int axis = context->axis;
42  const int size = calc_shape_size(f->inputs[0]->shape);
43  const int size_axis = axis <= 0 ? size
44  : shape_product_of(f->inputs[0], axis,
45  f->inputs[0]->shape.size);
46 
47  // axis must be less than ndim of inputs[0].
48  if (f->inputs[0]->shape.size <= axis) {
49  rt_free_func(p);
50  return RT_FUNCTION_ERROR_INVALID_SHAPE;
51  }
52  p->batch_size = size / size_axis;
53  p->specified_axis_size = f->inputs[0]->shape.data[axis];
54  p->output_size = size / p->batch_size / p->specified_axis_size;
55  if (p->batch_size * p->specified_axis_size * p->output_size != size) {
56  return RT_FUNCTION_ERROR_INVALID_SHAPE;
57  }
58  ((softmax_local_context_t *)(f->local_context))->data = (void *)p;
59  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&
60  f->outputs[0]->type == NN_DATA_TYPE_FLOAT) {
61 #ifdef CONFIG_SOFTMAX_FLOAT32
62  f->exec_func = exec_softmax;
63 #endif /* CONFIG_SOFTMAX_FLOAT32 */
64  } else {
65 #ifdef CONFIG_SOFTMAX_GENERIC
66  f->exec_func = exec_softmax_generic;
67 #endif /* CONFIG_SOFTMAX_GENERIC */
68  }
69  return RT_FUNCTION_ERROR_NOERROR;
70 }

78 rt_function_error_t exec_softmax(rt_function_t *f) {
79  softmax_local_context_t *context =
80  (softmax_local_context_t *)(f->local_context);
81  softmax_private_t *p = (softmax_private_t *)(context->data);
82  const float *const x = (float *)(f->inputs[0]->data);
83  float *const y = (float *)(f->outputs[0]->data);
84  const int batch_size = p->batch_size;
85  const int specified_axis_size = p->specified_axis_size;
86  const int output_size = p->output_size;
87 
88  int sample_index;
89  for (sample_index = 0; sample_index < batch_size; ++sample_index) {
90  int output_index;
91  for (output_index = 0; output_index < output_size; ++output_index) {
92  const int j =
93  sample_index * specified_axis_size * output_size + output_index;
94  // compute maximum
95  float max_input = x[j];
96  int specified_index;
97  for (specified_index = 0; specified_index < specified_axis_size;
98  ++specified_index) {
99  const int k = specified_index * output_size + j;
100  max_input = local_max(max_input, x[k]);
101  }
102  // Compute exponential and sum
103  float exp_sum = 0;
104  for (specified_index = 0; specified_index < specified_axis_size;
105  ++specified_index) {
106  const int k = specified_index * output_size + j;
107  const float tmp = expf(x[k] - max_input);
108  y[k] = tmp;
109  exp_sum += tmp;
110  }
111  // Compute softmax
112  for (specified_index = 0; specified_index < specified_axis_size;
113  ++specified_index) {
114  const int k = specified_index * output_size + j;
115  y[k] = y[k] / exp_sum;
116  }
117  }
118  }
119  return RT_FUNCTION_ERROR_NOERROR;
120 }
121 #endif /* CONFIG_SOFTMAX_FLOAT32 */

123 #ifdef CONFIG_SOFTMAX_GENERIC
124 rt_function_error_t exec_softmax_generic(rt_function_t *f) {
125  softmax_local_context_t *context =
126  (softmax_local_context_t *)(f->local_context);
127  softmax_private_t *p = (softmax_private_t *)(context->data);
128  rt_variable_t *input = f->inputs[0];
129  rt_variable_getter get_input = select_getter(input);
130  rt_variable_t *output = f->outputs[0];
131  rt_variable_getter get_output = select_getter(output);
132  rt_variable_setter set_output = select_setter(output);
133  const int batch_size = p->batch_size;
134  const int specified_axis_size = p->specified_axis_size;
135  const int output_size = p->output_size;
136 
137  int sample_index;
138  for (sample_index = 0; sample_index < batch_size; ++sample_index) {
139  int output_index;
140  for (output_index = 0; output_index < output_size; ++output_index) {
141  const int j =
142  sample_index * specified_axis_size * output_size + output_index;
143  // compute maximum
144  float max_input = get_input(input, j);
145  int specified_index;
146  for (specified_index = 0; specified_index < specified_axis_size;
147  ++specified_index) {
148  const int k = specified_index * output_size + j;
149  max_input = local_max(max_input, get_input(input, k));
150  }
151  // Compute exponential and sum
152  float exp_sum = 0;
153  for (specified_index = 0; specified_index < specified_axis_size;
154  ++specified_index) {
155  const int k = specified_index * output_size + j;
156  const float tmp = expf(get_input(input, k) - max_input);
157  set_output(output, k, tmp);
158  exp_sum += tmp;
159  }
160  // Compute softmax
161  for (specified_index = 0; specified_index < specified_axis_size;
162  ++specified_index) {
163  const int k = specified_index * output_size + j;
164  set_output(output, k, (get_output(output, k) / exp_sum));
165  }
166  }
167  }
168  return RT_FUNCTION_ERROR_NOERROR;
169 }
