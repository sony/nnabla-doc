19 // Common algorithm for arithmetic calculation between two vectors has same
20 // dimension.
21 void calc_dim_arithmetic(rt_variable_t *output, rt_variable_t *input1,
22  rt_variable_t *input2, int dim_index, int y_out,
23  int y_in1, int y_in2,
24  float (*calc_func)(float, float)) {
25  int dim_size = output->shape.size;
26  int x_size_out = output->shape.data[dim_index];
27  int x_size_in1 = input1->shape.data[dim_index];
28  int x_size_in2 = input2->shape.data[dim_index];
29  for (int x_out = 0; x_out < x_size_out; x_out++) {
30  int x_in1 = (x_out % x_size_in1);
31  int x_in2 = (x_out % x_size_in2);
32  int z_out = y_out * x_size_out + x_out;
33  int z_in1 = y_in1 * x_size_in1 + x_in1;
34  int z_in2 = y_in2 * x_size_in2 + x_in2;
35  if (dim_index + 1 < dim_size) {
36  calc_dim_arithmetic(output, input1, input2, dim_index + 1, z_out, z_in1,
37  z_in2, calc_func);
38  } else {
39  float *data_out = output->data;
40  float *data_in1 = input1->data;
41  float *data_in2 = input2->data;
42  data_out[z_out] = calc_func(data_in1[z_in1], data_in2[z_in2]);
43  }
44  }
45 }
