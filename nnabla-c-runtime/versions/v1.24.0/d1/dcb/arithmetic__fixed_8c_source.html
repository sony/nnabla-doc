21 // Common algorithm for arithmetic calculation between two vectors has same
22 // dimension.
23 void calc_dim_arithmetic_fixed8(rt_variable_t *output, rt_variable_t *input1,
24  rt_variable_t *input2, int dim_index, int y_out,
25  int y_in1, int y_in2,
26  int8_t (*calc_func)(int8_t, int8_t)) {
27  int dim_size = output->shape.size;
28  int x_size_out = output->shape.data[dim_index];
29  int x_size_in1 = input1->shape.data[dim_index];
30  int x_size_in2 = input2->shape.data[dim_index];
31  for (int x_out = 0; x_out < x_size_out; x_out++) {
32  int x_in1 = (x_out % x_size_in1);
33  int x_in2 = (x_out % x_size_in2);
34  int z_out = y_out * x_size_out + x_out;
35  int z_in1 = y_in1 * x_size_in1 + x_in1;
36  int z_in2 = y_in2 * x_size_in2 + x_in2;
37  if (dim_index + 1 < dim_size) {
38  calc_dim_arithmetic_fixed8(output, input1, input2, dim_index + 1, z_out,
39  z_in1, z_in2, calc_func);
40  } else {
41  int8_t *data_out = output->data;
42  int8_t *data_in1 = input1->data;
43  int8_t *data_in2 = input2->data;
44  data_out[z_out] = calc_func(data_in1[z_in1], data_in2[z_in2]);
45  }
46  }
47 }

137 void equalize_bits_fixed8(rt_variable_t *v1, rt_variable_t *v2) {
138  // Rescale two arrays to have the same number of range bits.
139  // Downscale the array having most precision bits.
140 
141  if (v1->fp_pos == v2->fp_pos) {
142  return;
143  }
144 
145  const rt_variable_t *to_keep = v1->fp_pos < v2->fp_pos ? v1 : v2;
146  rt_variable_t *to_rescale = to_keep == v1 ? v2 : v1;
147  const unsigned rescale_size = calc_shape_size(to_rescale->shape);
148 
149  rescale_fixed8(to_rescale->data, rescale_size, to_rescale->fp_pos,
150  to_keep->fp_pos);
151 
152  // Update fp_pos for rescaled variable
153  to_rescale->fp_pos = to_keep->fp_pos;
154 
155  return;
156 }

235 void calc_scalar_fixed8_largebuff(rt_function_t *f, float value,
236  int16_t (*calc_func)(int8_t, int8_t, int)) {
237  int out_size = calc_shape_size(f->outputs[0]->shape);
238  int8_t *input = f->inputs[0]->data;
239  int8_t *output = f->outputs[0]->data;
240 
241  // Compute needed number of range bits to represent given float.
242  // Notice that it should be log(abs(x)) instead of log(1+abs(x)), but the
243  // implemented
244  // formulation makes sure the output of the log is positive.
245  const short n_dec_bits_scalar =
246  roundf(fmaxf(0.0, 8.0 - (1.0 + log2f(1.0 + fabsf(value)))));
247 
248  // Convert scalar value to fixed-point.
249  const int8_t value_int = float_to_fixed8(value, n_dec_bits_scalar);
250  const int n_bits_rescale =
251  f->inputs[0]->fp_pos + n_dec_bits_scalar - f->outputs[0]->fp_pos;
252  const int rescaling_factor = 1 << n_bits_rescale;
253  int i; // Iterator
254  for (i = 0; i < out_size; i++) {
255  int16_t output_largebuff = calc_func(input[i], value_int, rescaling_factor);
256  output[i] = saturate16_to_8(output_largebuff);
257  }
258 }
