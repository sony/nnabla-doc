28 rt_function_error_t allocate_leaky_relu_local_context(rt_function_t *f) {
29  int input_size, output_size;
30 
31  if (f->num_of_inputs != 1) {
32  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
33  }
34  if (f->num_of_outputs != 1) {
35  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
36  }
37 
38  input_size = calc_shape_size(f->inputs[0]->shape);
39  output_size = calc_shape_size(f->outputs[0]->shape);
40 
41  if (input_size != output_size) {
42  return RT_FUNCTION_ERROR_INVALID_SHAPE;
43  }
44  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&
45  f->outputs[0]->type == NN_DATA_TYPE_FLOAT) {
46 #ifdef CONFIG_LEAKYRELU_FLOAT32
47  f->exec_func = exec_leaky_relu;
48 #endif /* CONFIG_LEAKYRELU_FLOAT32 */
49  } else {
50 #ifdef CONFIG_LEAKYRELU_GENERIC
51  f->exec_func = exec_leaky_relu_generic;
52 #endif /* CONFIG_LEAKYRELU_GENERIC */
53  }
54  return RT_FUNCTION_ERROR_NOERROR;
55 }

62 rt_function_error_t exec_leaky_relu(rt_function_t *f) {
63  leaky_relu_local_context_t *c =
64  (leaky_relu_local_context_t *)(f->local_context);
65  const float *x = (float *)(f->inputs[0]->data);
66  float *y = (float *)(f->outputs[0]->data);
67  const int output_size = calc_shape_size(f->inputs[0]->shape);
68  int i;
69 
70  for (i = 0; i < output_size; ++i) {
71  y[i] = x[i] > 0.0f ? x[i] : x[i] * c->alpha;
72  }
73  return RT_FUNCTION_ERROR_NOERROR;
74 }
75 #endif /* CONFIG_LEAKYRELU_FLOAT32 */

81  rt_variable_t *input = f->inputs[0];
82  rt_variable_getter get_input = select_getter(input);
83  rt_variable_t *output = f->outputs[0];
84  rt_variable_setter set_output = select_setter(output);
85  const int output_size = calc_shape_size(f->inputs[0]->shape);
86  int i;
87 
88  for (i = 0; i < output_size; ++i) {
89  float val_x = get_input(input, i);
90  float val_y = val_x > 0.0f ? val_x : val_x * c->alpha;
91  set_output(output, i, val_y);
92  }
93  return RT_FUNCTION_ERROR_NOERROR;
