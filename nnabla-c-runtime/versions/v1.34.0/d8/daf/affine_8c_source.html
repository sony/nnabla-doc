62  int i; // Iterator

119 rt_function_error_t exec_affine(rt_function_t *f) {
120  affine_private_t *p =
121  (affine_private_t *)(((affine_local_context_t *)(f->local_context))
122  ->data);
123  int i, j, k; // Iterators.
124  float *input = (float *)(p->input->data);
125  float *weight = (float *)(p->weight->data);
126  float *output = (float *)(p->output->data);
127  float *alpha = 0;
128 
129  if (p->alpha) {
130  alpha = (float *)p->alpha->data;
131  }
132 
133  // Clear output
134  memset(output, 0, sizeof(float) * p->output_size);
135 
136  for (k = 0; k < p->base_loop_size; k++) {
137  int output_offset = k * p->output_loop_size;
138  int input_offset = k * p->input_loop_size;
139  float *o_addr = output + output_offset;
140  float *alpha_addr = alpha;
141 
142  for (j = 0; j < p->output_loop_size; ++j) {
143  int weight_offset = j * p->input_loop_size;
144 
145  float *i_addr = input + input_offset;
146  float *w_addr = weight + weight_offset;
147  for (i = 0; i < p->input_loop_size; ++i) {
148  *o_addr += (*i_addr++) * (*w_addr++);
149  }
150  if (alpha) {
151  *o_addr *= *alpha_addr++;
152  }
153  ++o_addr;
154  }
155 
156  // Bias
157  if (p->bias) {
158  float *bias = (float *)(p->bias->data);
159  for (i = 0; i < p->output_loop_size; i++) {
160  int opos = output_offset + i;
161  int bpos = i;
162  *(output + opos) = *(output + opos) + *(bias + bpos);
163  }
164  }
165  }
166  return RT_FUNCTION_ERROR_NOERROR;
167 }
