33 static inline nn_size_t var_calc_offset(var_t *var, int *pos, int size) {
34  int *s = var->stride.data;
35  int len = var->stride.size > size ? size : var->stride.size;
36  int i, offset = 0;
37 
38  for (i = 0; i < len; ++i) {
39  offset += (*pos++) * (*s++);
40  }
41  return offset;
42 }

56 static inline void convnd(var_t *out, var_t *in, var_t *we,
57  rt_list_t input_shape, rt_list_t output_shape,
58  rt_list_t kernel_shape, rt_list_t in_position,
59  rt_list_t out_position, rt_list_t pad,
60  rt_list_t stride, rt_list_t dilation,
61  int spatial_dims) {
62  int output_size = calc_shape_size(output_shape);
63  int kernel_size = calc_shape_size(kernel_shape);
64 
65  for (int o = 0; o < output_size; o++) {
66  float sum = 0.0f;
67  pos_to_shape(out_position, output_shape, o);
68  for (int k = 0; k < kernel_size; k++) {
69  pos_to_shape(in_position, kernel_shape, k);
70  uint8_t condition = 1;
71  for (int j = 0; j < spatial_dims; j++) {
72  in_position.data[j] *= dilation.data[j];
73  in_position.data[j] -= pad.data[j];
74  in_position.data[j] += out_position.data[j] * stride.data[j];
75  if (in_position.data[j] < 0 ||
76  in_position.data[j] >= input_shape.data[j]) {
77  condition = 0;
78  break;
79  }
80  }
81  if (condition) {
82  float x =
83  var_get(in, in->offset + shape_to_pos(input_shape, in_position));
84  float w = var_get(we, we->offset + k);
85  sum += x * w;
86  }
87  }
88  float value = var_get(out, out->offset + o);
89  value += sum;
90  var_set(out, out->offset + o, value);
91  }
92 }

94 static inline void conv2d(var_t *out, var_t *in, var_t *we,
95  rt_list_t input_shape, rt_list_t output_shape,
96  rt_list_t kernel_shape, rt_list_t in_position,
97  rt_list_t out_position, rt_list_t pad,
98  rt_list_t stride, rt_list_t dilation,
99  int spatial_dims) {
100  int ix, iy, ox, oy, kx, ky;
101 
102  for (iy = -pad.data[0], oy = 0; oy < output_shape.data[0];
103  ++oy, iy += stride.data[0]) {
104  for (ix = -pad.data[1], ox = 0; ox < output_shape.data[1];
105  ++ox, ix += stride.data[1]) {
106  float sum = 0.0f;
107  for (ky = 0; ky < kernel_shape.data[0]; ++ky) {
108  for (kx = 0; kx < kernel_shape.data[1]; ++kx) {
109  int iky = ky * dilation.data[0] + iy;
110  int ikx = kx * dilation.data[1] + ix;
111  if (ikx >= 0 && ikx < input_shape.data[1] && iky >= 0 &&
112  iky < input_shape.data[0]) {
113  float x = var_get(in, in->offset + iky * input_shape.data[1] + ikx);
114  float w = var_get(we, we->offset + ky * kernel_shape.data[1] + kx);
115  sum += x * w;
116  }
117  }
118  }
119  float o = var_get(out, out->offset + oy * output_shape.data[1] + ox);
120  o += sum;
121  var_set(out, out->offset + oy * output_shape.data[1] + ox, o);
122  }
123  }
124 }

148 rt_function_error_t exec_convolution_generic(rt_function_t *f) {
149  convolution_local_context_t *c =
150  (convolution_local_context_t *)f->local_context;
151  convolution_private_t *p = (convolution_private_t *)(c->data);
152 
153  nn_size_t group = c->group;
154  nn_size_t in_vars = p->in_var.shape.data[I];
155  nn_size_t out_vars = p->out_var.shape.data[I];
156  nn_size_t batch_size;
157  nn_size_t om, im, g, b;
158  var_t *out_var = &p->out_var;
159  var_t *in_var = &p->in_var;
160  var_t *w_var = &p->w_var;
161  var_t *b_var = &p->b_var;
162  var_t *a_var = &p->a_var;
163 
164  int output_size = calc_shape_size(p->out_var.shape);
165 
166  memset(out_var->v->data, 0, sizeof(float) * output_size);
167 
168  batch_size = p->in_var.shape.data[0];
169  if (p->spatial_dims == 2) {
170  for (b = 0; b < batch_size; ++b) {
171  for (g = 0; g < group; ++g) {
172  for (om = 0; om < out_vars; ++om) {
173  int o_pos[] = {b, g, om};
174  var_setpos(out_var, o_pos, _S(o_pos));
175  for (im = 0; im < in_vars; ++im) {
176  int i_pos[] = {b, g, im};
177  int w_pos[] = {g, om, im};
178  var_setpos(in_var, i_pos, _S(i_pos));
179  var_setpos(w_var, w_pos, _S(w_pos));
180  conv2d(out_var, in_var, w_var, p->input_shape, p->output_shape,
181  p->kernel_shape, p->in_position, p->out_position, c->pad,
182  c->stride, c->dilation, p->spatial_dims);
183  }
184  {
185  int b_pos[] = {g, om};
186  if (p->a_var.v) {
187  var_setpos(a_var, b_pos, _S(b_pos));
188  mul_alpha(out_var, a_var);
189  }
190  if (p->b_var.v) {
191  var_setpos(b_var, b_pos, _S(b_pos));
192  add_bias(out_var, b_var);
193  }
194  }
195  }
196  }
197  }
198  } else {
199  for (b = 0; b < batch_size; ++b) {
200  for (g = 0; g < group; ++g) {
201  for (om = 0; om < out_vars; ++om) {
202  int o_pos[] = {b, g, om};
203  var_setpos(out_var, o_pos, _S(o_pos));
204  for (im = 0; im < in_vars; ++im) {
205  int i_pos[] = {b, g, im};
206  int w_pos[] = {g, om, im};
207  var_setpos(in_var, i_pos, _S(i_pos));
208  var_setpos(w_var, w_pos, _S(w_pos));
209  convnd(out_var, in_var, w_var, p->input_shape, p->output_shape,
210  p->kernel_shape, p->in_position, p->out_position, c->pad,
211  c->stride, c->dilation, p->spatial_dims);
212  }
213  {
214  int b_pos[] = {g, om};
215  if (p->a_var.v) {
216  var_setpos(a_var, b_pos, _S(b_pos));
217  mul_alpha(out_var, a_var);
218  }
219  if (p->b_var.v) {
220  var_setpos(b_var, b_pos, _S(b_pos));
221  add_bias(out_var, b_var);
222  }
223  }
224  }
225  }
226  }
227  }
