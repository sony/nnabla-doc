79 /**
80  * https://arxiv.org/pdf/1603.05201.pdf
81  * Definition 2.1. CReLU activation, denoted by ρc : R →R2
82 ,* is defined as follows: ∀x ∈ R, ρc(x) = ([x]+, [−x]+).
83  */
84 rt_function_error_t exec_crelu(rt_function_t *f) {
85  crelu_local_context_t *c = (crelu_local_context_t *)(f->local_context);
86  crelu_private_t *p = (crelu_private_t *)(c->data);
87  int i, j, s0 = 1, s1;
88 
89  for (i = c->axis; i < p->in_shape.size; ++i) {
90  s0 *= p->in_shape.data[i];
91  }
92  s1 = p->input_size / s0;
93 
94  for (i = 0; i < s1; ++i) {
95  for (j = 0; j < s0; ++j) {
96  float x = *((float *)(p->input->data) + i * s0 + j);
97  float *y = (float *)(p->output->data);
98  *(y + i * s0 * 2 + j) = x > 0.0f ? x : 0.0f;
99  *(y + i * s0 * 2 + s0 + j) = x < 0.0f ? -x : 0.0f;
100  }
101  }
102  return RT_FUNCTION_ERROR_NOERROR;
103 }

105 rt_function_error_t exec_crelu_generic(rt_function_t *f) {
106  crelu_local_context_t *c = (crelu_local_context_t *)(f->local_context);
107  crelu_private_t *p = (crelu_private_t *)(c->data);
108  int i, j, s0 = 1, s1;
109 
110  for (i = c->axis; i < p->in_shape.size; ++i) {
111  s0 *= p->in_shape.data[i];
112  }
113  s1 = p->input_size / s0;
114 
115  for (i = 0; i < s1; ++i) {
116  for (j = 0; j < s0; ++j) {
117  float x = p->get_input(p->input, i * s0 + j);
118  float value = x > 0.0f ? x : 0.0f;
119  p->set_output(p->output, i * s0 * 2 + j, value);
120  value = x < 0.0f ? -x : 0.0f;
121  p->set_output(p->output, i * s0 * 2 + s0 + j, value);
122  }
123  }
124  return RT_FUNCTION_ERROR_NOERROR;
125 }
