21 inline int sum_acc_sat8(int8_t *acc, int8_t b) {
22  // Accumulate b into *acc, after cheking for overflow (8-bit accumulator)
23  int8_t result = *acc + b;
24  if (*acc > 0 && b > 0 && result < 0) {
25  *acc = INT8_MAX;
26  return -1;
27  } else if (*acc < 0 && b < 0 && result > 0) {
28  *acc = INT8_MIN;
29  return -1;
30  } else {
31  *acc = result;
32  return 0;
33  }

121 inline void rescale_fixed8(int8_t *arr, unsigned arr_size,
122  unsigned old_decimal_bits,
123  unsigned new_decimal_bits) {
124  // Rescale an array of 8-bit fixed-point values,
125  // from old_decimal_bits precision, to new_decimal_bits precision.
126 
127  const short n_bits_shift = old_decimal_bits - new_decimal_bits;
128  if (n_bits_shift > 0) {
129  const int8_t rescaling = 1 << n_bits_shift;
130  for (int i = 0; i < arr_size; i++) {
131  *(arr + i) /= rescaling;
132  }
133  } else if (n_bits_shift < 0) {
134  const int16_t rescaling = 1 << -n_bits_shift;
135  for (int i = 0; i < arr_size; i++) {
136  *(arr + i) = saturate16_to_8(*(arr + i) * rescaling); // ovf could occur
137  }
138  }
139 }

268 inline int8_t float_to_fixed8(float value, int n_precision_bits) {
269  const float FIXED8_MAX =
270  (1 << (8 - n_precision_bits - 1)) - (1 / (1 << n_precision_bits));
271  const float FIXED8_MIN = -(1 << (8 - n_precision_bits - 1));
272 
273  if (value >= FIXED8_MAX) {
274  return (int8_t)FIXED8_MAX;
275  } else if (value <= FIXED8_MIN) {
276  return (int8_t)FIXED8_MIN;
277  } else {
278  return (int8_t)roundf(value * (1 << n_precision_bits));
279  }
