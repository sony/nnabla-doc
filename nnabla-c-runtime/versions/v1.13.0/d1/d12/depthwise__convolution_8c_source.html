23 rt_function_error_t
24 allocate_depthwise_convolution_local_context(rt_function_t *f) {
25  const int bias = 2;
26  const int x = 0;
27  const int weight = 1;
28  const int y0 = 0;
29  depthwise_convolution_local_context_t *c =
30  (depthwise_convolution_local_context_t *)(f->local_context);
31  int i;
32  int multiplier = c->multiplier;
33  int group;
34 
35  if (f->num_of_inputs != bias && f->num_of_inputs != bias + 1) {
36  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
37  }
38 
39  if (f->num_of_outputs != 1) {
40  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
41  }
42 
43  rt_list_t in_shape = f->inputs[x]->shape;
44  rt_list_t w_shape = f->inputs[weight]->shape;
45  int spatial_dims = in_shape.size - c->base_axis - 1;
46 
47  if (c->base_axis >= in_shape.size - 1) {
48  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
49  }
50 
51  convolution_private_t *p = rt_malloc_func(sizeof(convolution_private_t));
52  if (p == 0) {
53  return RT_FUNCTION_ERROR_MALLOC;
54  }
55  c->data = (void *)p;
56 
57  p->in_var.shape = allocate_list(spatial_dims + 3);
58  p->in_var.shape.data[B] = 1;
59  for (i = 0; i < c->base_axis; ++i) {
60  p->in_var.shape.data[B] *= in_shape.data[i];
61  }
62  p->in_var.shape.data[G] = in_shape.data[c->base_axis];
63  p->in_var.shape.data[I] = 1;
64  // Potential Issue: <to avoid duplicated convolution codes>
65  // c->multiplier is used as c->group.
66  group = c->multiplier = in_shape.data[c->base_axis];
67 
68  p->out_var.shape = allocate_list(spatial_dims + 3);
69  p->out_var.shape.data[B] = p->in_var.shape.data[B];
70  p->out_var.shape.data[G] = group;
71  p->out_var.shape.data[I] = multiplier;
72 
73  for (i = 0; i < spatial_dims; ++i) {
74  int k = w_shape.data[i + 1];
75  int dims = p->in_var.shape.data[i + 3] =
76  in_shape.data[c->base_axis + 1 + i];
77  int ks = c->dilation.data[i] * (k - 1) + 1;
78  int o = (dims + 2 * c->pad.data[i] - ks) / c->stride.data[i] + 1;
79  p->out_var.shape.data[i + 3] = o;
80  }
81  p->in_var.v = f->inputs[x];
82  p->in_var.get = select_getter(f->inputs[x]);
83  p->in_var.offset = 0;
84  p->in_var.stride = calc_contiguous_strides(p->in_var.shape);
85  p->out_var.v = f->outputs[y0];
86  p->out_var.get = select_getter(f->outputs[y0]);
87  p->out_var.set = select_setter(f->outputs[y0]);
88  p->out_var.offset = 0;
89  p->out_var.stride = calc_contiguous_strides(p->out_var.shape);
90 
91  p->w_var.v = f->inputs[weight];
92  p->w_var.get = select_getter(f->inputs[weight]);
93  p->w_var.offset = 0;
94  p->w_var.shape = allocate_list(spatial_dims + 3);
95  p->w_var.shape.data[KG] = group;
96  p->w_var.shape.data[KO] = multiplier;
97  p->w_var.shape.data[KI] = 1;
98  for (i = 0; i < spatial_dims; i++) {
99  p->w_var.shape.data[i + 3] = w_shape.data[i + 1];
100  }
101  p->w_var.stride = calc_contiguous_strides(p->w_var.shape);
102 
103  if (f->num_of_inputs > bias) {
104  p->b_var.v = f->inputs[bias];
105  p->b_var.get = select_getter(f->inputs[bias]);
106  p->b_var.offset = 0;
107  p->b_var.shape = allocate_list(2);
108  p->b_var.shape.data[KG] = group;
109  p->b_var.shape.data[KO] = multiplier;
110  p->b_var.stride = calc_contiguous_strides(p->b_var.shape);
111  } else {
112  p->b_var.v = 0;
113  }
114 
115  p->a_var.v = 0;
116  p->spatial_dims = spatial_dims;
117  p->input_shape = allocate_list(p->spatial_dims);
118  p->kernel_shape = allocate_list(p->spatial_dims);
119  p->output_shape = allocate_list(p->spatial_dims);
120  p->in_position = allocate_list(p->spatial_dims);
121  p->out_position = allocate_list(p->spatial_dims);
122 
123  for (i = 0; i < p->spatial_dims; i++) {
124  p->kernel_shape.data[i] = p->w_var.shape.data[i + 3];
125  p->input_shape.data[i] = p->in_var.shape.data[i + 3];
126  p->output_shape.data[i] = p->out_var.shape.data[i + 3];
127  }
128 
129 #ifdef CONFIG_DEPTHWISECONVOLUTION_FLOAT32
130  f->exec_func = exec_depthwise_convolution;
131 #endif /* CONFIG_DEPTHWISECONVOLUTION_FLOAT32 */
132 
133 #ifdef CONFIG_DEPTHWISECONVOLUTION_GENERIC
134  for (int i = 0; i < f->num_of_inputs; i++) {
135 
136 #ifdef CONFIG_DEPTHWISECONVOLUTION_FIXED8
137  if (f->inputs[i]->type == NN_DATA_TYPE_INT8) {
138  f->exec_func = exec_convolution_int8;
139  break;
140  }
141 #endif /* CONFIG_DEPTHWISECONVOLUTION_FIXED8 */
142 
143 #ifdef CONFIG_DEPTHWISECONVOLUTION_FIXED16
144  if (f->inputs[i]->type == NN_DATA_TYPE_INT16) {
145  f->exec_func = exec_convolution_int16;
146  break;
147  }
148 #endif /* CONFIG_DEPTHWISECONVOLUTION_FIXED16 */
149 
150  if (f->inputs[i]->type != NN_DATA_TYPE_FLOAT) {
151  f->exec_func = exec_convolution_generic;
152  break;
153  }
154  }
155 
156  for (int i = 0; i < f->num_of_outputs; i++) {
157  int conv_output_assigned = 0;
158 
159 #ifdef CONFIG_DEPTHWISECONVOLUTION_FIXED8
160  if ((f->outputs[i]->type == NN_DATA_TYPE_INT8) &&
161  (f->exec_func == exec_convolution_int8)) {
162  f->exec_func = exec_convolution_int8;
163  conv_output_assigned = 1;
164  break;
165  }
166 #endif /* CONFIG_DEPTHWISECONVOLUTION_FIXED8 */
167 
168 #ifdef CONFIG_DEPTHWISECONVOLUTION_FIXED16
169  if ((f->outputs[i]->type == NN_DATA_TYPE_INT16) &&
170  (f->exec_func == exec_convolution_int16)) {
171  f->exec_func = exec_convolution_int16;
172  conv_output_assigned = 1;
173  break;
174  }
175 #endif /* CONFIG_DEPTHWISECONVOLUTION_FIXED16 */
176 
177 #ifdef CONFIG_DEPTHWISECONVOLUTION_FLOAT32
178  if ((f->outputs[i]->type == NN_DATA_TYPE_FLOAT) &&
179  (f->exec_func == exec_convolution_float)) {
180  f->exec_func = exec_convolution_float;
181  conv_output_assigned = 1;
182  break;
183  }
184 #endif /* CONFIG_DEPTHWISECONVOLUTION_FLOAT32 */
185 
186  if (!conv_output_assigned) {
187  f->exec_func = exec_convolution_generic;
188  }
189  }
190 #endif /* CONFIG_DEPTHWISECONVOLUTION_GENERIC */
191 
192  c->multiplier = group;
193 
194  return RT_FUNCTION_ERROR_NOERROR;
195 }

202 rt_function_error_t free_depthwise_convolution_local_context(rt_function_t *f) {
203  depthwise_convolution_local_context_t *c =
204  (depthwise_convolution_local_context_t *)(f->local_context);
205  convolution_private_t *p = (convolution_private_t *)c->data;
206  var_free(&p->out_var);
207  var_free(&p->in_var);
208  var_free(&p->w_var);
209  if (p->b_var.v != 0)
210  var_free(&p->b_var);
211  if (p->a_var.v != 0)
212  var_free(&p->a_var);
213  free_list(p->input_shape);
214  free_list(p->kernel_shape);
215  free_list(p->in_position);
216  free_list(p->out_position);
217  free_list(p->output_shape);
218  rt_free_func(p);
219  return RT_FUNCTION_ERROR_NOERROR;
220 }
