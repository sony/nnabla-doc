24 static int check_data_size(int actual_size, int expected_size) {
25  if (actual_size != expected_size) {
26  printf("Input data size is invalid.\n");
27  printf("Expected data size is: %d\n", expected_size);
28  printf("Actual data size is: %d\n", actual_size);
29  return -1;
30  }
31  return 0;
32 }

34 int infer(nn_network_t *net, int argc, char *argv[]) {
35  int i, j; // Iterator
36  rt_return_value_t ret;
37  const char *data_type[] = {"NN_DATA_TYPE_FLOAT", "NN_DATA_TYPE_INT16",
38  "NN_DATA_TYPE_INT8", "NN_DATA_TYPE_SIGN",
39  "END_OF_NN_DATA_TYPE"};
40 
41  for (i = 0; i < argc; i++) {
42  printf(" %d: %s\n", i, argv[i]);
43  }
44 
45  rt_context_pointer context = 0;
46 
47  ret = rt_allocate_context(&context);
48  assert(ret == RT_RET_NOERROR);
49 
50  ret = rt_initialize_context(context, net);
51  if (ret != RT_RET_NOERROR) {
52  switch (ret) {
53  case RT_RET_ERROR_VERSION_UNMATCH:
54  printf("input network(.nnb) is mismatch to current runtime library.\n");
55  return -1;
56  case RT_RET_ERROR_INITIALIZE_CONTEXT_TWICE:
57  printf("cannot initialize context twice.\n");
58  return -1;
59  default:
60  printf("rt_initialize_context() failed with %d.\n", ret);
61  return -1;
62  }
63  }
64 
65  if (argc != rt_num_of_input(context) + 1) {
66  printf("Incorrect input parameters.\n");
67  printf(" Required inputs: %d and output: 1\n", rt_num_of_input(context));
68  printf(" Actual inputs: %d\n", argc);
69  return -1;
70  }
71 
72  for (i = 0; i < rt_num_of_input(context); i++) {
73  printf("Input[%d] size:%d\n", i, rt_input_size(context, i));
74  FILE *input = 0;
75 #ifdef _MSC_VER
76  fopen_s(&input, *argv, "rb");
77 #else
78  input = fopen(*argv, "rb");
79 #endif
80  argv += 1;
81  if (input == NULL) {
82  printf("Cannot open input file: %s.\n", *(argv - 1));
83  return -1;
84  }
85  fseek(input, 0L, SEEK_END);
86  size_t input_data_size = ftell(input);
87  int read_size = 0;
88 
89  nn_variable_t *variable = rt_input_variable(context, i);
90  printf("Input[%d] data type:%s, fp:%d\n", i, data_type[variable->type],
91  variable->fp_pos);
92 
93  switch (variable->type) {
94  case NN_DATA_TYPE_FLOAT:
95  if (check_data_size(input_data_size,
96  rt_input_size(context, i) * sizeof(float)) < 0)
97  return -1;
98  fseek(input, 0L, SEEK_SET);
99  read_size = (int)fread(rt_input_buffer(context, i), sizeof(uint8_t),
100  input_data_size, input);
101  break;
102  case NN_DATA_TYPE_INT8:
103  if (check_data_size(input_data_size,
104  rt_input_size(context, i) * sizeof(uint8_t)) < 0)
105  return -1;
106  fseek(input, 0L, SEEK_SET);
107  read_size = (int)fread(rt_input_buffer(context, i), sizeof(uint8_t),
108  input_data_size, input);
109  break;
110  case NN_DATA_TYPE_INT16:
111  if (check_data_size(input_data_size,
112  rt_input_size(context, i) * sizeof(uint16_t)) < 0)
113  return -1;
114  fseek(input, 0L, SEEK_SET);
115  read_size = (int)fread(rt_input_buffer(context, i), sizeof(uint8_t),
116  input_data_size, input);
117  break;
118  case NN_DATA_TYPE_SIGN:
119  if (check_data_size(input_data_size, rt_input_size(context, i) >> 3) < 0)
120  return -1;
121  fseek(input, 0L, SEEK_SET);
122  read_size = (int)fread(rt_input_buffer(context, i), sizeof(uint8_t),
123  input_data_size, input);
124  break;
125  default:
126  printf("Type: %d is not yet supported.", variable->type);
127  return -1;
128  }
129 
130  assert(read_size == input_data_size);
131  if (read_size != input_data_size) {
132  printf("Failed to read input data, expected to read: %d bytes, actual "
133  "read: %d bytes\n",
134  (int)input_data_size, (int)read_size);
135  }
136  fclose(input);
137 
138  printf("Input[%d] Shape (", i);
139  for (j = 0; j < rt_input_dimension(context, i); j++) {
140  printf(" %d", rt_input_shape(context, i, j));
141  }
142  printf(" )\n");
143  }
144 
145  ret = rt_forward(context);
146  if (ret != RT_RET_NOERROR) {
147  printf("Error occurs in infer process. %d\n", ret);
148  return -1;
149  }
150 
151  for (i = 0; i < rt_num_of_output(context); i++) {
152  printf("Output[%d] size:%d\n", i, rt_output_size(context, i));
153  size_t output_filename_length = strlen(*argv) + 10;
154  char *output_filename = malloc(output_filename_length);
155  assert(output_filename);
156 #ifdef _MSC_VER
157  sprintf_s(output_filename, output_filename_length, "%s_%d.bin", *argv, i);
158 #else
159  snprintf(output_filename, output_filename_length, "%s_%d.bin", *argv, i);
160 #endif
161  printf("Output[%d] filename %s\n", i, output_filename);
162 
163  FILE *output = 0;
164 #ifdef _MSC_VER
165  fopen_s(&output, output_filename, "wb");
166 #else
167  output = fopen(output_filename, "wb");
168 #endif
169  if (output == NULL) {
170  printf("Cannot open output file:%s.\n", output_filename);
171  return -1;
172  }
173  int output_data_type = rt_output_variable(context, i)->type;
174  int output_data_size;
175 
176  switch (output_data_type) {
177  case NN_DATA_TYPE_FLOAT:
178  output_data_size = rt_output_size(context, i) * sizeof(float);
179  break;
180  case NN_DATA_TYPE_INT8:
181  output_data_size = rt_output_size(context, i) * sizeof(int8_t);
182  break;
183  case NN_DATA_TYPE_INT16:
184  output_data_size = rt_output_size(context, i) * sizeof(int16_t);
185  break;
186  case NN_DATA_TYPE_SIGN:
187  output_data_size = rt_output_size(context, i) >> 3;
188  break;
189  default:
190  printf("Type: %d is not yet supported.", output_data_type);
191  return -1;
192  }
193 
194  int output_write_size =
195  (int)fwrite(rt_output_buffer(context, i), sizeof(uint8_t),
196  output_data_size, output);
197  if (output_write_size != output_data_size) {
198  printf("Failed to write output. Expected writing: %d bytes, actual "
199  "writing: %d bytes\n",
200  output_data_size, output_write_size);
201  return -1;
202  }
203 
204  fclose(output);
205  free(output_filename);
206 
207  printf("Output[%d] Shape (", i);
208  for (j = 0; j < rt_output_dimension(context, i); j++) {
209  printf(" %d", rt_output_shape(context, i, j));
210  }
211  printf(" )\n");
212 
213  nn_variable_t *variable = rt_output_variable(context, i);
214  printf("Output[%d] data type:%s, fp:%d\n", i, data_type[variable->type],
215  variable->fp_pos);
216  }
217  ret = rt_free_context(&context);
218  assert(ret == RT_RET_NOERROR);
219  return 0;
220 }
