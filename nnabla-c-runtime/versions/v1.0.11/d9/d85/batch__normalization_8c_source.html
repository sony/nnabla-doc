37 allocate_batch_normalization_local_context(rt_function_t *f) {
38  batch_normalization_local_context_t *context =
39  (batch_normalization_local_context_t *)(f->local_context);
40  batch_normalization_private_t *p =
41  rt_malloc_func(sizeof(batch_normalization_private_t));
42  if (p == 0) {
43  return RT_FUNCTION_ERROR_MALLOC;
44  }
45  // Check axes
46  if (context->axes.size != 1) {
47  return RT_FUNCTION_ERROR_INVALID_SHAPE;
48  }
49  // Check and parse shapes
50  rt_list_t input_shape = clone_list(f->inputs[0]->shape);
51  const int axis = context->axes.data[0];
52  const int size = calc_shape_size(input_shape);
53  const int size_axis =
54  axis <= 0 ? size : shape_product_of(f->inputs[0], axis,
55  f->inputs[0]->shape.size);
56  p->batch_size = size / size_axis;
57  p->specified_axis_size = input_shape.data[context->axes.data[0]];
58  p->output_size = size / p->batch_size / p->specified_axis_size;
59  p->multiplication_axis_output = p->specified_axis_size * p->output_size;
60  p->multiplication_batch_axis = p->batch_size * p->output_size;
61  if (p->batch_size * p->specified_axis_size * p->output_size != size) {
62  return RT_FUNCTION_ERROR_INVALID_SHAPE;
63  }
64 
65  p->batch_mean.shape = clone_list(f->inputs[1]->shape);
66  p->batch_var.shape = clone_list(f->inputs[2]->shape);
67  p->batch_mean.data =
68  rt_malloc_func(sizeof(float) * calc_shape_size(p->batch_mean.shape));
69  p->batch_var.data =
70  rt_malloc_func(sizeof(float) * calc_shape_size(p->batch_var.shape));
71  free_list(input_shape);
72  ((batch_normalization_local_context_t *)(f->local_context))->data = (void *)p;
73 
74 #ifdef CONFIG_BATCHNORMALIZATION_FLOAT32
75  f->exec_func = exec_batch_normalization;
76 #endif /* CONFIG_BATCHNORMALIZATION_FLOAT32 */
77 #ifdef CONFIG_BATCHNORMALIZATION_GENERIC
78  for (int i = 0; i < f->num_of_inputs; i++) {

88  return RT_FUNCTION_ERROR_NOERROR;
89 }

105 static void forward_impl_batch(rt_function_t *f,
106  batch_normalization_local_context_t *context,
107  batch_normalization_private_t *p) {
108  const float *x = (float *)(f->inputs[0]->data);
109  const float *beta = (float *)(f->inputs[1]->data);
110  const float *gamma = (float *)(f->inputs[2]->data);
111  float *y = (float *)(f->outputs[0]->data);
112  float *m = (float *)p->batch_mean.data; // batch mean
113  float *v = (float *)p->batch_var.data; // batch var
114  float *rm = (float *)(f->inputs[3]->data); // running mean
115  float *rv = (float *)(f->inputs[4]->data); // running var
116  const int specified_axis_size = p->specified_axis_size;
117  const int output_size = p->output_size;
118  const int multiplication_axis_output = p->multiplication_axis_output;
119  const int multiplication_batch_axis = p->multiplication_batch_axis;
120 
121  if (f->num_of_outputs == 3) {
122  m = (float *)(f->outputs[1]->data);
123  v = (float *)(f->outputs[2]->data);
124  }
125 
126  int i1;
127  for (i1 = 0; i1 < specified_axis_size; i1++) {
128  m[i1] = 0;
129  v[i1] = 0;
130  int i02;
131  for (i02 = 0; i02 < multiplication_batch_axis; i02++) {
132  const int i0 = i02 / output_size;
133  const int i2 = i02 % output_size;
134  const int i = i0 * multiplication_axis_output + i1 * output_size + i2;
135  const float value = x[i];
136  m[i1] += value;
137  v[i1] += value * value;
138  }
139  m[i1] /= multiplication_batch_axis;
140  v[i1] = v[i1] / multiplication_batch_axis - m[i1] * m[i1];
141 
142  // Moving mean and var
143  rm[i1] = context->decay_rate * rm[i1] + (1 - context->decay_rate) * m[i1];
144  rv[i1] = context->decay_rate * rv[i1] +
145  (1 - context->decay_rate) * v[i1] * multiplication_batch_axis /
146  (multiplication_batch_axis - 1);
147 
148  const float stdvar = (float)sqrt(v[i1] + context->eps);
149  // Subtract mean and divide by std, and apply beta and gamma.
150  for (i02 = 0; i02 < multiplication_batch_axis; i02++) {
151  const int i0 = i02 / output_size;
152  const int i2 = i02 % output_size;
153  const int i = i0 * multiplication_axis_output + i1 * output_size + i2;
154  y[i] = (x[i] - m[i1]) * gamma[i1] / stdvar + beta[i1];
155  }
156  }
157 }

205 forward_impl_batch_generic(rt_function_t *f,
206  batch_normalization_local_context_t *context,
207  batch_normalization_private_t *p) {
208  rt_variable_t *input_x = f->inputs[0];
209  rt_variable_getter get_x = select_getter(input_x);
210  rt_variable_t *input_beta = f->inputs[1];
211  rt_variable_getter get_beta = select_getter(input_beta);
212  rt_variable_t *input_gamma = f->inputs[2];
213  rt_variable_getter get_gamma = select_getter(input_gamma);
214  rt_variable_t *input_rm = f->inputs[3]; // running mean
215  rt_variable_getter get_rm = select_getter(input_rm);
216  rt_variable_setter set_rm = select_setter(input_rm);
217  rt_variable_t *input_rv = f->inputs[4]; // running var
218  rt_variable_getter get_rv = select_getter(input_rv);
219  rt_variable_setter set_rv = select_setter(input_rv);
220  rt_variable_t *output = f->outputs[0];
221  rt_variable_setter set_output = select_setter(output);
222  float *m = (float *)p->batch_mean.data; // batch mean
223  float *v = (float *)p->batch_var.data; // batch var
224  const int specified_axis_size = p->specified_axis_size;
225  const int output_size = p->output_size;
226  const int multiplication_axis_output = p->multiplication_axis_output;
227  const int multiplication_batch_axis = p->multiplication_batch_axis;
228 
229  if (f->num_of_outputs == 3) {
230  m = (float *)(f->outputs[1]->data);
231  v = (float *)(f->outputs[2]->data);
232  }
233 
234  int i1;
235  for (i1 = 0; i1 < specified_axis_size; i1++) {
236  m[i1] = 0;
237  v[i1] = 0;
238  int i02;
239  for (i02 = 0; i02 < multiplication_batch_axis; i02++) {
240  const int i0 = i02 / output_size;
241  const int i2 = i02 % output_size;
242  const int i = i0 * multiplication_axis_output + i1 * output_size + i2;
243  const float value = get_x(input_x, i);
244  m[i1] += value;
245  v[i1] += value * value;
246  }
247  m[i1] /= multiplication_batch_axis;
248  v[i1] = v[i1] / multiplication_batch_axis - m[i1] * m[i1];
249 
250  // Moving mean and var
251  float rm = get_rm(input_rm, i1);
252  float rv = get_rv(input_rv, i1);
253  rm = context->decay_rate * rm + (1 - context->decay_rate) * m[i1];
254  rv = context->decay_rate * rv +
255  (1 - context->decay_rate) * v[i1] * multiplication_batch_axis /
256  (multiplication_batch_axis - 1);
257  set_rm(input_rm, i1, rm);
258  set_rv(input_rv, i1, rv);
259 
260  const float stdvar = (float)sqrt(v[i1] + context->eps);
261  // Subtract mean and divide by std, and apply beta and gamma.
262  for (i02 = 0; i02 < multiplication_batch_axis; i02++) {
263  const int i0 = i02 / output_size;
264  const int i2 = i02 % output_size;
265  const int i = i0 * multiplication_axis_output + i1 * output_size + i2;
266  float x = get_x(input_x, i);
267  float beta = get_beta(input_beta, i1);
268  float gamma = get_gamma(input_gamma, i1);
269  float y = (x - m[i1]) * gamma / stdvar + beta;
270  set_output(output, i, y);
271  }
272  }
273 }
