19 rt_function_error_t allocate_pooling(rt_function_t *f,
20  pooling_context_t *context,
21  pooling_private_t *p) {
22  if (p == 0) {
23  return RT_FUNCTION_ERROR_MALLOC;
24  }
25  p->input_shape = clone_list(f->inputs[0]->shape);
26  p->output_shape = clone_list(f->outputs[0]->shape);
27  p->input_n_kernel_size_diff = p->input_shape.size - context->kernel.size;
28  if (context->stride.size == 0) {
29  context->stride = clone_list(context->kernel);
30  } else {
31  if (context->kernel.size != context->stride.size) {
32  return RT_FUNCTION_ERROR_INVALID_SHAPE;
33  }
34  if (context->kernel.size > p->input_shape.size) {
35  return RT_FUNCTION_ERROR_INVALID_SHAPE;
36  }
37  }
38  if (context->kernel.size != 2) {
39  return RT_FUNCTION_ERROR_INVALID_SHAPE;
40  }
41  if (context->kernel.size != context->pad.size) {
42  return RT_FUNCTION_ERROR_INVALID_SHAPE;
43  }
44 
45  // Calc and set output shape.
46  rt_list_t shape = allocate_list(context->kernel.size);
47  int i;
48  for (i = 0; i < context->kernel.size; i++) {
49  shape.data[i] = p->input_shape.data[i + p->input_n_kernel_size_diff];
50  }
51  for (i = 0; i < shape.size; i++) {
52  shape.data[i] += 2 * context->pad.data[i];
53  if (context->ignore_border) {
54  shape.data[i] = (int)((shape.data[i] - context->kernel.data[i]) /
55  context->stride.data[i]) +
56  1;
57  } else {
58  shape.data[i] = (int)ceil(shape.data[i] * 1.0 / context->stride.data[i]);
59  }
60  }
61  for (i = 0; i < p->input_shape.size; i++) {
62  if (i < p->input_n_kernel_size_diff) {
63  p->output_shape.data[i] = p->input_shape.data[i];
64  } else {
65  p->output_shape.data[i] = shape.data[i - p->input_n_kernel_size_diff];
66  }
67  }
68  free_list(shape);
69 
70  // Calc x_stride and y_stride.
71  const rt_list_t input_strides = calc_contiguous_strides(f->inputs[0]->shape);
72  const rt_list_t output_strides =
73  calc_contiguous_strides(f->outputs[0]->shape);
74  p->x_stride = (p->input_n_kernel_size_diff == 0)
75  ? calc_shape_size(f->inputs[0]->shape)
76  : input_strides.data[p->input_n_kernel_size_diff - 1];
77  p->y_stride = (p->input_n_kernel_size_diff == 0)
78  ? calc_shape_size(f->outputs[0]->shape)
79  : output_strides.data[p->input_n_kernel_size_diff - 1];
80  free_list(input_strides);
81  free_list(output_strides);
82 
83  // Init calc_context.
84  p->calc_context.hstart = 0;
85  p->calc_context.hend = 0;
86  p->calc_context.wstart = 0;
87  p->calc_context.wend = 0;
88  p->calc_context.wx = p->input_shape.data[p->input_n_kernel_size_diff + 1];
89  p->calc_context.pool_size = 0;
90  p->calc_context.x = f->inputs[0];
91  p->calc_context.get_x = select_getter(p->calc_context.x);
92  p->calc_context.y = f->outputs[0];
93  p->calc_context.set_y = select_setter(p->calc_context.y);
94  p->calc_context.including_pad = context->including_pad;
95 
96  return RT_FUNCTION_ERROR_NOERROR;
97 }

106 rt_function_error_t exec_pooling(rt_function_t *f, pooling_context_t *context,
107  pooling_private_t *p,
108  exec_pooling_func_t exec) {
109  float *y = (float *)(p->calc_context.y->data);
110  const int hx = p->input_shape.data[p->input_n_kernel_size_diff + 0];
111  const int wx = p->calc_context.wx;
112  const int hy = p->output_shape.data[p->input_n_kernel_size_diff + 0];
113  const int wy = p->output_shape.data[p->input_n_kernel_size_diff + 1];
114  const int hkernel = context->kernel.data[0];
115  const int wkernel = context->kernel.data[1];
116  const int hstride = context->stride.data[0];
117  const int wstride = context->stride.data[1];
118  const int hpad = context->pad.data[0];
119  const int wpad = context->pad.data[1];
120  const int n_map = calc_shape_size(f->inputs[0]->shape) / p->x_stride;
121  int n;
122  p->calc_context.offset_x = 0;
123  p->calc_context.offset_y = 0;
124  for (n = 0; n < n_map; n++) {
125  int iy;
126  for (iy = 0; iy < hy; iy++) {
127  int jy;
128  for (jy = 0; jy < wy; jy++) {
129  int hstart = iy * hstride - hpad;
130  int wstart = jy * wstride - wpad;
131  int hend = (int)fminf((float)(hstart + hkernel), (float)(hx + hpad));
132  int wend = (int)fminf((float)(wstart + wkernel), (float)(wx + wpad));
133  p->calc_context.pool_size = (hend - hstart) * (wend - wstart);
134  p->calc_context.hstart = (int)fmaxf((float)hstart, 0);
135  p->calc_context.wstart = (int)fmaxf((float)wstart, 0);
136  p->calc_context.hend = (int)fminf((float)hend, (float)hx);
137  p->calc_context.wend = (int)fminf((float)wend, (float)wx);
138  int k = iy * wy + jy;
139  float val = exec(p->calc_context);
140  *(y + k + p->calc_context.offset_y) = val;
141  }
142  }
143  p->calc_context.offset_x += p->x_stride;
144  p->calc_context.offset_y += p->y_stride;
145  }
146  return RT_FUNCTION_ERROR_NOERROR;
147 }

201  float val = *(x + jx + calc.offset_x);

203  max_val = val;
204  }
205  }
206  }
