33 rt_function_error_t allocate_round_local_context(rt_function_t *f) {

35  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;

37  if (f->num_of_outputs != 1) {
38  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
39  }
40 
41  round_private_t *p = malloc(sizeof(round_private_t));
42  if (p == 0) {
43  return RT_FUNCTION_ERROR_MALLOC;
44  }

46  f->local_context = (void *)p;
47  p->input = f->inputs[0];
