56 rt_return_value_t rt_initialize_context(rt_context_pointer context,
57  nn_network_t *n) {
58  rt_context_t *c = context;
59  int i, j; // Iterator
60 
61  //////////////////////////////////////////////////////////////////////////////
62  // Binary format version check
63  if (n->version != NN_BINARY_FORMAT_VERSION) {
64  return RT_RET_ERROR_VERSION_UNMATCH;
65  }
66 
67  //////////////////////////////////////////////////////////////////////////////
68  // Buffer list
69  c->num_of_buffers = n->buffers.size;
70  c->buffers = malloc(sizeof(rt_variable_buffer_context_t) * c->num_of_buffers);
71 
72  if (c->buffers == 0) {
73  return RT_RET_ERROR_ALLOCATE_CONTEXT;
74  }
75  for (i = 0; i < c->num_of_buffers; i++) {
76  c->buffers[i].allocate_type = RT_BUFFER_ALLOCATE_TYPE_INITIAL;
77  }
78 
79  //////////////////////////////////////////////////////////////////////////////
80  // Inputs
81  rt_list_t inputs = create_rt_list_from_nn_list(n, n->inputs);
82  c->num_of_inputs = inputs.size;
83  c->input_variable_ids = malloc(sizeof(int *) * c->num_of_inputs);
84  for (i = 0; i < c->num_of_inputs; i++) {
85  c->input_variable_ids[i] = inputs.data[i];
86  }
87 
88  //////////////////////////////////////////////////////////////////////////////
89  // Outputs
90  rt_list_t outputs = create_rt_list_from_nn_list(n, n->outputs);
91  c->num_of_outputs = outputs.size;
92  c->output_variable_ids = malloc(sizeof(int *) * c->num_of_outputs);
93  for (i = 0; i < c->num_of_outputs; i++) {
94  c->output_variable_ids[i] = outputs.data[i];
95  }
96 
97  //////////////////////////////////////////////////////////////////////////////
98  // Allocate buffers
99  int *list = (int *)NN_GET(n, n->buffers.list);
100  for (i = 0; i < c->num_of_buffers; i++) {
101  if (c->buffers[i].allocate_type == RT_BUFFER_ALLOCATE_TYPE_INITIAL) {
102  c->buffers[i].allocate_type = RT_BUFFER_ALLOCATE_TYPE_MALLOC;
103  c->buffers[i].buffer =
104  calloc(*(list + i) * sizeof(float), 1); // TODO float only.
105  if (c->buffers[i].buffer == 0) {
106  return RT_RET_ERROR_ALLOCATE_CONTEXT;
107  }
108  }
109  }
110 
111  //////////////////////////////////////////////////////////////////////////////
112  // Variables
113  c->num_of_variables = n->variables.size;
114  c->variables = malloc(sizeof(rt_variable_t) * c->num_of_variables);
115  if (c->variables == 0) {
116  return RT_RET_ERROR_ALLOCATE_CONTEXT;
117  }
118 
119  list = (int *)NN_GET(n, n->variables.list);
120  for (i = 0; i < c->num_of_variables; i++) {
121  nn_variable_t *var = (nn_variable_t *)(NN_GET(n, *(list + i)));
122  c->variables[i].shape = create_rt_list_from_nn_list(n, var->shape);
123  c->variables[i].type = var->type;
124  c->variables[i].fp_pos = var->fp_pos;
125 
126  if (var->type == NN_DATA_TYPE_INT8 || var->type == NN_DATA_TYPE_INT16) {
127  c->variables[i].coefficient = (1.0 / (1 << var->fp_pos));
128  } else {
129  c->variables[i].coefficient = 0;
130  }
131 
132  if (var->data_index < 0) {
133  int index = (-1 * var->data_index) - 1;
134  if (index >= c->num_of_buffers) {
135  return RT_RET_ERROR_INVALID_BUFFER_INDEX;
136  }
137  c->variables[i].data = c->buffers[index].buffer;
138  } else {
139  c->variables[i].data = NN_GET(n, var->data_index);
140  }
141  }
142 
143  //////////////////////////////////////////////////////////////////////////////
144  // Functions
145  c->num_of_functions = n->functions.size;
146  c->functions = malloc(sizeof(rt_function_context_t) * c->num_of_functions);
147  list = (int *)NN_GET(n, n->functions.list);
148  for (i = 0; i < c->num_of_functions; i++) {
149  nn_function_t *func = (nn_function_t *)(NN_GET(n, *(list + i)));
150  c->functions[i] = allocate_function_io(n, c, func);
151 
152  int callback_registered_flag = 0;
153  if (func->impl <= NN_END_OF_USER_DEFINED_FUNCTION_IMPLEMENT) {
154  for (j = 0; j < c->num_of_callbacks; j++) {
155  if ((c->callbacks + j)->type == func->type) {
156  rt_return_value_t ret =
157  (c->callbacks +
158  j)->allocate_local_context(n, (void *)(&(c->functions[i])));
159  if (ret == RT_RET_FUNCTION_MATCH) {
160  callback_registered_flag = 1;
161  break;
162  }
163  }
164  }
165  }
166  if (!callback_registered_flag) {
167  allocate_function_context(n, func, c->functions + i);
168  }
169  }
170 
171  c->network = n;
172 
173  return RT_RET_NOERROR;
174 }

176 rt_return_value_t rt_free_context(rt_context_pointer *context) {
177  rt_context_t *c = *context;
178 
179  int i; // Iterator
180 
181  // Buffers
182  for (i = 0; i < c->num_of_buffers; i++) {
183  if (c->buffers[i].allocate_type == RT_BUFFER_ALLOCATE_TYPE_MALLOC) {
184  free(c->buffers[i].buffer);
185  }
186  }
187  free(c->buffers);
188 
189  // Variables
190  free(c->variables);
191 
192  // Functions
193  for (i = 0; i < c->num_of_functions; i++) {
194  free(c->functions[i].func.inputs);
195  free(c->functions[i].func.outputs);
196 
197  c->functions[i].func.free_local_context_func(&(c->functions[i].func));
198  if (c->functions[i].func.local_context != 0) {
199  free(c->functions[i].func.local_context);
200  }
201  }
202  free(c->functions);
203 
204  free(c->input_variable_ids);
205  free(c->output_variable_ids);
206 
207  // Callback
208  if (c->callbacks) {
209  free(c->callbacks);
210  }
211 
212  free(*context);
213  return RT_RET_NOERROR;
214 }
