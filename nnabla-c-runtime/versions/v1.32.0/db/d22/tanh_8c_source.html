38 rt_function_error_t allocate_tanh_local_context(rt_function_t *f) {
39  if (f->num_of_inputs != 1) {
40  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
41  }
42  if (f->num_of_outputs != 1) {
43  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
44  }
45  tanh_local_context_t *c = rt_malloc_func(sizeof(tanh_local_context_t));
46  if (c == 0) {
47  return RT_FUNCTION_ERROR_MALLOC;
48  }
49  f->local_context = c;
50  c->input = f->inputs[0];
51  c->get_input = select_getter(c->input);
52  c->input_size = calc_shape_size(f->inputs[0]->shape);
53  c->output = f->outputs[0];
54  c->set_output = select_setter(c->output);
55  c->output_size = calc_shape_size(f->outputs[0]->shape);
56  if (c->input_size != c->output_size) {
57  return RT_FUNCTION_ERROR_INVALID_SHAPE;
58  }
59  if (c->input->type == NN_DATA_TYPE_FLOAT &&
60  c->output->type == NN_DATA_TYPE_FLOAT) {
61 #ifdef CONFIG_TANH_FLOAT32
62  f->exec_func = exec_tanh;
63 #endif /* CONFIG_TANH_FLOAT32 */
64  } else {
65 #ifdef CONFIG_TANH_GENERIC
66  f->exec_func = exec_tanh_generic;
67 #endif /* CONFIG_TANH_GENERIC */
68  }
69  return RT_FUNCTION_ERROR_NOERROR;
70 }

94  int i; // Iterator
