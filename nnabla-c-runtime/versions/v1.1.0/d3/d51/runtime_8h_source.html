168 /// @brief Initialize runtime context with parsing @ref nn_network_t.
169 /// Initialize all functions in context and prepare forward calculation.

171 ///
172 /// Callback selection rule.
173 /// @startuml
174 /// skinparam monochrome true
175 /// start
176 ///
177 /// if (impl <= NN_END_OF_SYSTEM_DEFINED_FUNCTION_IMPLEMENT) then (yes)
178 /// if (Found callback->type is same as func->type) then (yes)
179 /// :Call user defined allocator;
180 /// if (return value == RT_RET_FUNCTION_MATCH) then (yes)
181 /// stop
182 /// else (no)
183 /// :Call default allocator;
184 /// stop
185 /// endif
186 /// else (no)
187 /// :Call default allocator;
188 /// stop

193 /// endif

195 /// @enduml
