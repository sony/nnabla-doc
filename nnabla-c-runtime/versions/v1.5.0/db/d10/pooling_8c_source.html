19 rt_function_error_t allocate_pooling(rt_function_t *f,
20  pooling_context_t *context,
21  pooling_private_t *p) {
22  if (p == 0) {
23  return RT_FUNCTION_ERROR_MALLOC;
24  }
25  p->input_shape = clone_list(f->inputs[0]->shape);
26  p->output_shape = clone_list(f->outputs[0]->shape);
27  p->input_n_kernel_size_diff = p->input_shape.size - context->kernel.size;
28  if (context->stride.size == 0) {
29  context->stride = clone_list(context->kernel);
30  } else {
31  if (context->kernel.size != context->stride.size) {
32  return RT_FUNCTION_ERROR_INVALID_SHAPE;
33  }
34  if (context->kernel.size > p->input_shape.size) {
35  return RT_FUNCTION_ERROR_INVALID_SHAPE;
36  }
37  }
38  if (context->kernel.size != 2 && context->kernel.size != 3) {
39  return RT_FUNCTION_ERROR_INVALID_SHAPE;
40  }
41  if (context->kernel.size != context->pad.size) {
42  return RT_FUNCTION_ERROR_INVALID_SHAPE;
43  }
44 
45  // Calc and set output shape.
46  rt_list_t shape = allocate_list(context->kernel.size);
47  int i;
48  for (i = 0; i < shape.size; i++) {
49  int _w = p->input_shape.data[i + p->input_n_kernel_size_diff];
50  int _k = context->kernel.data[i];
51  int _s = context->stride.data[i];
52  int _p = context->pad.data[i];
53  shape.data[i] = (_w + _p - (context->ignore_border ? _k - _p : 1)) / _s + 1;
54  }
55  for (i = 0; i < p->input_shape.size; i++) {
56  if (i < p->input_n_kernel_size_diff) {
57  p->output_shape.data[i] = p->input_shape.data[i];
58  } else {
59  p->output_shape.data[i] = shape.data[i - p->input_n_kernel_size_diff];
60  }
61  }
62  free_list(shape);
63 
64  // Calc x_map_size and y_map_size.
65  p->input_strides = calc_contiguous_strides(f->inputs[0]->shape);
66  p->output_strides = calc_contiguous_strides(f->outputs[0]->shape);
67  p->x_map_size = (p->input_n_kernel_size_diff == 0)
68  ? calc_shape_size(f->inputs[0]->shape)
69  : p->input_strides.data[p->input_n_kernel_size_diff - 1];
70  p->y_map_size = (p->input_n_kernel_size_diff == 0)
71  ? calc_shape_size(f->outputs[0]->shape)
72  : p->output_strides.data[p->input_n_kernel_size_diff - 1];
73 
74  // Init calc_context.
75  p->calc_context.hstart = 0;
76  p->calc_context.hend = 0;
77  p->calc_context.wstart = 0;
78  p->calc_context.wend = 0;
79  p->calc_context.dstart = 0;
80  p->calc_context.dend = 0;
81  p->calc_context.pool_size = 0;
82  p->calc_context.x = f->inputs[0];
83  p->calc_context.get_x = select_getter(p->calc_context.x);
84  p->calc_context.y = f->outputs[0];
85  p->calc_context.set_y = select_setter(p->calc_context.y);
86  p->calc_context.including_pad = context->including_pad;
87 
88  return RT_FUNCTION_ERROR_NOERROR;
89 }

100 rt_function_error_t exec_pooling(rt_function_t *f, pooling_context_t *context,
101  pooling_private_t *p,
102  exec_pooling_func_t exec) {
103  float *y = (float *)(p->calc_context.y->data);
104  const int hx = p->input_shape.data[p->input_n_kernel_size_diff + 0];
105  const int wx = p->input_shape.data[p->input_n_kernel_size_diff + 1];
106  const int hy = p->output_shape.data[p->input_n_kernel_size_diff + 0];
107  const int wy = p->output_shape.data[p->input_n_kernel_size_diff + 1];
108  const int hkernel = context->kernel.data[0];
109  const int wkernel = context->kernel.data[1];
110  const int hstride = context->stride.data[0];
111  const int wstride = context->stride.data[1];
112  const int hpad = context->pad.data[0];
113  const int wpad = context->pad.data[1];
114  const int n_map = calc_shape_size(f->inputs[0]->shape) / p->x_map_size;
115  p->calc_context.offset_x = 0;
116  p->calc_context.offset_y = 0;
117  p->calc_context.kernel_size = context->kernel.size;
118 
119  if (context->kernel.size == 2) {
120  for (int n = 0; n < n_map; n++) {
121  for (int iy = 0; iy < hy; iy++) {
122  for (int jy = 0; jy < wy; jy++) {
123  int hstart = iy * hstride - hpad;
124  int wstart = jy * wstride - wpad;
125  int hend = (int)fminf((float)(hstart + hkernel), (float)(hx + hpad));
126  int wend = (int)fminf((float)(wstart + wkernel), (float)(wx + wpad));
127  p->calc_context.pool_size = (hend - hstart) * (wend - wstart);
128  p->calc_context.hstart = (int)fmaxf((float)hstart, 0);
129  p->calc_context.wstart = (int)fmaxf((float)wstart, 0);
130  p->calc_context.hend = (int)fminf((float)hend, (float)hx);
131  p->calc_context.wend = (int)fminf((float)wend, (float)wx);
132  p->calc_context.hstride =
133  p->input_strides.data[p->input_n_kernel_size_diff + 0];
134  int k =
135  iy * p->output_strides.data[p->input_n_kernel_size_diff + 0] + jy;
136  float val = exec(p->calc_context);
137  *(y + k + p->calc_context.offset_y) = val;
138  }
139  }
140  p->calc_context.offset_x += p->x_map_size;
141  p->calc_context.offset_y += p->y_map_size;
142  }
143  } else if (context->kernel.size == 3) {
144  const int dx = p->input_shape.data[p->input_n_kernel_size_diff + 2];
145  const int dy = p->output_shape.data[p->input_n_kernel_size_diff + 2];
146  const int dkernel = context->kernel.data[2];
147  const int dstride = context->stride.data[2];
148  const int dpad = context->pad.data[2];
149 
150  for (int n = 0; n < n_map; n++) {
151  for (int iy = 0; iy < hy; iy++) {
152  for (int jy = 0; jy < wy; jy++) {
153  for (int ky = 0; ky < dy; ky++) {
154  int hstart = iy * hstride - hpad;
155  int wstart = jy * wstride - wpad;
156  int dstart = ky * dstride - dpad;
157  int hend =
158  (int)fminf((float)(hstart + hkernel), (float)(hx + hpad));
159  int wend =
160  (int)fminf((float)(wstart + wkernel), (float)(wx + wpad));
161  int dend =
162  (int)fminf((float)(dstart + dkernel), (float)(dx + dpad));
163  p->calc_context.pool_size =
164  (hend - hstart) * (wend - wstart) * (dend - dstart);
165  p->calc_context.hstart = (int)fmaxf((float)hstart, 0);
166  p->calc_context.wstart = (int)fmaxf((float)wstart, 0);
167  p->calc_context.dstart = (int)fmaxf((float)dstart, 0);
168  p->calc_context.hend = (int)fminf((float)hend, (float)hx);
169  p->calc_context.wend = (int)fminf((float)wend, (float)wx);
170  p->calc_context.dend = (int)fminf((float)dend, (float)dx);
171  p->calc_context.hstride =
172  p->input_strides.data[p->input_n_kernel_size_diff + 0];
173  p->calc_context.wstride =
174  p->input_strides.data[p->input_n_kernel_size_diff + 1];
175  int k =
176  iy * p->output_strides.data[p->input_n_kernel_size_diff + 0] +
177  jy * p->output_strides.data[p->input_n_kernel_size_diff + 1] +
178  ky;
179  float val = exec(p->calc_context);
180  *(y + k + p->calc_context.offset_y) = val;
181  }
182  }
183  }
184  p->calc_context.offset_x += p->x_map_size;
185  p->calc_context.offset_y += p->y_map_size;
186  }
187  }
188  return RT_FUNCTION_ERROR_NOERROR;

286  float *x = (float *)(calc.x->data);
287  if (calc.kernel_size == 2) {
288  int l = calc.hstart * calc.hstride + calc.wstart;
289  max_val = *(x + l + calc.offset_x);
290  for (int ix = calc.hstart; ix < calc.hend; ix++) {
291  for (int jx = ix * calc.hstride + calc.wstart;
292  jx < ix * calc.hstride + calc.wend; jx++) {
293  float val = *(x + jx + calc.offset_x);
294  if (max_val < val) {
295  max_val = val;
296  }
297  }
298  }
299  } else if (calc.kernel_size == 3) {
