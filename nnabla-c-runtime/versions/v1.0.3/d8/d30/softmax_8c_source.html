31 rt_function_error_t allocate_softmax_local_context(rt_function_t *f) {
32  softmax_local_context_t *context =
33  (softmax_local_context_t *)(f->local_context);
34  softmax_private_t *p = malloc(sizeof(softmax_private_t));
35  if (p == 0) {
36  return RT_FUNCTION_ERROR_MALLOC;
37  }
38  const int axis = context->axis;
39  const int size = calc_shape_size(f->inputs[0]->shape);
40  const int size_axis =
41  axis <= 0 ? size : shape_product_of(f->inputs[0], axis,
42  f->inputs[0]->shape.size);
43 
44  // axis must be less than ndim of inputs[0].
45  if (f->inputs[0]->shape.size <= axis) {
46  return RT_FUNCTION_ERROR_INVALID_SHAPE;
47  }
48  p->batch_size = size / size_axis;
49  p->specified_axis_size = f->inputs[0]->shape.data[axis];
50  p->output_size = size / p->batch_size / p->specified_axis_size;
51  if (p->batch_size * p->specified_axis_size * p->output_size != size) {
52  return RT_FUNCTION_ERROR_INVALID_SHAPE;
53  }
54  ((softmax_local_context_t *)(f->local_context))->data = (void *)p;
55  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&
56  f->outputs[0]->type == NN_DATA_TYPE_FLOAT) {
57  f->exec_func = exec_softmax;
58  } else {
59  f->exec_func = exec_softmax_generic;
60  }
61  return RT_FUNCTION_ERROR_NOERROR;
62 }

69 rt_function_error_t exec_softmax(rt_function_t *f) {
70  softmax_local_context_t *context =
71  (softmax_local_context_t *)(f->local_context);
72  softmax_private_t *p = (softmax_private_t *)(context->data);
73  const float *const x = (float *)(f->inputs[0]->data);
74  float *const y = (float *)(f->outputs[0]->data);
75  const int batch_size = p->batch_size;
76  const int specified_axis_size = p->specified_axis_size;
77  const int output_size = p->output_size;
78 
79  int sample_index;
80  for (sample_index = 0; sample_index < batch_size; ++sample_index) {
81  int output_index;
82  for (output_index = 0; output_index < output_size; ++output_index) {
83  const int j =
84  sample_index * specified_axis_size * output_size + output_index;
85  // compute maximum
86  float max_input = x[j];
87  int specified_index;
88  for (specified_index = 0; specified_index < specified_axis_size;
89  ++specified_index) {
90  const int k = specified_index * output_size + j;
91  max_input = local_max(max_input, x[k]);
92  }
93  // Compute exponential and sum
94  float exp_sum = 0;
95  for (specified_index = 0; specified_index < specified_axis_size;
96  ++specified_index) {
97  const int k = specified_index * output_size + j;
98  const float tmp = expf(x[k] - max_input);
99  y[k] = tmp;
100  exp_sum += tmp;
101  }
102  // Compute softmax
103  for (specified_index = 0; specified_index < specified_axis_size;
104  ++specified_index) {
105  const int k = specified_index * output_size + j;
106  y[k] = y[k] / exp_sum;
107  }
108  }
109  }
110  return RT_FUNCTION_ERROR_NOERROR;
111 }

113 rt_function_error_t exec_softmax_generic(rt_function_t *f) {
114  softmax_local_context_t *context =
115  (softmax_local_context_t *)(f->local_context);
116  softmax_private_t *p = (softmax_private_t *)(context->data);
117  rt_variable_t *input = f->inputs[0];
118  rt_variable_getter get_input = select_getter(input);
119  rt_variable_t *output = f->outputs[0];
120  rt_variable_getter get_output = select_getter(output);
121  rt_variable_setter set_output = select_setter(output);
122  const int batch_size = p->batch_size;
123  const int specified_axis_size = p->specified_axis_size;
124  const int output_size = p->output_size;
125 
126  int sample_index;
127  for (sample_index = 0; sample_index < batch_size; ++sample_index) {
128  int output_index;
129  for (output_index = 0; output_index < output_size; ++output_index) {
130  const int j =
131  sample_index * specified_axis_size * output_size + output_index;
132  // compute maximum
133  float max_input = get_input(input, j);
134  int specified_index;
135  for (specified_index = 0; specified_index < specified_axis_size;
136  ++specified_index) {
137  const int k = specified_index * output_size + j;
138  max_input = local_max(max_input, get_input(input, k));
139  }
140  // Compute exponential and sum
141  float exp_sum = 0;
142  for (specified_index = 0; specified_index < specified_axis_size;
143  ++specified_index) {
144  const int k = specified_index * output_size + j;
145  const float tmp = expf(get_input(input, k) - max_input);
146  set_output(output, k, tmp);
147  exp_sum += tmp;
148  }
149  // Compute softmax
150  for (specified_index = 0; specified_index < specified_axis_size;
151  ++specified_index) {
152  const int k = specified_index * output_size + j;
153  set_output(output, k, (get_output(output, k) / exp_sum));
154  }
155  }
156  }
157  return RT_FUNCTION_ERROR_NOERROR;
158 }
