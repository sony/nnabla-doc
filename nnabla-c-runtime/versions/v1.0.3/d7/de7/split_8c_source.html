70 rt_function_error_t exec_split(rt_function_t *f) {

72  split_private_t *p = (split_private_t *)(c->data);
73 
74  const float *x = (float *)(f->inputs[0]->data);
75  for (int i = 0; i < p->num_outputs; i++) {
