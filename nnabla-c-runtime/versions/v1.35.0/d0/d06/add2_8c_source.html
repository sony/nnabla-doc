28 rt_function_error_t allocate_add2_local_context(rt_function_t *f) {
29  if (f->num_of_inputs != 2) {
30  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
31  }
32  if (f->num_of_outputs != 1) {
33  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
34  }
35  if (f->inputs[0]->shape.size != f->inputs[1]->shape.size) {
36  return RT_FUNCTION_ERROR_INVALID_SHAPE;
37  }
38  if (f->outputs[0]->shape.size != f->inputs[0]->shape.size) {
39  return RT_FUNCTION_ERROR_INVALID_SHAPE;
40  }
41 
42  if (f->inputs[0]->type == NN_DATA_TYPE_FLOAT &&
43  f->inputs[1]->type == NN_DATA_TYPE_FLOAT &&
44  f->outputs[0]->type == NN_DATA_TYPE_FLOAT) {
45 #ifdef CONFIG_ADD2_FLOAT32
46  f->exec_func = exec_add2;
47 #endif /* CONFIG_ADD2_FLOAT32 */
48  }
49 
50  else if (f->inputs[0]->type == NN_DATA_TYPE_INT16 &&
51  f->inputs[1]->type == NN_DATA_TYPE_INT16 &&
52  f->outputs[0]->type == NN_DATA_TYPE_INT16) {
53 #ifdef CONFIG_ADD2_FIXED16
54  f->exec_func = exec_add2_fixed16;
55 #endif /* CONFIG_ADD2_FIXED16 */
56  }
57 
58  else if (f->inputs[0]->type == NN_DATA_TYPE_INT8 &&
59  f->inputs[1]->type == NN_DATA_TYPE_INT8 &&
60  f->outputs[0]->type == NN_DATA_TYPE_INT8) {
61 #ifdef CONFIG_ADD2_FIXED8
62  f->exec_func = exec_add2_fixed8;
63 #endif /* CONFIG_ADD2_FIXED8 */
64  }
65 
66  else {
67 #ifdef CONFIG_ADD2_GENERIC
68  f->exec_func = exec_add2_generic;
69 #endif /* CONFIG_ADD2_GENERIC */
70  }
71 
72  return RT_FUNCTION_ERROR_NOERROR;
73 }
