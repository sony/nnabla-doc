48 rt_function_error_t allocate_deconvolution_local_context(rt_function_t *f) {
49  deconvolution_local_context_t *c =
50  (deconvolution_local_context_t *)(f->local_context);
51 
52  if (f->num_of_inputs != 2 && f->num_of_inputs != 3) {
53  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
54  }
55 
56  if (f->num_of_outputs != 1) {
57  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
58  }
59 
60  deconvolution_private_t *p = malloc(sizeof(deconvolution_private_t));
61  if (p == 0) {
62  return RT_FUNCTION_ERROR_MALLOC;
63  }
64  c->data = (void *)p;
65  p->input = f->inputs[0];
66  p->get_input = select_getter(p->input);
67 
68  p->weight = f->inputs[1];
69  p->get_weight = select_getter(p->weight);
70 
71  p->output = f->outputs[0];
72  p->get_output = select_getter(p->output);
73  p->set_output = select_setter(p->output);
74 
75  if (f->num_of_inputs > 2) {
76  p->bias = f->inputs[2];
77  p->get_bias = select_getter(p->bias);
78  } else {
79  p->bias = 0;
80  }
81 
82  p->base_loop_size = 1;
83  for (int i = 0; i < c->base_axis; i++) {
84  p->base_loop_size *= p->input->shape.data[i];
85  }
86  p->spatial_dims = p->input->shape.size - c->base_axis - 1;
87 
88  p->input_shape = allocate_list(p->spatial_dims);
89  p->kernel_shape = allocate_list(p->spatial_dims);
90  p->output_shape = allocate_list(p->spatial_dims);
91  p->in_position = allocate_list(p->spatial_dims);
92  p->out_position = allocate_list(p->spatial_dims);
93 
94  for (int i = 0; i < p->spatial_dims; i++) {
95  p->kernel_shape.data[i] = p->weight->shape.data[i + 2];
96  p->input_shape.data[i] = p->input->shape.data[i + c->base_axis + 1];
97  p->output_shape.data[i] = p->output->shape.data[i + c->base_axis + 1];
98  }
99 
100  if (p->input->type == NN_DATA_TYPE_FLOAT &&
101  p->output->type == NN_DATA_TYPE_FLOAT &&
102  p->weight->type == NN_DATA_TYPE_FLOAT &&
103  ((p->bias && p->bias->type == NN_DATA_TYPE_FLOAT) || !p->bias)) {
104  f->exec_func = exec_deconvolution;
105  } else {
106  f->exec_func = exec_deconvolution_generic;
107  }
108  return RT_FUNCTION_ERROR_NOERROR;
109 }

124 rt_function_error_t exec_deconvolution(rt_function_t *f) {
125  deconvolution_local_context_t *c =
126  (deconvolution_local_context_t *)(f->local_context);
127  deconvolution_private_t *p = (deconvolution_private_t *)(c->data);
128 
129  int output_size = calc_shape_size(p->output_shape);
130  int kernel_size = calc_shape_size(p->kernel_shape);
131  int input_size = calc_shape_size(p->input_shape);
132 
133  memset(p->output->data, 0, sizeof(float) * calc_shape_size(p->output->shape));
134 
135  for (int b = 0; b < p->base_loop_size; b++) {
136  for (int g = 0; g < c->group; g++) {
137  for (int om = 0; om < p->weight->shape.data[1]; om++) {
138  int output_offset = (b * c->group * p->weight->shape.data[1] +
139  g * p->weight->shape.data[1] + om) *
140  output_size;
141  for (int im = 0; im < p->weight->shape.data[0] / c->group; im++) {
142  int input_offset = (b * p->weight->shape.data[0] +
143  g * p->weight->shape.data[0] / c->group + im) *
144  input_size;
145  int kernel_offset = (g * p->weight->shape.data[1] *
146  p->weight->shape.data[0] / c->group +
147  im * p->weight->shape.data[1] + om) *
148  kernel_size;
149  for (int o = 0; o < output_size; o++) {
150  float sum = 0.0f;
151  pos_to_shape(p->out_position, p->output_shape, o);
152  for (int k = 0; k < kernel_size; k++) {
153  pos_to_shape(p->in_position, p->kernel_shape, k);
154  uint8_t condition = 1;
155  for (int j = 0; j < p->spatial_dims; j++) {
156  int k1 =
157  c->dilation.data[j] * (p->kernel_shape.data[j] - 1) + 1;
158  p->in_position.data[j] *= c->dilation.data[j];
159  p->in_position.data[j] -= k1 - c->pad.data[j] - 1;
160  p->in_position.data[j] += p->out_position.data[j];
161  if (p->in_position.data[j] % c->stride.data[j] != 0) {
162  condition = 0;
163  break;
164  }
165  p->in_position.data[j] =
166  p->in_position.data[j] / c->stride.data[j];
167  if (p->in_position.data[j] < 0 ||
168  p->in_position.data[j] >= p->input_shape.data[j]) {
169  condition = 0;
170  break;
171  }
172  }
173  if (condition) {
174  float x = *((float *)(p->input->data) + input_offset +
175  shape_to_pos(p->input_shape, p->in_position));
176  float w = *((float *)(p->weight->data) + kernel_offset +
177  (kernel_size - k - 1));
178  sum += x * w;
179  }
180  }
181  *((float *)(p->output->data) + output_offset + o) += sum;
182  }
183  }
184  if (p->bias) {
185  for (int o = 0; o < output_size; o++) {
186  *((float *)(p->output->data) + output_offset + o) +=
187  *((float *)(p->bias->data) + g * p->weight->shape.data[1] + om);
188  }
189  }
190  }
191  }
192  }
193 
194  return RT_FUNCTION_ERROR_NOERROR;
195 }
