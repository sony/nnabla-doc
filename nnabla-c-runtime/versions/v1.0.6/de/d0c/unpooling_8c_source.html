35 rt_function_error_t allocate_unpooling_local_context(rt_function_t *f) {
36  unpooling_local_context_t *context =
37  (unpooling_local_context_t *)(f->local_context);
38  unpooling_private_t *p = malloc(sizeof(unpooling_private_t));
39  if (p == 0) {
40  return RT_FUNCTION_ERROR_MALLOC;
41  }
42  p->input_shape = clone_list(f->inputs[0]->shape);
43  p->output_shape = clone_list(f->outputs[0]->shape);
44  p->input_strides = calc_contiguous_strides(f->inputs[0]->shape);
45  p->output_strides = calc_contiguous_strides(f->outputs[0]->shape);
46  p->input = f->inputs[0];
47  p->get_input = select_getter(p->input);
48  p->output = f->outputs[0];
49  p->set_output = select_setter(p->output);
50 
51  if (context->kernel.size > p->input_shape.size) {
52  return RT_FUNCTION_ERROR_INVALID_SHAPE;
53  }
54 
55  // Calc and set output shape.
56  rt_list_t shape = allocate_list(p->input_shape.size);
57  int diff = p->input_shape.size - context->kernel.size;
58  int i;
59  for (i = 0; i < diff; i++) {
60  shape.data[i] = 1;
61  }
62  for (i = diff; i < p->input_shape.size; i++) {
63  shape.data[i] = context->kernel.data[i - diff];
64  }
65  p->kernel = clone_list(shape);
66  for (i = 0; i < p->input_shape.size; i++) {
67  p->output_shape.data[i] = p->input_shape.data[i] * p->kernel.data[i];
68  }
69  free_list(shape);
70 
71  ((unpooling_local_context_t *)(f->local_context))->data = (void *)p;
72 
73  return RT_FUNCTION_ERROR_NOERROR;
74 }

89 static void unpooling_forward_recursive_generic(unpooling_private_t *p,
90  int x_offset, int y_offset,
91  int dim) {
92  int current_x_offset = x_offset;
93  int current_y_offset = y_offset;
94  const int x_stride = p->input_strides.data[dim];
95  const int y_stride = p->output_strides.data[dim];
96  const int kernel = p->kernel.data[dim];
97  const int size = p->output_shape.data[dim];
98 
99  if (dim == p->input_shape.size - 1) {
100  if (x_stride == 1 && kernel == 1) {
101  for (int i = 0; i < size; i++) {
102  float value = p->get_input(p->input, i + current_x_offset);
103  p->set_output(p->output, i + current_y_offset, value);
104  }
105  } else {
106  int count = 0;
107  for (int i = 0; i < size; i++) {
108  float value = p->get_input(p->input, current_x_offset);
109  p->set_output(p->output, current_y_offset, value);
110  if (++count >= kernel) {
111  count = 0;
112  current_x_offset += x_stride;
113  }
114  current_y_offset += y_stride;
115  }
116  }
117  } else {
118  int count = 0;
119  int i;
120  for (i = 0; i < size; i++) {
121  unpooling_forward_recursive_generic(p, current_x_offset, current_y_offset,
122  dim + 1);
123  if (++count >= kernel) {
124  count = 0;
125  current_x_offset += x_stride;
126  }
127  current_y_offset += y_stride;
128  }
129  }
130 }

132 static void unpooling_forward_recursive(unpooling_private_t *p, int x_offset,
133  int y_offset, int dim) {
134  int current_x_offset = x_offset;
135  int current_y_offset = y_offset;
136  const int x_stride = p->input_strides.data[dim];
137  const int y_stride = p->output_strides.data[dim];
138  const int kernel = p->kernel.data[dim];
139  const int size = p->output_shape.data[dim];
140  const float *x = (float *)(p->input->data);
141  float *y = (float *)(p->output->data);
142 
143  if (dim == p->input_shape.size - 1) {
144  const float *current_x = x + current_x_offset;
145  float *current_y = y + current_y_offset;
146  if (x_stride == 1 && kernel == 1) {
147  memcpy(current_y, current_x, sizeof(float) * size);
148  } else {
149  const float *end_y = current_y + size * y_stride;
150  int count = 0;
151  while (current_y != end_y) {
152  *current_y = *current_x;
153  if (++count >= kernel) {
154  count = 0;
155  current_x += x_stride;
156  }
157  current_y += y_stride;
158  }
159  }
160  } else {
161  int count = 0;
162  int i;
163  for (i = 0; i < size; i++) {
164  unpooling_forward_recursive(p, current_x_offset, current_y_offset,
165  dim + 1);
166  if (++count >= kernel) {
167  count = 0;
168  current_x_offset += x_stride;
169  }
170  current_y_offset += y_stride;
171  }
172  }
173 }
