44 rt_function_error_t allocate_batch_matmul_local_context(rt_function_t *f) {
45  if (f->num_of_inputs != 2) {
46  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
47  }
48  if (f->num_of_outputs != 1) {
49  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
50  }
51 
52  batch_matmul_local_context_t *context =
53  (batch_matmul_local_context_t *)(f->local_context);
54  batch_matmul_private_t *p = malloc(sizeof(batch_matmul_private_t));
55  if (p == 0) {
56  return RT_FUNCTION_ERROR_MALLOC;
57  }
58  ((batch_matmul_local_context_t *)(f->local_context))->data = (void *)p;
59  p->row_a = f->inputs[0]->shape.data[f->inputs[0]->shape.size - 2];
60  p->col_a = f->inputs[0]->shape.data[f->inputs[0]->shape.size - 1];
61  p->row_b = f->inputs[1]->shape.data[f->inputs[1]->shape.size - 2];
62  p->col_b = f->inputs[1]->shape.data[f->inputs[1]->shape.size - 1];
63  p->row_y = context->transpose_a ? p->col_a : p->row_a;
64  p->col_y = context->transpose_b ? p->row_b : p->col_b;
65  p->offset_a = p->row_a * p->col_a;
66  p->offset_b = p->row_b * p->col_b;
67  p->offset_y = p->row_y * p->col_y;
68 
69  p->input_a = f->inputs[0];
70  p->get_input_a = select_getter(p->input_a);
71  p->input_b = f->inputs[1];
72  p->get_input_b = select_getter(p->input_b);
73 
74  p->samples = 1;
75  for (int i = 0; i < f->inputs[0]->shape.size - 2; ++i) {
76  p->samples *= f->inputs[0]->shape.data[i];
77  }
78 
79  int samples_b = 1;
80  for (int j = 0; j < f->inputs[1]->shape.size - 2; ++j) {
81  samples_b *= f->inputs[1]->shape.data[j];
82  }
83 
84  p->output = f->outputs[0];
85  p->get_output = select_getter(p->output);
86  p->set_output = select_setter(p->output);
87  p->output_size = calc_shape_size(f->outputs[0]->shape);
88 
89  if (p->output_size != p->row_y * p->col_y || p->col_a != p->row_b ||
90  p->samples != samples_b) {
91  return RT_FUNCTION_ERROR_INVALID_SHAPE;
92  }
93  if (p->input_a->type == NN_DATA_TYPE_FLOAT &&
94  p->input_b->type == NN_DATA_TYPE_FLOAT &&
95  p->output->type == NN_DATA_TYPE_FLOAT) {
96  f->exec_func = exec_batch_matmul;
97  } else {
98  f->exec_func = exec_batch_matmul_generic;
99  }
100  return RT_FUNCTION_ERROR_NOERROR;
101 }

112 static void move_data(float *mtx, int i, int m, int n) {
113  float temp = mtx[i];
114  int cur = i;
115  int pre = get_pre(cur, m, n);
116  while (pre != i) {
117  mtx[cur] = mtx[pre];
118  cur = pre;
119  pre = get_pre(cur, m, n);
120  }
121  mtx[cur] = temp;
122 }

124 static void transpose(float *mtx, int m, int n) {
125  for (int i = 0; i < m * n; i++) {
126  int next = get_next(i, m, n);
127  while (next > i)
128  next = get_next(next, m, n);
129  if (next == i)
130  move_data(mtx, i, m, n);
131  }
132 }

134 rt_function_error_t exec_batch_matmul(rt_function_t *f) {
135  batch_matmul_local_context_t *context =
136  (batch_matmul_local_context_t *)(f->local_context);
137  batch_matmul_private_t *p = (batch_matmul_private_t *)(context->data);
138  float *input_a = (float *)(p->input_a->data);
139  float *input_b = (float *)(p->input_b->data);
140  float *output = (float *)(p->output->data);
141 
142  int i;
143  if (context->transpose_a) {
144  for (i = 0; i < p->samples; i++) {
145  transpose(input_a + p->offset_a * i, p->row_a, p->col_a);
146  }
147  p->row_a = p->row_a ^ p->col_a;
148  p->col_a = p->row_a ^ p->col_a;
149  p->row_a = p->row_a ^ p->col_a;
150  }
151  if (context->transpose_b) {
152  for (i = 0; i < p->samples; i++) {
153  transpose(input_b + p->offset_b * i, p->row_b, p->col_b);
154  }
155  p->row_b = p->row_b ^ p->col_b;
156  p->col_b = p->row_b ^ p->col_b;
157  p->row_b = p->row_b ^ p->col_b;
158  }
159  for (i = 0; i < p->samples; i++) {
160  float *mtx_y = output + p->offset_y * i;
161  float *mtx_a = input_a + p->offset_a * i;
162  float *mtx_b = input_b + p->offset_b * i;
163  int row_a = p->row_a;
164  int col_b = p->col_b;
165  int col_a = p->col_a;
166  for (int j = 0; j < row_a; j++) {
167  for (int k = 0; k < col_b; k++) {
168  for (int l = 0; l < col_a; l++) {
169  float a = *(mtx_a + col_a * j + l);
170  float b = *(mtx_b + col_b * l + k);
171  *(mtx_y + col_b * j + k) += a * b;
172  }
173  }
174  }
175  }
176 
177  return RT_FUNCTION_ERROR_NOERROR;
178 }

181  batch_matmul_local_context_t *context =
182  (batch_matmul_local_context_t *)(f->local_context);
183  batch_matmul_private_t *p = (batch_matmul_private_t *)(context->data);

185  float *input_b = (float *)(p->input_b->data);
186 
187  int i;

189  for (i = 0; i < p->samples; i++) {
190  transpose(input_a + p->offset_a * i, p->row_a, p->col_a);
191  }
192  p->row_a = p->row_a ^ p->col_a;
193  p->col_a = p->row_a ^ p->col_a;
194  p->row_a = p->row_a ^ p->col_a;
195  }
196  if (context->transpose_b) {
197  for (i = 0; i < p->samples; i++) {
198  transpose(input_b + p->offset_b * i, p->row_b, p->col_b);
199  }
200  p->row_b = p->row_b ^ p->col_b;
201  p->col_b = p->row_b ^ p->col_b;
202  p->row_b = p->row_b ^ p->col_b;
203  }
204  for (i = 0; i < p->samples; i++) {
205  int row_a = p->row_a;
206  int col_b = p->col_b;
207  int col_a = p->col_a;
208  for (int j = 0; j < row_a; j++) {
209  for (int k = 0; k < col_b; k++) {
210  for (int l = 0; l < col_a; l++) {
211  float a = p->get_input_a(p->input_a, p->offset_y * i + col_a * j + l);
212  float b = p->get_input_b(p->input_b, p->offset_b * i + col_b * l + k);
213  float y = p->get_output(p->output, p->offset_y * i + col_b * j + k);
214  y += a * b;
215  p->set_output(p->output, p->offset_y * i + col_b * j + k, y);
216  }
217  }
218  }
219  }
220 
221  return RT_FUNCTION_ERROR_NOERROR;
222 }
