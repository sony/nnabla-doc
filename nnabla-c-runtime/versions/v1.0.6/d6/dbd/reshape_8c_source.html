31 rt_function_error_t allocate_reshape_local_context(rt_function_t *f) {
32  if (f->num_of_inputs != 1) {
33  return RT_FUNCTION_ERROR_INVALID_NUM_OF_INPUTS;
34  }
35  if (f->num_of_outputs != 1) {
36  return RT_FUNCTION_ERROR_INVALID_NUM_OF_OUTPUTS;
37  }
38  reshape_private_t *p = malloc(sizeof(reshape_private_t));
39  if (p == 0) {
40  return RT_FUNCTION_ERROR_MALLOC;
41  }
42  p->input = f->inputs[0];
43  p->get_input = select_getter(p->input);
44  p->input_size = calc_shape_size(f->inputs[0]->shape);
45  p->output = f->outputs[0];
46  p->set_output = select_setter(p->output);
47  p->output_size = calc_shape_size(f->outputs[0]->shape);
48  if (p->input_size != p->output_size) {
49  return RT_FUNCTION_ERROR_INVALID_SHAPE;
50  }
51  ((reshape_local_context_t *)(f->local_context))->data = (void *)p;
52  if (p->input->type == NN_DATA_TYPE_FLOAT &&
53  p->output->type == NN_DATA_TYPE_FLOAT) {
54  f->exec_func = exec_reshape;
55  } else {
56  f->exec_func = exec_reshape_generic;
57  }
58  return RT_FUNCTION_ERROR_NOERROR;
59 }
