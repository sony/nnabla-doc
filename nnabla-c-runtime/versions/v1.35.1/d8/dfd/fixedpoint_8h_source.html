22 inline int sum_acc_sat8(int8_t *acc, int8_t b) {
23  // Accumulate b into *acc, after cheking for overflow (8-bit accumulator)
24  int8_t result = *acc + b;
25  if (*acc > 0 && b > 0 && result < 0) {
26  *acc = INT8_MAX;
27  return -1;
28  } else if (*acc < 0 && b < 0 && result > 0) {
29  *acc = INT8_MIN;
30  return -1;
31  } else {
32  *acc = result;
33  return 0;
34  }
35 }

122 inline void rescale_fixed8(int8_t *arr, unsigned arr_size,
123  unsigned old_decimal_bits,
124  unsigned new_decimal_bits) {
125  // Rescale an array of 8-bit fixed-point values,
126  // from old_decimal_bits precision, to new_decimal_bits precision.
127 
128  const short n_bits_shift = old_decimal_bits - new_decimal_bits;
129  if (n_bits_shift > 0) {
130  const int8_t rescaling = 1 << n_bits_shift;
131  for (int i = 0; i < arr_size; i++) {
132  *(arr + i) /= rescaling;
133  }
134  } else if (n_bits_shift < 0) {
135  const int16_t rescaling = 1 << -n_bits_shift;
136  for (int i = 0; i < arr_size; i++) {
137  *(arr + i) = saturate16_to_8(*(arr + i) * rescaling); // ovf could occur
138  }
139  }
140 }

269 inline int8_t float_to_fixed8(float value, int n_precision_bits) {
270  const float FIXED8_MAX =
271  (1 << (8 - n_precision_bits - 1)) - (1 / (1 << n_precision_bits));
272  const float FIXED8_MIN = -(1 << (8 - n_precision_bits - 1));
273 
274  if (value >= FIXED8_MAX) {
275  return (int8_t)FIXED8_MAX;
276  } else if (value <= FIXED8_MIN) {
277  return (int8_t)FIXED8_MIN;
278  } else {
279  return (int8_t)roundf(value * (1 << n_precision_bits));
280  }
