94 rt_function_error_t exec_affine(rt_function_t *f) {
95  affine_private_t *p =
96  (affine_private_t *)(((affine_local_context_t *)(f->local_context))
97  ->data);
98  int i, j, k; // Iterators.
99  float *input = (float *)(p->input->data);
100  float *weight = (float *)(p->weight->data);
101  float *output = (float *)(p->output->data);
102  float *alpha = 0;
103 
104  if (p->alpha) {
105  alpha = (float *)p->alpha->data;
106  }
107 
108  // Clear output
109  memset(output, 0, sizeof(float) * p->output_size);
110 
111  for (k = 0; k < p->base_loop_size; k++) {
112  int output_offset = k * p->output_loop_size;
113  int input_offset = k * p->input_loop_size;
114  float *o_addr = output + output_offset;
115  float *alpha_addr = alpha;
116 
117  for (j = 0; j < p->output_loop_size; ++j) {
118  int weight_offset = j * p->input_loop_size;
119 
120  float *i_addr = input + input_offset;
121  float *w_addr = weight + weight_offset;
122  for (i = 0; i < p->input_loop_size; ++i) {
123  *o_addr += (*i_addr++) * (*w_addr++);
124  }
125  if (alpha) {
126  *o_addr *= *alpha_addr++;
127  }
128  ++o_addr;
129  }
130 
131  // Bias
132  if (p->bias) {
133  float *bias = (float *)(p->bias->data);
134  for (i = 0; i < p->output_loop_size; i++) {
135  int opos = output_offset + i;
136  int bpos = i;
137  *(output + opos) = *(output + opos) + *(bias + bpos);
138  }
139  }
140  }
141  return RT_FUNCTION_ERROR_NOERROR;
142 }
