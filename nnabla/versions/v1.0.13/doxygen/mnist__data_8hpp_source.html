32 vector<vector<uint8_t>> read_images(const string &train,
33  const string &data_root,
34  const string &filename) {
35 
36  gzFile fp = gzopen((data_root + filename).c_str(), "rb");
37  if (fp == NULL) {
38  cerr << "This sample requires mnist data downloaded before.";
39  exit(0);
40  }
41 
42  int header;
43  gzread(fp, (char *)&header, 4 * sizeof(int));
44  int num_images = 60000;
45  if (train == "test")
46  num_images = 10000;
47 
48  int num_rows = 28;
49  int num_cols = 28;
50 
51  vector<vector<uint8_t>> images;
52  for (int i = 0; i < num_images; ++i) {
53  vector<uint8_t> x;
54  for (int r = 0; r < num_rows; ++r) {
55  for (int c = 0; c < num_cols; ++c) {
56  unsigned char temp = 0;
57  gzread(fp, (char *)&temp, sizeof(temp));
58  x.push_back(temp);
59  }
60  }
61  images.push_back(x);
62  }
63  gzclose(fp);
64  return images;
65 }

112 MnistDataIterator::MnistDataIterator(string train)
113  : current_idx_(0), train_(train) {
114  string image_gz = "train-images-idx3-ubyte.gz";
115  string label_gz = "train-labels-idx1-ubyte.gz";
116  if (train == "test") {
117  image_gz = "t10k-images-idx3-ubyte.gz";
118  label_gz = "t10k-labels-idx1-ubyte.gz";
119  }
120  vector<vector<uint8_t>> images = read_images(train, "./", image_gz);
121  vector<uint8_t> labels = read_labels(train, "./", label_gz);
122  for (int i = 0; i < images.size(); i++) {
123  Pair xt;
124  xt.data = images[i];
125  xt.label = labels[i];
126  dataset_.push_back(xt);
127  }
128  permutation();
129 }

139 const vector<Pair> MnistDataIterator::get_batch(int batch_size) {
140 
141  const int n_data = dataset_.size();
142 
143  vector<Pair> batch_data(batch_size);
144 
145  int n_count = 0;
146  if (n_data <= current_idx_ + batch_size)
147  n_count = n_data - current_idx_;
148  for (int i = 0; i < n_count; i++)
149  batch_data[i] = dataset_[current_idx_++];
150 
151  if (n_data <= current_idx_) {
152  permutation();
153  current_idx_ = 0;
154  }
155 
156  for (int i = n_count; i < batch_size; i++)
157  batch_data[i] = dataset_[current_idx_++];
158 
159  return batch_data;
160 }

162 void set_data(Context cpu_ctx, vector<Pair> batch_data, CgVariablePtr x,
163  CgVariablePtr t) {
164 
165  int batch_size = batch_data.size();
166 
167  float_t *x_d =
168  x->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, true);
169  const int stride_x = x->variable()->strides()[0];
170  for (int i = 0; i < batch_size; i++) {
171  for (int j = 0; j < stride_x; j++)
172  x_d[i * stride_x + j] = (float_t)(batch_data[i].data[j]) / 255.;
173  }
174 
175  uint8_t *t_d =
176  t->variable()->cast_data_and_get_pointer<uint8_t>(cpu_ctx, true);
177  for (int i = 0; i < batch_size; i++)
178  t_d[i] = batch_data[i].label;
179 }
