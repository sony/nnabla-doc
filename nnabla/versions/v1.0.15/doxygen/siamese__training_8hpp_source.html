35 CgVariablePtr mnist_lenet_feature(CgVariablePtr image,
36  ParameterDirectory params, bool test) {
37  // from python example
38  // def mnist_lenet_feature(image, test=False):
39  // """
40  // Construct LeNet for MNIST.
41  // """
42  auto c1 = f::elu(pf::convolution(image, 1, 20, {5, 5}, params["conv1"]), 1.0);
43  c1 = f::average_pooling(c1, {2, 2}, {2, 2});
44  auto c2 = f::elu(pf::convolution(c1, 1, 50, {5, 5}, params["conv2"]), 1.0);
45  c2 = f::average_pooling(c2, {2, 2}, {2, 2});
46  auto c3 = f::elu(pf::affine(c2, 1, 500, params["fc3"]), 1.0);
47  auto c4 = f::elu(pf::affine(c3, 1, 10, params["fc4"]), 1.0);
48  auto c5 = pf::affine(c4, 1, 2, params["fc_embed"]);
49  return c5;
50 }

52 CgVariablePtr mnist_lenet_siamese(CgVariablePtr x0, CgVariablePtr x1,
53  ParameterDirectory params,
54  bool test = false) {
55  // from python example
56  // def mnist_lenet_siamese(x0, x1, test=False):
57  // """"""
58  auto h0 = mnist_lenet_feature(x0, params, test);
59  auto h1 = mnist_lenet_feature(x1, params, test);
60  auto h = f::squared_error(h0, h1);
61  auto p = f::sum(h, {1}, true);
62  return p;
63 }

65 CgVariablePtr contrastive_loss(CgVariablePtr sd, CgVariablePtr l,
66  float margin = 1.0, float eps = 1e-4) {
67  // from python example
68  // def contrastive_loss(sd, l, margin=1.0, eps=1e-4):
69  // """
70  // This implements contrastive loss function given squared difference `sd`
71  // and labels `l` in {0, 1}.
72  //
73  // f(sd, l) = l * sd + (1 - l) * max(0, margin - sqrt(sd))^2
74  //
75  // NNabla implements various basic arithmetic operations. That helps write
76  // custom operations
77  // with composition like this. This is handy, but still implementing NNabla
78  // Function in C++
79  // gives you better performance advantage.
80  // """
81  auto sim_cost = l * sd;
82  auto dissim_cost =
83  (1 - l) *
84  (f::pow_scalar(
85  f::maximum_scalar(margin - f::pow_scalar(sd + eps, 0.5), 0), 2.0));
86  return sim_cost + dissim_cost;
87 }

114  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
115  m1_->provide_data(cpu_ctx, batch_size_, x1, t1_);
116  m2_->provide_data(cpu_ctx, batch_size_, x2, t2_);

118  t1_->variable()->cast_data_and_get_pointer<uint8_t>(cpu_ctx, false);
119  uint8_t *t2_d =
120  t2_->variable()->cast_data_and_get_pointer<uint8_t>(cpu_ctx, false);
121  uint8_t *t_d =
122  t->variable()->cast_data_and_get_pointer<uint8_t>(cpu_ctx, true);
123  for (int i = 0; i < t->variable()->size(); i++)
124  t_d[i] = (t1_d[i] == t2_d[i]);
125  }

131 bool siamese_training(nbla::Context ctx) {
132 
133  // Create a context for cpu
134  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
135 
136  // # Create CNN network for both training and testing.
137  float margin = 1.0; //# Margin for contrastive loss.
138 
139  // # Create input variables.
140  SingletonManager::get<GlobalContext>()->set_current_context(ctx);
141  ParameterDirectory params;
142  int batch_size = 128;
143  auto timage0 =
144  make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
145  auto timage1 =
146  make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
147  auto tlabel = make_shared<CgVariable>(Shape_t({batch_size, 1}), false);
148 
149  // # Create prediction graph.
150  auto tpred = mnist_lenet_siamese(timage0, timage1, params, false);
151 
152  // # Create loss function.
153  auto tloss = f::mean(contrastive_loss(tpred, tlabel, margin), {0}, false);
154 
155  // # TEST
156  // # Create input variables.
157  auto vimage0 =
158  make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
159  auto vimage1 =
160  make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
161  auto vlabel = make_shared<CgVariable>(Shape_t({batch_size, 1}), false);
162 
163  // # Create prediction graph.
164  auto vpred = mnist_lenet_siamese(vimage0, vimage1, params, true);
165  auto vloss = f::mean(contrastive_loss(vpred, vlabel, margin), {0}, false);
166 
167  // # Create Solver.
168  float learning_rate = 1.0e-3;
169  auto solver = create_AdamSolver(ctx, learning_rate, 0.9, 0.999, 1.0e-8);
170  solver->set_parameters(params.get_parameters());
171 
172  // # Initialize DataIterator for MNIST.
173  MnistSiameseDataIterator tdata_iterator(batch_size, true);
174  MnistSiameseDataIterator vdata_iterator(batch_size, false);
175 
176  //# Training loop.
177  FILE *fp;
178  fp = fopen("log.txt", "wt");
179  if (fp == NULL) {
180  fprintf(stderr, "Error in opening log file.");
181  return false;
182  }
183 
184  // # Training loop.
185  int max_iter = 5000;
186  int n_val_interval = 10;
187  int n_val_iter = 10;
188  float weight_decay = 0.0e-4;
189  float mean_tloss = 0.;
190  for (int iter = 0; iter < max_iter; iter++) {
191 
192  // Get batch and copy to input variables
193  tdata_iterator.provide_data(timage0, timage1, tlabel);
194 
195  // # Training forward, backward and update.
196  solver->zero_grad();
197  tloss->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
198  tloss->variable()->grad()->fill(1.0);
199  tloss->backward(/*NdArrayPtr grad=*/nullptr, /*clear_buffer=*/true);
200  solver->weight_decay(weight_decay);
201  solver->update();
202 
203  // Get and print the average loss
204  float_t *tloss_d =
205  tloss->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, false);
206  mean_tloss += tloss_d[0];
207 
208  if ((iter + 1) % n_val_interval == 0) {
209  float mean_vloss = 0.0;
210  for (int i = 0; i < n_val_iter; i++) {
211  vdata_iterator.provide_data(vimage0, vimage1, vlabel);
212  vloss->forward(/*clear_buffer=*/true, /*clear_no_need_grad=*/true);
213  float_t *vloss_d =
214  vloss->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx,
215  false);
216  mean_vloss += vloss_d[0];
217  }
218  mean_tloss /= n_val_interval;
219  mean_vloss /= n_val_iter;
220  fprintf(fp, "iter: %d, tloss: %f, vloss: %f\n", iter, mean_tloss,
221  mean_vloss);
222  fprintf(stdout, "iter: %d, tloss: %f, vloss: %f\n", iter, mean_tloss,
223  mean_vloss);
224  mean_tloss = 0.;
225  mean_vloss = 0.;
226  }
227  }
228  fclose(fp);
229  return true;
230 }
