37 // Example of Lenet Model Design
38 /******************************************/
39 CgVariablePtr model(CgVariablePtr x, ParameterDirectory parameters) {
40 
41  //############################################
42  // Encoder of 2 fully connected layers
43  //############################################
44 
45  // Normalize input
46  auto xa = x;
47 
48  // Fully connected layers, and Elu replaced from original Softplus.
49  auto h1 = f::elu(pf::affine(xa, 1, 500, parameters["fc1"]), 1.0);
50  auto h2 = f::elu(pf::affine(h1, 1, 500, parameters["fc2"]), 1.0);
51 
52  // The outputs are the parameters of Gauss probability density.
53  auto mu = pf::affine(h2, 1, 50, parameters["fc_mu"]);
54  auto logvar = pf::affine(h2, 1, 50, parameters["fc_logvar"]);
55  auto sigma = f::exp(logvar * 0.5);
56 
57  // The prior variable and the reparameterization trick
58  Shape_t shape_mu = mu->variable()->shape();
59  vector<int> shape_z(shape_mu.size());
60  for (int i = 0; i < shape_z.size(); i++)
61  shape_z[i] = shape_mu[i];
62  auto epsilon = f::randn(0.0, 1.0, shape_z, 706);
63  auto z = mu + sigma * epsilon;
64 
65  //#############################################
66  // Decoder of 2 fully connected layers #
67  //#############################################
68 
69  // Fully connected layers, and Elu replaced from original Softplus.
70  auto h3 = f::elu(pf::affine(z, 1, 500, parameters["fc3"]), 1.0);
71  auto h4 = f::elu(pf::affine(h3, 1, 500, parameters["fc4"]), 1.0);
72 
73  // The outputs are the parameters of Bernoulli probabilities for each pixel.
74  Shape_t shape_xa = xa->variable()->shape();
75  int n_pb = 1;
76  vector<int> shape_pb(shape_xa.size());
77  for (int i = 0; i < shape_xa.size(); i++) {
78  if (0 < i)
79  n_pb *= shape_xa[i];
80  shape_pb[i] = shape_xa[i];
81  }
82  auto h5 = pf::affine(h4, 1, n_pb, parameters["fc5"]);
83  auto prob = f::reshape(h5, shape_pb, true);
84 
85  //############################################
86  // Elbo components and loss objective #
87  //############################################
88 
89  // Binarized input
90  auto xb = f::greater_equal_scalar(xa, 0.5);
91 
92  // E_q(z|x)[log(q(z|x))]
93  // without some constant terms that will canceled after summation of loss
94  auto logqz = 0.5 * f::sum(1.0 + logvar, {1}, false);
95 
96  // E_q(z|x)[log(p(z))]
97  // without some constant terms that will canceled after summation of loss
98  auto logpz = 0.5 * f::sum(mu * mu + sigma * sigma, {1}, false);
99 
100  // E_q(z|x)[log(p(x|z))]
101  auto logpx = f::sum(f::sigmoid_cross_entropy(prob, xb), {1, 2, 3}, false);
102 
103  // Vae loss, the negative evidence lowerbound
104  auto loss = f::mean(logpx + logpz - logqz, {0}, false);
105 
106  return loss;
107 }

112 bool vae_training(nbla::Context ctx) {
113 
114  // Create a context for cpu
115  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
116 
117  // Create mnist data iterator
118  MnistDataIterator train_data_provider("train");
119  MnistDataIterator test_data_provider("test");
120 
121  // Build network
122  SingletonManager::get<GlobalContext>()->set_current_context(ctx);
123  ParameterDirectory params;
124  int batch_size = 100;
125  auto x = make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
126  auto t = make_shared<CgVariable>(Shape_t({batch_size, 1}), false);
127  auto loss = model(x, params);
128 
129  // Setup solver and input learnable parameters
130  float learning_rate = 3.0e-4;
131  auto adam = create_AdamSolver(ctx, learning_rate, 0.9, 0.999, 1.0e-8);
132  adam->set_parameters(params.get_parameters());
133 
134  // Execute training
135  FILE *fp;
136  fp = fopen("log.txt", "wt");
137  if (fp == NULL) {
138  fprintf(stderr, "Error in opening log file.");
139  return false;
140  }
141 
142  int max_iter = 60000;
143  int n_val_interval = 10;
144  int n_val_iter = 10;
145  float mean_loss_t = 0.;
146  for (int iter = 0; iter < max_iter; iter++) {
147 
148  // Get batch and copy to input variables
149  train_data_provider.provide_data(cpu_ctx, batch_size, x, t);
150 
151  // Execute forward, backward and update
152  adam->zero_grad();
153  loss->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
154  loss->variable()->grad()->fill(1.0);
155  loss->backward(/*NdArrayPtr grad =*/nullptr,
156  /*bool clear_buffer = */ true);
157  adam->update();
158 
159  // Get and print the average loss
160  float_t *loss_d =
161  loss->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, false);
162  mean_loss_t += loss_d[0];
163 
164  if ((iter + 1) % n_val_interval == 0) {
165 
166  float mean_loss_v = 0.0;
167  for (int v_iter = 0; v_iter < n_val_iter; v_iter++) {
168  test_data_provider.provide_data(cpu_ctx, batch_size, x, t);
169  loss->forward(/*clear_buffer=*/true, /*clear_no_need_grad=*/true);
170  float_t *loss_d = loss->variable()->cast_data_and_get_pointer<float_t>(
171  cpu_ctx, false);
172  mean_loss_v += loss_d[0];
173  }
174  mean_loss_t /= n_val_interval;
175  mean_loss_v /= n_val_iter;
176 
177  fprintf(fp, "iter: %d, tloss: %f, vloss: %f\n", iter + 0, mean_loss_t,
178  mean_loss_v);
179  fprintf(stdout, "iter: %d, tloss: %f, vloss: %f\n", iter + 0, mean_loss_t,
180  mean_loss_v);
181  mean_loss_t = 0;
182  }
183  }
184  fclose(fp);
185  return true;
186 }
