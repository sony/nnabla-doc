42 /******************************************/
43 CgVariablePtr model(CgVariablePtr x, ParameterDirectory parameters) {
44  auto h = pf::convolution(x, 1, 16, {5, 5}, parameters["conv1"]);
45  h = f::max_pooling(h, {2, 2}, {2, 2}, true, {0, 0}, false);
46  h = f::relu(h, false);
47  h = pf::convolution(h, 1, 16, {5, 5}, parameters["conv2"]);
48  h = f::max_pooling(h, {2, 2}, {2, 2}, true, {0, 0}, false);
49  h = f::relu(h, false);
50  h = pf::affine(h, 1, 50, parameters["affine3"]);
51  h = f::relu(h, false);
52  h = pf::affine(h, 1, 10, parameters["affine4"]);
53  return h;
54 }

59 bool lenet_training_with_static_graph(nbla::Context ctx) {
60 
61  // Create a context for cpu
62  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
63 
64  // Create mnist data iterator
65  MnistDataIterator train_data_provider("train");
66  MnistDataIterator test_data_provider("test");
67 
68  // Setup context
69  SingletonManager::get<GlobalContext>()->set_current_context(ctx);
70 
71  // Build network
72  ParameterDirectory params;
73 
74  // Load pretrained parameter if it has.
75  utl::load_parameters(params, "lenet_param.protobuf");
76 
77  int batch_size = 128;
78  auto x = make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
79  auto t = make_shared<CgVariable>(Shape_t({batch_size, 1}), false);
80  auto h = model(x, params);
81  auto loss = f::mean(f::softmax_cross_entropy(h, t, 1), {0, 1}, false);
82  auto err = f::mean(f::top_n_error(h, t, 1, 1), {0, 1}, false);
83 
84  // Setup solver and input learnable parameters
85  auto adam = create_AdamSolver(ctx, 0.001, 0.9, 0.999, 1.0e-8);
86  adam->set_parameters(params.get_parameters());
87 
88  // Execute training
89  FILE *fp;
90  fp = fopen("log.txt", "wt");
91  if (fp == NULL) {
92  fprintf(stderr, "Error in opening log file.");
93  return false;
94  }
95 
96  try {
97  int max_iter = 10000;
98  int n_val_iter = 10;
99  float mean_t_loss = 0.;
100  for (int iter = 0; iter < max_iter; iter++) {
101 
102  // Get batch and copy to input variables
103  train_data_provider.provide_data(cpu_ctx, batch_size, x, t);
104 
105  // Execute forward, backward and update
106  adam->zero_grad();
107  loss->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
108  loss->variable()->grad()->fill(1.0);
109  loss->backward(/*NdArrayPtr grad =*/nullptr,
110  /*bool clear_buffer = */ true);
111  adam->update();
112 
113  // Get and print the average loss
114  float_t *t_loss_d =
115  loss->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, false);
116  mean_t_loss += t_loss_d[0];
117 
118  if ((iter + 1) % n_val_iter == 0) {
119 
120  float mean_v_err = 0.0;
121  for (int v_iter = 0; v_iter < 10; v_iter++) {
122  test_data_provider.provide_data(cpu_ctx, batch_size, x, t);
123  err->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
124  float_t *v_err_d =
125  err->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx,
126  false);
127  mean_v_err += v_err_d[0];
128  }
129  mean_t_loss /= n_val_iter;
130  mean_v_err /= 10;
131 
132  fprintf(fp, "iter: %d, tloss: %f, verr: %f\n", iter + 0, mean_t_loss,
133  mean_v_err);
134  fprintf(stdout, "iter: %d, tloss: %f, verr: %f\n", iter + 0,
135  mean_t_loss, mean_v_err);
136  mean_t_loss = 0;
137 
138  // Save parameters as a snapshot.
139  utl::save_parameters(params, "saved_lenet_param.protobuf");
140  }
141  }
142  } catch (...) {
143  cout << "Exception in lenet_training_with_static_graph.\n";
144  fclose(fp);
145  return false;
146  }
147  fclose(fp);
148  return true;
149 }
