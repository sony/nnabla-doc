42 CgVariablePtr generator(CgVariablePtr z, int max_h, bool test,
43  ParameterDirectory params) {
44  // Building generator network which takes (B, Z, 1, 1) inputs and generates
45  //(B, 1, 28, 28) outputs.
46 
47  assert(max_h / 4 > 0);
48 
49  // # (Z, 1, 1) --> (256, 4, 4)
50  pf::ConvolutionOpts opts1 = pf::ConvolutionOpts().with_bias(false);
51  auto h1 = pf::deconvolution(z, 1, max_h, {4, 4}, params["deconv1"], opts1);
52  auto b1 = pf::batch_normalization(h1, !test, params["deconv1"]);
53  auto e1 = f::elu(b1, 1.0);
54 
55  // # (256, 4, 4) --> (128, 8, 8)
56  pf::ConvolutionOpts opts2 =
57  pf::ConvolutionOpts().with_bias(false).pad({1, 1}).stride({2, 2});
58  auto h2 =
59  pf::deconvolution(e1, 1, max_h / 2, {4, 4}, params["deconv2"], opts2);
60  auto b2 = pf::batch_normalization(h2, !test, params["deconv2"]);
61  auto e2 = f::elu(b2, 1.0);
62 
63  // # (128, 8, 8) --> (64, 16, 16)
64  pf::ConvolutionOpts opts3 =
65  pf::ConvolutionOpts().with_bias(false).pad({1, 1}).stride({2, 2});
66  auto h3 =
67  pf::deconvolution(e2, 1, max_h / 4, {4, 4}, params["deconv3"], opts3);
68  auto b3 = pf::batch_normalization(h3, !test, params["deconv3"]);
69  auto e3 = f::elu(b3, 1.0);
70 
71  // # (64, 16, 16) --> (32, 28, 28)
72  // # Convolution with kernel=4, pad=3 and stride=2 transforms a 28 x 28 map
73  // # to a 16 x 16 map. Deconvolution with those parameters behaves like an
74  // # inverse operation, i.e. maps 16 x 16 to 28 x 28.
75  pf::ConvolutionOpts opts4 =
76  pf::ConvolutionOpts().with_bias(false).pad({3, 3}).stride({2, 2});
77  auto h4 =
78  pf::deconvolution(e3, 1, max_h / 4, {4, 4}, params["deconv4"], opts4);
79  auto b4 = pf::batch_normalization(h4, !test, params["deconv4"]);
80  auto e4 = f::elu(b4, 1.0);
81 
82  // # (32, 28, 28) --> (1, 28, 28)
83  pf::ConvolutionOpts opts5 = pf::ConvolutionOpts().pad({1, 1});
84  auto h5 = pf::convolution(e4, 1, 1, {3, 3}, params["conv5"], opts5);
85  auto x = f::sigmoid(h5);
86  return x;
87 }

89 CgVariablePtr discriminator(CgVariablePtr x, int max_h, bool test,
90  ParameterDirectory params) {
91  // Building discriminator network which maps a (B, 1, 28, 28) input to
92  // a (B, 1).
93 
94  assert(max_h / 8 > 0);
95 
96  // # (1, 28, 28) --> (32, 16, 16)
97  pf::ConvolutionOpts opts1 =
98  pf::ConvolutionOpts().pad({3, 3}).stride({2, 2}).with_bias(false);
99  auto h1 = pf::convolution(x, 1, max_h / 8, {3, 3}, params["conv1"], opts1);
100  auto b1 = pf::batch_normalization(h1, !test, params["conv1"]);
101  auto e1 = f::elu(b1, 1.0);
102 
103  // # (32, 16, 16) --> (64, 8, 8)
104  pf::ConvolutionOpts opts2 =
105  pf::ConvolutionOpts().pad({1, 1}).stride({2, 2}).with_bias(false);
106  auto h2 = pf::convolution(e1, 1, max_h / 4, {3, 3}, params["conv2"], opts2);
107  auto b2 = pf::batch_normalization(h2, !test, params["conv2"]);
108  auto e2 = f::elu(b2, 1.0);
109 
110  // # (64, 8, 8) --> (128, 4, 4)
111  pf::ConvolutionOpts opts3 =
112  pf::ConvolutionOpts().pad({1, 1}).stride({2, 2}).with_bias(false);
113  auto h3 = pf::convolution(e2, 1, max_h / 2, {3, 3}, params["conv3"], opts3);
114  auto b3 = pf::batch_normalization(h3, !test, params["conv3"]);
115  auto e3 = f::elu(b3, 1.0);
116 
117  // # (128, 4, 4) --> (256, 4, 4)
118  pf::ConvolutionOpts opts4 =
119  pf::ConvolutionOpts().pad({1, 1}).with_bias(false);
120  auto h4 = pf::convolution(e3, 1, max_h, {3, 3}, params["conv4"], opts4);
121  auto b4 = pf::batch_normalization(h4, !test, params["conv4"]);
122 
123  // # (256, 4, 4) --> (1,)
124  auto f1 = pf::affine(b4, 1, 1, params["fc1"]);
125  return f1;
126 }

148 /******************************************/
149 // Example of DCGAN Training
150 /******************************************/
151 bool dcgan_training(nbla::Context ctx) {
152 
153  // Create a context for cpu
154  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
155 
156  // Build network
157  SingletonManager::get<GlobalContext>()->set_current_context(ctx);
158  int max_h = 256;
159  int batch_size = 64;
160  ParameterDirectory params;
161 
162  // # Fake path
163  auto z = make_shared<CgVariable>(Shape_t({batch_size, 100, 1, 1}), false);
164  auto fake = generator(z, max_h, false, params["gen"]);
165  fake->set_persistent(true);
166 
167  auto pred_fake = discriminator(fake, max_h, false, params["dis"]);
168  auto loss_gen = f::mean(
169  f::sigmoid_cross_entropy(pred_fake, f::constant(1, {batch_size, 1})),
170  {0, 1}, false);
171 
172  auto fake_dis = make_shared<CgVariable>(fake->variable(), true);
173  fake_dis->set_need_grad(true);
174  auto pred_fake_dis = discriminator(fake, max_h, false, params["dis"]);
175  pred_fake_dis->set_persistent(true);
176  auto loss_dis = f::mean(
177  f::sigmoid_cross_entropy(pred_fake_dis, f::constant(0, {batch_size, 1})),
178  {0, 1}, false);
179 
180  // # Real path
181  auto x = make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
182  auto t = make_shared<CgVariable>(Shape_t({batch_size, 1}), false);
183  auto pred_real = discriminator(x, max_h, false, params["dis"]);
184  loss_dis = loss_dis + f::mean(f::sigmoid_cross_entropy(
185  pred_real, f::constant(1, {batch_size, 1})),
186  {0, 1}, false);
187 
188  // # Create Solver.
189  float learning_rate = 2.0e-4;
190  auto solver_gen = create_AdamSolver(ctx, learning_rate, 0.5, 0.999, 1.0e-8);
191  auto solver_dis = create_AdamSolver(ctx, learning_rate, 0.5, 0.999, 1.0e-8);
192  solver_gen->set_parameters(params["gen"].get_parameters());
193  solver_dis->set_parameters(params["dis"].get_parameters());
194 
195  // Create mnist data iterator
196  MnistDataIterator train_data_provider("train");
197 
198  //# Training loop.
199  FILE *fp;
200  fp = fopen("log.txt", "wt");
201  if (fp == NULL) {
202  fprintf(stderr, "Error in opening log file.");
203  return false;
204  }
205 
206  int max_iter = 20000;
207  int n_val_iter = 10;
208  float weight_decay = 1.0e-4;
209  float mean_loss_gen = 0.;
210  float mean_loss_dis = 0.;
211  for (int iter = 0; iter < max_iter; iter++) {
212 
213  // Get batch and copy to input variables
214  train_data_provider.provide_data(cpu_ctx, batch_size, x, t);
215 
216  float_t *z_d =
217  z->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, true);
218  for (int i = 0; i < z->variable()->size(); i++, z_d++)
219  *z_d = normal(engine);
220 
221  // # Generator update.
222  solver_gen->zero_grad();
223  loss_gen->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
224  loss_gen->variable()->grad()->fill(1.0);
225  loss_gen->backward(/*NdArrayPtr grad=*/nullptr, /*clear_buffer=*/true);
226  solver_gen->weight_decay(weight_decay);
227  solver_gen->update();
228 
229  // # Discriminator update.
230  solver_dis->zero_grad();
231  loss_dis->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
232  loss_dis->variable()->grad()->fill(1.0);
233  loss_dis->backward(/*NdArrayPtr grad=*/nullptr, /*clear_buffer=*/true);
234  solver_dis->weight_decay(weight_decay);
235  solver_dis->update();
236 
237  // Get and print the average loss
238  float_t *loss_gen_d =
239  loss_gen->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx,
240  false);
241  mean_loss_gen += loss_gen_d[0];
242 
243  float_t *loss_dis_d =
244  loss_dis->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx,
245  false);
246  mean_loss_dis += loss_dis_d[0];
247 
248  if ((iter + 1) % n_val_iter == 0) {
249  mean_loss_gen /= n_val_iter;
250  mean_loss_dis /= n_val_iter;
251  fprintf(fp, "iter: %d, loss_gen: %f, loss_dis: %f\n", iter + 0,
252  mean_loss_gen, mean_loss_dis);
253  fprintf(stdout, "iter: %d, loss_gen: %f, loss_dis: %f\n", iter + 0,
254  mean_loss_gen, mean_loss_dis);
255  mean_loss_gen = 0.;
256  mean_loss_dis = 0.;
257  }
258 
259  // Get generated images"
260  if ((iter + 1) % max_iter == 0)
261  print_array(fake);
262  }
263  fclose(fp);
264  return true;
265 }
