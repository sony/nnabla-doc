113 template <typename T, typename... Args>
114 string format_string(const string &format, T first, Args... rest) {
115  int size = snprintf(nullptr, 0, format.c_str(), first, rest...);
116  if (size < 0) {
117  std::printf("fatal error in format_string function: snprintf failed\n");
118  std::abort();
119  }
120  vector<char> buffer(size + 1);
121  snprintf(buffer.data(), size + 1, format.c_str(), first, rest...);
122  return string(buffer.data(), buffer.data() + size);
123 }
