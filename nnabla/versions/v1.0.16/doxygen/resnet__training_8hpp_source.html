39 CgVariablePtr res_unit(CgVariablePtr x, bool test,
40  ParameterDirectory parameters) {
41  // from python examples
42  // def res_unit(x, scope):
43  // C = x.shape[1]
44  // with nn.parameter_scope(scope):
45  // with nn.parameter_scope('conv1'):
46  // h = F.elu(bn(PF.convolution(x, C / 2, (1, 1),
47  // with_bias=False)))
48  // with nn.parameter_scope('conv2'):
49  // h = F.elu(
50  // bn(PF.convolution(h, C / 2, (3, 3), pad=(1, 1),
51  // with_bias=False)))
52  // with nn.parameter_scope('conv3'):
53  // h = bn(PF.convolution(h, C, (1, 1), with_bias=False))
54  // return F.elu(F.add2(h, x, inplace=True))
55  Shape_t shape_x = x->variable()->shape();
56  const int n_channel = shape_x[1];
57  pf::ConvolutionOpts opts1 = pf::ConvolutionOpts().with_bias(false);
58  pf::ConvolutionOpts opts2 =
59  pf::ConvolutionOpts().with_bias(false).pad({1, 1});
60  pf::ConvolutionOpts opts3 = pf::ConvolutionOpts().with_bias(false);
61  auto c1 =
62  pf::convolution(x, 1, n_channel / 2, {1, 1}, parameters["conv1"], opts1);
63  auto e1 =
64  f::elu(pf::batch_normalization(c1, !test, parameters["conv1"]), 1.0);
65  auto c2 =
66  pf::convolution(e1, 1, n_channel / 2, {3, 3}, parameters["conv2"], opts2);
67  auto e2 =
68  f::elu(pf::batch_normalization(c2, !test, parameters["conv2"]), 1.0);
69  auto c3 =
70  pf::convolution(e2, 1, n_channel, {1, 1}, parameters["conv3"], opts3);
71  auto e3 =
72  f::elu(pf::batch_normalization(c3, !test, parameters["conv3"]), 1.0);
73  return f::elu(f::add2(e3, x, true), 1.0);
74 }

86 CgVariablePtr model(CgVariablePtr x, bool test, ParameterDirectory parameters) {
87  // from python exmaples
88  // xa = /= 255.0;
89  // xa = augmentation(xa, test, aug);
90  // # Conv1 --> 64 x 32 x 32
91  // with nn.parameter_scope("conv1"):
92  // c1 = F.elu(
93  // bn(PF.convolution(image, 64, (3, 3), pad=(3, 3),
94  // with_bias=False)))
95  // # Conv2 --> 64 x 16 x 16
96  // c2 = F.max_pooling(res_unit(c1, "conv2"), (2, 2))
97  // # Conv3 --> 64 x 8 x 8
98  // c3 = F.max_pooling(res_unit(c2, "conv3"), (2, 2))
99  // # Conv4 --> 64 x 8 x 8
100  // c4 = res_unit(c3, "conv4")
101  // # Conv5 --> 64 x 4 x 4
102  // c5 = F.max_pooling(res_unit(c4, "conv5"), (2, 2))
103  // # Conv5 --> 64 x 4 x 4
104  // c6 = res_unit(c5, "conv6")
105  // pl = F.average_pooling(c6, (4, 4))
106  // with nn.parameter_scope("classifier"):
107  // y = PF.affine(pl, 10)
108  // return y
109 
110  auto xa = x * (1.0 / 255.0);
111  xa = augmentation(x, !test);
112 
113  // Conv1 --> 64 x 32 x 32
114  pf::ConvolutionOpts opts = pf::ConvolutionOpts().with_bias(false).pad({3, 3});
115  auto c1 = pf::convolution(x, 1, 64, {3, 3}, parameters["conv1"], opts);
116  c1 = f::elu(pf::batch_normalization(c1, !test, parameters["conv1"]), 1.0);
117 
118  // Conv2 --> 64 x 16 x 16
119  auto c2 =
120  f::max_pooling(res_unit(c1, test, parameters["conv2"]), {2, 2}, {2, 2});
121 
122  // Conv3 --> 64 x 8 x 8
123  auto c3 =
124  f::max_pooling(res_unit(c2, test, parameters["conv3"]), {2, 2}, {2, 2});
125 
126  // Conv4 --> 64 x 8 x 8
127  auto c4 = res_unit(c3, test, parameters["conv4"]);
128 
129  // Conv5 --> 64 x 4 x 4
130  auto c5 =
131  f::max_pooling(res_unit(c4, test, parameters["conv5"]), {2, 2}, {2, 2});
132 
133  // Conv6 --> 64 x 4 x 4
134  auto c6 = res_unit(c5, test, parameters["conv6"]);
135 
136  // affine
137  auto h = f::average_pooling(c6, {4, 4}, {4, 4});
138  auto y = pf::affine(h, 1, 10, parameters["classifier"]);
139  return y;
140 }

145 bool resnet_training(nbla::Context ctx) {
146 
147  // Create a context for cpu
148  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
149 
150  // Create mnist data iterator
151  MnistDataIterator train_data_provider("train");
152  MnistDataIterator test_data_provider("test");
153 
154  // Build network
155  SingletonManager::get<GlobalContext>()->set_current_context(ctx);
156  ParameterDirectory params;
157  int batch_size = 128;
158  auto x = make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
159  auto t = make_shared<CgVariable>(Shape_t({batch_size, 1}), false);
160  auto h = model(x, false, params);
161  auto loss = f::mean(f::softmax_cross_entropy(h, t, 1), {0, 1}, false);
162  auto err = f::mean(f::top_n_error(h, t, 1, 1), {0, 1}, false);
163 
164  // Setup solver and input learnable parameters
165  auto adam = create_AdamSolver(ctx, 0.001, 0.9, 0.999, 1.0e-8);
166  adam->set_parameters(params.get_parameters());
167 
168  // Execute training
169  FILE *fp;
170  fp = fopen("log.txt", "wt");
171  if (fp == NULL) {
172  fprintf(stderr, "Error in opening log file.");
173  return false;
174  }
175 
176  int max_iter = 10000;
177  int n_val_iter = 10;
178  float mean_t_loss = 0.;
179  for (int iter = 0; iter < max_iter; iter++) {
180 
181  // Get batch and copy to input variables
182  train_data_provider.provide_data(cpu_ctx, batch_size, x, t);
183 
184  // Execute forward, backward and update
185  adam->zero_grad();
186  loss->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
187  loss->variable()->grad()->fill(1.0);
188  loss->backward(/*NdArrayPtr grad =*/nullptr, /*bool clear_buffer = */ true);
189  adam->update();
190 
191  // Get and print the average loss
192  float_t *t_loss_d =
193  loss->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, false);
194  mean_t_loss += t_loss_d[0];
195 
196  if ((iter + 1) % n_val_iter == 0) {
197 
198  float mean_v_err = 0.0;
199  for (int v_iter = 0; v_iter < 10; v_iter++) {
200  test_data_provider.provide_data(cpu_ctx, batch_size, x, t);
201  err->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
202  float_t *v_err_d =
203  err->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, false);
204  mean_v_err += v_err_d[0];
205  }
206  mean_t_loss /= n_val_iter;
207  mean_v_err /= 10;
208 
209  fprintf(fp, "iter: %d, tloss: %f, verr: %f\n", iter + 1, mean_t_loss,
210  mean_v_err);
211  fprintf(stdout, "iter: %d, tloss: %f, verr: %f\n", iter + 1, mean_t_loss,
212  mean_v_err);
213  mean_t_loss = 0;
214  }
215  }
216  fclose(fp);
217  return true;
218 }
