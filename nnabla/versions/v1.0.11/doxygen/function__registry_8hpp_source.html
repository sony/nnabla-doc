48 template <typename Item> class NBLA_API FunctionDb {
49  vector<shared_ptr<Item>> items_;
50 
51 public:
56  typename Item::function_t query(const vector<string> &backend) {
57  auto it = items_.end();
58  for (auto &be : backend) {
59  it = std::find_if(
60  items_.begin(), items_.end(),
61  [be](const shared_ptr<Item> &item) { return item->backend == be; });
62  if (it != items_.end()) {
63  break;
64  }
65  }
66  NBLA_CHECK(it != items_.end(), error_code::value,
67  "Any of [%s] could not be found in %s",
68  string_join(backend, ", ").c_str(),
69  print_function_items<Item>(items_).c_str());
70  return (*it)->function;
71  }
72 
76  void add(shared_ptr<Item> item) { items_.push_back(item); }
77 };
