30 template <typename... Args>
31 class BaseTransformBinary : public BaseFunction<Args...> {
32 protected:
33  shared_ptr<Function> f_bc0_, f_bc1_;
34  shared_ptr<Variable> o_bc0_, o_bc1_;
35 
36 public:
37  BaseTransformBinary(const Context &ctx, Args... args)
38  : BaseFunction<Args...>(ctx, args...) {}
39  virtual ~BaseTransformBinary() {}
40  virtual int min_inputs() { return 2; }
41  virtual int min_outputs() { return 1; }
42 
43 protected:
44  virtual void setup_impl(const Variables &inputs, const Variables &outputs) {
45  int ndim = inputs[0]->ndim();
46  NBLA_CHECK(ndim == inputs[1]->ndim(), error_code::value,
47  "Dimensions of inputs must match. "
48  "inputs[0]: %d != inputs[1]: %d.",
49  ndim, inputs[1]->ndim());
50  Shape_t s0 = inputs[0]->shape();
51  Shape_t s1 = inputs[1]->shape();
52  Shape_t oshape(ndim);
53  bool bc0 = false;
54  bool bc1 = false;
55  for (int i = 0; i < ndim; ++i) {
56  if (s0[i] != s1[i]) {
57  NBLA_CHECK(std::min(s0[i], s1[i]) == 1, error_code::value,
58  "Broadcast dimension must be one. shape[%d]: %d.", i,
59  std::min(s0[i], s1[i]));
60  if (s0[i] == 1) {
61  bc0 = true;
62  }
63  if (s1[i] == 1) {
64  bc1 = true;
65  }
66  }
67  oshape[i] = std::max(s0[i], s1[i]);
68  }
69  outputs[0]->reshape(oshape, true);
70  if (bc0) {
71  o_bc0_ = make_shared<Variable>(Shape_t{});
72  f_bc0_ = create_Broadcast(this->ctx_,
73  vector<int>(oshape.cbegin(), oshape.cend()));
74  f_bc0_->setup(Variables{inputs[0]}, Variables{o_bc0_.get()});
75  }
76  if (bc1) {
77  o_bc1_ = make_shared<Variable>(Shape_t{});
78  f_bc1_ = create_Broadcast(this->ctx_,
79  vector<int>(oshape.cbegin(), oshape.cend()));
80  f_bc1_->setup(Variables{inputs[1]}, Variables{o_bc1_.get()});
81  }
82  }
83 };

153 template <typename T, typename BinaryOp, typename... Args>
154 void TransformBinary<T, BinaryOp, Args...>::forward_impl(
155  const Variables &inputs, const Variables &outputs) {
156  auto _get = [this](Variable *v) {
157  return v->get_data_pointer<T>(this->ctx_);
158  };
159  if (this->f_bc0_) {
160  this->f_bc0_->forward(Variables{inputs[0]}, Variables{this->o_bc0_.get()});
161  }
162  if (this->f_bc1_) {
163  this->f_bc1_->forward(Variables{inputs[1]}, Variables{this->o_bc1_.get()});
164  }
165  const T *x0 = _get((this->f_bc0_) ? (this->o_bc0_.get()) : (inputs[0]));
166  const T *x1 = _get((this->f_bc1_) ? (this->o_bc1_.get()) : (inputs[1]));
167  T *y = outputs[0]->cast_data_and_get_pointer<T>(this->ctx_, true);
168  transform_binary(outputs[0]->size(), x0, x1, y, binary_op_);
169 }

171 template <typename T, typename BinaryOp, typename... Args>
172 void TransformBinary<T, BinaryOp, Args...>::backward_impl(
173  const Variables &inputs, const Variables &outputs,
174  const vector<bool> &propagate_down, const vector<bool> &accum) {
175  if (!(propagate_down[0] || propagate_down[1])) {
176  return;
177  }
178  auto _get_data = [this](Variable *v) {
179  return v->get_data_pointer<T>(this->ctx_);
180  };
181  auto _cast_grad = [this](Variable *v, bool wo) {
182  return v->cast_grad_and_get_pointer<T>(this->ctx_, wo);
183  };
184  const T *x0 = _get_data((this->f_bc0_) ? (this->o_bc0_.get()) : (inputs[0]));
185  const T *x1 = _get_data((this->f_bc1_) ? (this->o_bc1_.get()) : (inputs[1]));
186  const T *dy = outputs[0]->get_grad_pointer<T>(this->ctx_);
187  const T *y = _get_data(outputs[0]);
188  Size_t size = outputs[0]->size();
189  if (propagate_down[0]) {
190  T *dx0 = (this->f_bc0_) ? _cast_grad(this->o_bc0_.get(), true)
191  : _cast_grad(inputs[0], !accum[0]);
192  if ((!this->f_bc0_) && accum[0])
193  transform_binary_grad0<T, BinaryOp, true>(size, dy, x0, x1, y, dx0,
194  binary_op_);
195  else
196  transform_binary_grad0<T, BinaryOp, false>(size, dy, x0, x1, y, dx0,
197  binary_op_);
198  if (this->f_bc0_) {
199  this->f_bc0_->backward(Variables{inputs[0]},
200  Variables{this->o_bc0_.get()}, {true}, {accum[0]});
201  }
202  }
203  if (propagate_down[1]) {
204  T *dx1 = (this->f_bc1_) ? _cast_grad(this->o_bc1_.get(), true)
205  : _cast_grad(inputs[1], !accum[1]);
206  if ((!this->f_bc1_) && accum[1])
207  transform_binary_grad1<T, BinaryOp, true>(size, dy, x0, x1, y, dx1,
208  binary_op_);
209  else
210  transform_binary_grad1<T, BinaryOp, false>(size, dy, x0, x1, y, dx1,
211  binary_op_);
212  if (this->f_bc1_) {
213  this->f_bc1_->backward(Variables{inputs[1]},
214  Variables{this->o_bc1_.get()}, {true}, {accum[1]});
215  }
216  }
