79 template <typename T, typename UnaryOp>
80 void transform_unary(int size, const T *x, T *y, UnaryOp op) {
81 #ifdef _OPENMP
82 #pragma omp parallel for schedule(static)
83 #endif
84  for (int idx = 0; idx < size; ++idx) {
85  y[idx] = op(x[idx]);
86  }
87 }

89 template <typename T, typename UnaryOp, bool accum>
90 void transform_unary_grad(int size, const T *dy, const T *x, const T *y, T *g,
91  UnaryOp op) {
92  for (int idx = 0; idx < size; ++idx) {
93  g[idx] = (accum ? g[idx] : (T)0) + op.g(dy[idx], x[idx], y[idx]);
94  }
95 }

98 void TransformUnary<T, UnaryOp, Args...>::forward_impl(

100  const T *x = inputs[0]->get_data_pointer<T>(this->ctx_);

105 template <typename T, typename UnaryOp, typename... Args>
106 void TransformUnary<T, UnaryOp, Args...>::backward_impl(
107  const Variables &inputs, const Variables &outputs,
108  const vector<bool> &propagate_down, const vector<bool> &accum) {
109  if (!propagate_down[0]) {
110  return;
111  }
112  const T *dy = outputs[0]->get_grad_pointer<T>(this->ctx_);
113  const T *x = inputs[0]->get_data_pointer<T>(this->ctx_);
114  const T *y = outputs[0]->get_data_pointer<T>(this->ctx_);
115  Size_t size = inputs[0]->size();
116  T *g = inputs[0]->cast_grad_and_get_pointer<T>(this->ctx_, !accum[0]);
117  if (accum[0])
118  transform_unary_grad<T, UnaryOp, true>(size, dy, x, y, g, unary_op_);
119  else
120  transform_unary_grad<T, UnaryOp, false>(size, dy, x, y, g, unary_op_);
121 }
