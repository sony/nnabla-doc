39 CgVariablePtr mlp_net(CgVariablePtr x, int n_h, int n_y,
40  ParameterDirectory params, bool test) {
41  // from python examples
42  //"""
43  // Function for building multi-layer-perceptron with batch_normalization
44  //
45  // Args:
46  // x(`~nnabla.Variable`): N-D array
47  // n_h(int): number of units in an intermediate layer
48  // n_y(int): number of classes
49  // test: operation type train=True, test=False
50  //
51  // Returns:
52  // ~nnabla.Variable: h
53  //"""
54  auto h1 = pf::affine(x, 1, n_h, params["fc1"]);
55  auto r1 = f::relu(pf::batch_normalization(h1, !test, params["fc1"]), false);
56  auto h2 = pf::affine(r1, 1, n_h, params["fc2"]);
57  auto r2 = f::relu(pf::batch_normalization(h2, !test, params["fc3"]), false);
58  auto h3 = pf::affine(r2, 1, n_y, params["fc3"]);
59  return h3;
60 }

73 // Example of Virtual Adversarial Training
74 /******************************************/
75 bool vat_training(nbla::Context ctx) {
76  // from python examples
77  //"""
78  // Main script.
79  //
80  // Steps:
81  // * Get and set context.
82  // * Load Dataset
83  // * Initialize DataIterator.
84  // * Create Networks
85  // * Network for training with labeled data
86  // * Network for training with unlabeled data
87  // * Network for evaluation with validation data
88  // * Create Solver.
89  // * Training Loop.
90  // * Training
91  // * by Labeled Data
92  // * Calculate loss with labeled data
93  // * by Unlabeled Data
94  // * Calculate virtual adversarial loss
95  // * Calculate loss with virtual adversarial noise
96  // """
97 
98  // Create a context for cpu
99  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
100 
101  // # Create networks
102  // # Network
103  SingletonManager::get<GlobalContext>()->set_current_context(ctx);
104  ParameterDirectory params;
105  int n_h = 1200;
106  int n_y = 10;
107 
108  // # Network for training with labeled data
109  int batch_size_l = 100;
110  auto xl = make_shared<CgVariable>(Shape_t({batch_size_l, 1, 28, 28}), false);
111  auto tl = make_shared<CgVariable>(Shape_t({batch_size_l, 1}), false);
112  auto yl = mlp_net(xl, n_h, n_y, params, false);
113  auto loss_l = f::mean(f::softmax_cross_entropy(yl, tl, 1), {0, 1}, false);
114 
115  // # Network for training with unlabeled data
116  int batch_size_u = 250;
117  float xi = 10.0;
118  float eps = 1.5;
119  auto xu = make_shared<CgVariable>(Shape_t({batch_size_u, 1, 28, 28}), false);
120  auto tu = make_shared<CgVariable>(Shape_t({batch_size_u, 1}), false);
121  auto yu = mlp_net(xu, n_h, n_y, params, false);
122  auto y1 = make_shared<CgVariable>(yu->variable(), true);
123  y1->set_need_grad(false);
124 
125  auto noise =
126  make_shared<CgVariable>(Shape_t({batch_size_u, 1, 28, 28}), true);
127  auto r = noise / f::pow_scalar(
128  (f::sum(f::pow_scalar(noise, 2), {1, 2, 3}, true)), 0.5);
129  r->set_persistent(true);
130  auto y2 = mlp_net(xu + xi * r, n_h, n_y, params, false);
131  auto y3 = mlp_net(xu + eps * r, n_h, n_y, params, false);
132  auto loss_k = f::mean(distance(y1, y2), {0}, false);
133  auto loss_u = f::mean(distance(y1, y3), {0}, false);
134 
135  // # Network for evaluation with validation data
136  int batch_size_v = 10000;
137  auto xv = make_shared<CgVariable>(Shape_t({batch_size_v, 1, 28, 28}), false);
138  auto hv = mlp_net(xv, n_h, n_y, params, true);
139  auto tv = make_shared<CgVariable>(Shape_t({batch_size_v, 1}), false);
140  auto err = f::mean(f::top_n_error(hv, tv, 1, 1), {0, 1}, false);
141 
142  // # Solver.
143  float learning_rate = 2.0e-3;
144  auto solver = create_AdamSolver(ctx, learning_rate, 0.9, 0.999, 1.0e-8);
145  solver->set_parameters(params.get_parameters());
146 
147  // # Training loop.
148  MnistDataIterator train_data_iterator("train");
149  MnistDataIterator valid_data_iterator("valid");
150  auto labeled_set = train_data_iterator.get_batch(100);
151  int max_iter = 24000;
152  float learning_rate_decay = 0.9;
153  int learning_rate_decay_interval = 240;
154  float weight_decay = 0.0e-4;
155  int max_iter_power_method = 1;
156  int n_val_interval = 240;
157  float mean_tloss = 0.;
158  FILE *fp;
159  fp = fopen("log.txt", "wt");
160  if (fp == NULL) {
161  fprintf(stderr, "Error in opening log file.");
162  return false;
163  }
164  for (int iter = 0; iter < max_iter; iter++) {
165  // Training with labeled data
166  set_data(cpu_ctx, labeled_set, xl, tl);
167  solver->zero_grad();
168  loss_l->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
169  loss_l->variable()->grad()->fill(1.0);
170  loss_l->backward(/*NdArrayPtr grad=*/nullptr, /*clear_buffer=*/true);
171  solver->weight_decay(weight_decay);
172  solver->update();
173 
174  // Training with unlabeled data
175  train_data_iterator.provide_data(cpu_ctx, batch_size_u, xu, tu);
176  yu->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/false);
177 
178  // Calculating virtual adversarial noise
179  float_t *n_d =
180  noise->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, true);
181  for (int i = 0; i < noise->variable()->size(); i++, n_d++)
182  *n_d = normal(engine);
183 
184  for (int k = 0; k < max_iter_power_method; k++) {
185  r->variable()->grad()->fill(0.0);
186  loss_k->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
187  loss_k->variable()->grad()->fill(1.0);
188  loss_k->backward(/*NdArrayPtr grad=*/nullptr, /*clear_buffer=*/true);
189 
190  n_d =
191  noise->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, true);
192  float_t *r_g =
193  r->variable()->cast_grad_and_get_pointer<float_t>(cpu_ctx, false);
194  for (int i = 0; i < noise->variable()->size(); i++, n_d++, r_g++)
195  *n_d = *r_g;
196  }
197 
198  // Updating with virtual adversarial noise
199  solver->zero_grad();
200  loss_u->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
201  loss_u->variable()->grad()->fill(1.0);
202  loss_u->backward(/*NdArrayPtr grad=*/nullptr, /*clear_buffer=*/true);
203  solver->weight_decay(weight_decay);
204  solver->update();
205 
206  // ##### Learning rate update #####
207  if (iter % learning_rate_decay_interval == 0) {
208  solver->set_learning_rate(solver->learning_rate() * learning_rate_decay);
209  }
210 
211  // Monitor loss and error rate
212  float_t *loss_l_d =
213  loss_l->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, false);
214  float_t *loss_u_d =
215  loss_u->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, false);
216  mean_tloss += *loss_l_d + *loss_u_d;
217 
218  if ((iter + 1) % n_val_interval == 0) {
219 
220  mean_tloss /= n_val_interval;
221 
222  valid_data_iterator.provide_data(cpu_ctx, batch_size_v, xv, tv);
223  err->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
224  float_t *err_d =
225  err->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, false);
226 
227  fprintf(fp, "iter: %d, tloss: %f, verr: %f\n", iter + 1, mean_tloss,
228  *err_d);
229  fprintf(stdout, "iter: %d, tloss: %f, verr: %f\n", iter + 1, mean_tloss,
230  *err_d);
231  mean_tloss = 0;
232  }
233  }
234  fclose(fp);
235  return true;
236 }
