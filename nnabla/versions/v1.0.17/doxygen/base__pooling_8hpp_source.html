86  // compute out shape
87  const Shape_t inshape = inputs[0]->shape();
88  const int s = inshape.size() - kernel_.size();
89 
90  if (stride_.size() == 0) {
91  std::copy(stride_.begin(), stride_.end(), kernel_.begin());
92  } else {
93  NBLA_CHECK(kernel_.size() == stride_.size(), error_code::value,
94  "Length of kernel and stride must be same. "
95  "kernel: %d != stride: %d.",
96  kernel_.size(), stride_.size());
97  NBLA_CHECK(kernel_.size() <= inshape.size(), error_code::value,
98  "Length of kernel must be less than length of inshape. "
99  "kernel: %d != inshape: %d.",
100  kernel_.size(), inshape.size());
101  }
102  NBLA_CHECK(kernel_.size() >= 2 && kernel_.size() <= 3,
103  error_code::not_implemented,
104  "2D and 3D Pooling are only supported so far.");
105 
106  NBLA_CHECK(kernel_.size() == pad_.size(), error_code::value,
107  "Size of kernel and pad must be same. "
108  "kernel: %d != pad: %d).",
109  kernel_.size(), pad_.size());
110 
111  vector<int> shape(kernel_.size());
112  for (int i = 0; i < kernel_.size(); i++) {
113  int _w = inshape[i + s];
114  int _k = kernel_[i];
115  int _s = stride_[i];
116  int _p = pad_[i];
117  shape[i] = (_w + _p - (ignore_border_ ? _k - _p : 1)) / _s + 1;
118  }
119 
120  Shape_t outshape(inshape.size());
121  for (int i = 0; i < inshape.size(); i++) {
122  if (i < s)
123  outshape[i] = inshape[i];
124  else
125  outshape[i] = shape[i - s];
126  }
127 
128  outputs[0]->reshape(outshape, true);
129  }
