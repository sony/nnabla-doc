40 CgVariablePtr res_unit(CgVariablePtr x, bool test,
41  ParameterDirectory parameters) {
42  // from python examples
43  // def res_unit(x, scope):
44  // C = x.shape[1]
45  // with nn.parameter_scope(scope):
46  // with nn.parameter_scope('conv1'):
47  // h = F.elu(bn(PF.convolution(x, C / 2, (1, 1),
48  // with_bias=False)))
49  // with nn.parameter_scope('conv2'):
50  // h = F.elu(
51  // bn(PF.convolution(h, C / 2, (3, 3), pad=(1, 1),
52  // with_bias=False)))
53  // with nn.parameter_scope('conv3'):
54  // h = bn(PF.convolution(h, C, (1, 1), with_bias=False))
55  // return F.elu(F.add2(h, x, inplace=True))
56  Shape_t shape_x = x->variable()->shape();
57  const int n_channel = shape_x[1];
58  pf::ConvolutionOpts opts1 = pf::ConvolutionOpts().with_bias(false);
59  pf::ConvolutionOpts opts2 =
60  pf::ConvolutionOpts().with_bias(false).pad({1, 1});
61  pf::ConvolutionOpts opts3 = pf::ConvolutionOpts().with_bias(false);
62  auto c1 =
63  pf::convolution(x, 1, n_channel / 2, {1, 1}, parameters["conv1"], opts1);
64  auto e1 =
65  f::elu(pf::batch_normalization(c1, !test, parameters["conv1"]), 1.0);
66  auto c2 =
67  pf::convolution(e1, 1, n_channel / 2, {3, 3}, parameters["conv2"], opts2);
68  auto e2 =
69  f::elu(pf::batch_normalization(c2, !test, parameters["conv2"]), 1.0);
70  auto c3 =
71  pf::convolution(e2, 1, n_channel, {1, 1}, parameters["conv3"], opts3);
72  auto e3 =
73  f::elu(pf::batch_normalization(c3, !test, parameters["conv3"]), 1.0);
74  return f::elu(f::add2(e3, x, true), 1.0);
75 }

87 CgVariablePtr model(CgVariablePtr x, bool test, ParameterDirectory parameters) {
88  // from python exmaples
89  // xa = /= 255.0;
90  // xa = augmentation(xa, test, aug);
91  // # Conv1 --> 64 x 32 x 32
92  // with nn.parameter_scope("conv1"):
93  // c1 = F.elu(
94  // bn(PF.convolution(image, 64, (3, 3), pad=(3, 3),
95  // with_bias=False)))
96  // # Conv2 --> 64 x 16 x 16
97  // c2 = F.max_pooling(res_unit(c1, "conv2"), (2, 2))
98  // # Conv3 --> 64 x 8 x 8
99  // c3 = F.max_pooling(res_unit(c2, "conv3"), (2, 2))
100  // # Conv4 --> 64 x 8 x 8
101  // c4 = res_unit(c3, "conv4")
102  // # Conv5 --> 64 x 4 x 4
103  // c5 = F.max_pooling(res_unit(c4, "conv5"), (2, 2))
104  // # Conv5 --> 64 x 4 x 4
105  // c6 = res_unit(c5, "conv6")
106  // pl = F.average_pooling(c6, (4, 4))
107  // with nn.parameter_scope("classifier"):
108  // y = PF.affine(pl, 10)
109  // return y
110 
111  auto xa = x * (1.0 / 255.0);
112  xa = augmentation(x, !test);
113 
114  // Conv1 --> 64 x 32 x 32
115  pf::ConvolutionOpts opts = pf::ConvolutionOpts().with_bias(false).pad({3, 3});
116  auto c1 = pf::convolution(x, 1, 64, {3, 3}, parameters["conv1"], opts);
117  c1 = f::elu(pf::batch_normalization(c1, !test, parameters["conv1"]), 1.0);
118 
119  // Conv2 --> 64 x 16 x 16
120  auto c2 =
121  f::max_pooling(res_unit(c1, test, parameters["conv2"]), {2, 2}, {2, 2});
122 
123  // Conv3 --> 64 x 8 x 8
124  auto c3 =
125  f::max_pooling(res_unit(c2, test, parameters["conv3"]), {2, 2}, {2, 2});
126 
127  // Conv4 --> 64 x 8 x 8
128  auto c4 = res_unit(c3, test, parameters["conv4"]);
129 
130  // Conv5 --> 64 x 4 x 4
131  auto c5 =
132  f::max_pooling(res_unit(c4, test, parameters["conv5"]), {2, 2}, {2, 2});
133 
134  // Conv6 --> 64 x 4 x 4
135  auto c6 = res_unit(c5, test, parameters["conv6"]);
136 
137  // affine
138  auto h = f::average_pooling(c6, {4, 4}, {4, 4});
139  auto y = pf::affine(h, 1, 10, parameters["classifier"]);
140  return y;
141 }

146 bool resnet_training_with_static_graph(nbla::Context ctx) {
147 
148  // Create a context for cpu
149  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
150 
151  // Create mnist data iterator
152  MnistDataIterator train_data_provider("train");
153  MnistDataIterator test_data_provider("test");
154 
155  // Setup context
156  SingletonManager::get<GlobalContext>()->set_current_context(ctx);
157 
158  // Build network
159  ParameterDirectory params;
160  int batch_size = 128;
161  auto x = make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
162  auto t = make_shared<CgVariable>(Shape_t({batch_size, 1}), false);
163  auto h_train = model(x, false, params);
164  auto loss = f::mean(f::softmax_cross_entropy(h_train, t, 1), {0, 1}, false);
165  auto h_valid = model(x, true, params);
166  auto err = f::mean(f::top_n_error(h_valid, t, 1, 1), {0, 1}, false);
167 
168  // Setup solver and input learnable parameters
169  auto adam = create_AdamSolver(ctx, 0.001, 0.9, 0.999, 1.0e-8);
170  adam->set_parameters(params.get_parameters());
171 
172  // Execute training
173  FILE *fp;
174  fp = fopen("log.txt", "wt");
175  if (fp == NULL) {
176  fprintf(stderr, "Error in opening log file.");
177  return false;
178  }
179 
180  try {
181  int max_iter = 10000;
182  int n_val_iter = 10;
183  float mean_t_loss = 0.;
184  for (int iter = 0; iter < max_iter; iter++) {
185 
186  // Get batch and copy to input variables
187  train_data_provider.provide_data(cpu_ctx, batch_size, x, t);
188 
189  // Execute forward, backward and update
190  adam->zero_grad();
191  loss->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
192  loss->variable()->grad()->fill(1.0);
193  loss->backward(/*NdArrayPtr grad =*/nullptr,
194  /*bool clear_buffer = */ true);
195  adam->update();
196 
197  // Get and print the average loss
198  float_t *t_loss_d =
199  loss->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, false);
200  mean_t_loss += t_loss_d[0];
201 
202  if ((iter + 1) % n_val_iter == 0) {
203 
204  float mean_v_err = 0.0;
205  for (int v_iter = 0; v_iter < 10; v_iter++) {
206  test_data_provider.provide_data(cpu_ctx, batch_size, x, t);
207  err->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
208  float_t *v_err_d =
209  err->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx,
210  false);
211  mean_v_err += v_err_d[0];
212  }
213  mean_t_loss /= n_val_iter;
214  mean_v_err /= 10;
215 
216  fprintf(fp, "iter: %d, tloss: %f, verr: %f\n", iter + 1, mean_t_loss,
217  mean_v_err);
218  fprintf(stdout, "iter: %d, tloss: %f, verr: %f\n", iter + 1,
219  mean_t_loss, mean_v_err);
220  mean_t_loss = 0;
221  }
222  }
223  } catch (...) {
224  cout << "Exception in resnet_training_with_static_graph.\n";
225  fclose(fp);
226  return false;
227  }
228  fclose(fp);
229  return true;
230 }
