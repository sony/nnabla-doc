128 const vector<Pair> MnistDataIterator::get_batch(int batch_size) {
129 
130  const int n_data = dataset_.size();
131 
132  vector<Pair> batch_data(batch_size);
133 
134  int n_count = 0;
135  if (n_data <= current_idx_ + batch_size)
136  n_count = n_data - current_idx_;
137  for (int i = 0; i < n_count; i++)
138  batch_data[i] = dataset_[current_idx_++];
139 
140  if (n_data <= current_idx_) {
141  permutation();
142  current_idx_ = 0;
143  }
144 
145  for (int i = n_count; i < batch_size; i++)
146  batch_data[i] = dataset_[current_idx_++];
147 
148  return batch_data;
149 }

154 bool mnist_training(nbla::Context ctx, string nnp_file_name) {
155 
156  // Create a context for cpu
157  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
158 
159  // Create mnist data iterator
160  MnistDataIterator data_iterator;
161 
162  // Get filename of the initialized model.
163  const std::string nnp_file(nnp_file_name);
164 
165  // Create a Nnp object
166  nbla::utils::nnp::Nnp nnp(ctx);
167 
168  // Set nnp file to Nnp object.
169  if (!nnp.add(nnp_file)) {
170  fprintf(stderr, "Error in loading nnp file.");
171  return false;
172  }
173 
174  // Get network.
175  auto net = nnp.get_network("training");
176  int batch_size = 128;
177  net->set_batch_size(batch_size);
178 
179  // Get input data as a CPU array.
180  nbla::CgVariablePtr x = net->get_variable("x");
181  nbla::CgVariablePtr t = net->get_variable("t");
182  nbla::CgVariablePtr loss = net->get_variable("loss");
183 
184  // Setup solver and input learnable parameters
185  auto adam = create_AdamSolver(ctx, 0.001, 0.9, 0.999, 1.0e-8);
186  adam->set_parameters(nnp.get_parameters());
187 
188  // Execute training
189  FILE *fp;
190  fp = fopen("log.txt", "wt");
191  if (fp == NULL) {
192  fprintf(stderr, "Error in opening log file.");
193  return false;
194  }
195  int max_iter = 10000;
196  int n_val_iter = 10;
197  float mean_loss = 0.;
198  for (int iter = 0; iter < max_iter; iter++) {
199 
200  // Get batch and copy to input variables
201  const vector<Pair> batch = data_iterator.get_batch(batch_size);
202  float_t *x_d = x->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx);
203  uint8_t *t_d = t->variable()->cast_data_and_get_pointer<uint8_t>(cpu_ctx);
204 
205  // The following stride represents the tensor size less than the specified
206  // index.
207  // e.g. If the tensor size of x is (b, c, w, h) then stride_x becomes c * w
208  // * h
209  const int stride_x = x->variable()->strides()[0];
210  for (int i = 0; i < batch_size; i++) {
211  for (int j = 0; j < stride_x; j++)
212  x_d[i * stride_x + j] = (float_t)(batch[i].data[j]);
213  t_d[i] = batch[i].label;
214  }
215 
216  // Execute forward, backward and update
217  adam->zero_grad();
218  loss->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
219  loss->variable()->grad()->fill(1.0);
220  loss->backward(/*NdArrayPtr grad =*/nullptr, /*bool clear_buffer = */ true);
221  adam->update();
222 
223  // Get and print the average loss
224  float_t *loss_d =
225  loss->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx);
226  mean_loss += loss_d[0];
227  if ((iter + 1) % n_val_iter == 0) {
228  mean_loss /= n_val_iter;
229  fprintf(fp, "iter: %d, loss: %f\n", iter + 0, mean_loss);
230  std::cout << "iter: " << iter + 0 << ", loss: " << mean_loss << std::endl;
231  mean_loss = 0;
232  }
233  }
234  fclose(fp);
235 
236  // Save parameters
237  if (!nnp.save_parameters("parameter.protobuf")) {
238  fprintf(stderr, "Error in saving parameters.");
239  return false;
240  }
241 
242  return true;
243 }
