46 CgVariablePtr generator(CgVariablePtr z, int max_h, bool test,
47  ParameterDirectory params) {
48  // Building generator network which takes (B, Z, 1, 1) inputs and generates
49  //(B, 1, 28, 28) outputs.
50 
51  assert(max_h / 4 > 0);
52 
53  // # (Z, 1, 1) --> (256, 4, 4)
54  pf::ConvolutionOpts opts1 = pf::ConvolutionOpts().with_bias(false);
55  auto h1 = pf::deconvolution(z, 1, max_h, {4, 4}, params["deconv1"], opts1);
56  auto b1 = pf::batch_normalization(h1, !test, params["deconv1"]);
57  auto e1 = f::elu(b1, 1.0);
58 
59  // # (256, 4, 4) --> (128, 8, 8)
60  pf::ConvolutionOpts opts2 =
61  pf::ConvolutionOpts().with_bias(false).pad({1, 1}).stride({2, 2});
62  auto h2 =
63  pf::deconvolution(e1, 1, max_h / 2, {4, 4}, params["deconv2"], opts2);
64  auto b2 = pf::batch_normalization(h2, !test, params["deconv2"]);
65  auto e2 = f::elu(b2, 1.0);
66 
67  // # (128, 8, 8) --> (64, 16, 16)
68  pf::ConvolutionOpts opts3 =
69  pf::ConvolutionOpts().with_bias(false).pad({1, 1}).stride({2, 2});
70  auto h3 =
71  pf::deconvolution(e2, 1, max_h / 4, {4, 4}, params["deconv3"], opts3);
72  auto b3 = pf::batch_normalization(h3, !test, params["deconv3"]);
73  auto e3 = f::elu(b3, 1.0);
74 
75  // # (64, 16, 16) --> (32, 28, 28)
76  // # Convolution with kernel=4, pad=3 and stride=2 transforms a 28 x 28 map
77  // # to a 16 x 16 map. Deconvolution with those parameters behaves like an
78  // # inverse operation, i.e. maps 16 x 16 to 28 x 28.
79  pf::ConvolutionOpts opts4 =
80  pf::ConvolutionOpts().with_bias(false).pad({3, 3}).stride({2, 2});
81  auto h4 =
82  pf::deconvolution(e3, 1, max_h / 4, {4, 4}, params["deconv4"], opts4);
83  auto b4 = pf::batch_normalization(h4, !test, params["deconv4"]);
84  auto e4 = f::elu(b4, 1.0);
85 
86  // # (32, 28, 28) --> (1, 28, 28)
87  pf::ConvolutionOpts opts5 = pf::ConvolutionOpts().pad({1, 1});
88  auto h5 = pf::convolution(e4, 1, 1, {3, 3}, params["conv5"], opts5);
89  auto x = f::sigmoid(h5);
90  return x;
91 }

93 CgVariablePtr discriminator(CgVariablePtr x, int max_h, bool test,
94  ParameterDirectory params) {
95  // Building discriminator network which maps a (B, 1, 28, 28) input to
96  // a (B, 1).
97 
98  assert(max_h / 8 > 0);
99 
100  // # (1, 28, 28) --> (32, 16, 16)
101  pf::ConvolutionOpts opts1 =
102  pf::ConvolutionOpts().pad({3, 3}).stride({2, 2}).with_bias(false);
103  auto h1 = pf::convolution(x, 1, max_h / 8, {3, 3}, params["conv1"], opts1);
104  auto b1 = pf::batch_normalization(h1, !test, params["conv1"]);
105  auto e1 = f::elu(b1, 1.0);
106 
107  // # (32, 16, 16) --> (64, 8, 8)
108  pf::ConvolutionOpts opts2 =
109  pf::ConvolutionOpts().pad({1, 1}).stride({2, 2}).with_bias(false);
110  auto h2 = pf::convolution(e1, 1, max_h / 4, {3, 3}, params["conv2"], opts2);
111  auto b2 = pf::batch_normalization(h2, !test, params["conv2"]);
112  auto e2 = f::elu(b2, 1.0);
113 
114  // # (64, 8, 8) --> (128, 4, 4)
115  pf::ConvolutionOpts opts3 =
116  pf::ConvolutionOpts().pad({1, 1}).stride({2, 2}).with_bias(false);
117  auto h3 = pf::convolution(e2, 1, max_h / 2, {3, 3}, params["conv3"], opts3);
118  auto b3 = pf::batch_normalization(h3, !test, params["conv3"]);
119  auto e3 = f::elu(b3, 1.0);
120 
121  // # (128, 4, 4) --> (256, 4, 4)
122  pf::ConvolutionOpts opts4 =
123  pf::ConvolutionOpts().pad({1, 1}).with_bias(false);
124  auto h4 = pf::convolution(e3, 1, max_h, {3, 3}, params["conv4"], opts4);
125  auto b4 = pf::batch_normalization(h4, !test, params["conv4"]);
126 
127  // # (256, 4, 4) --> (1,)
128  auto f1 = pf::affine(b4, 1, 1, params["fc1"]);
129  return f1;
130 }

155 bool dcgan_training_with_static_graph(nbla::Context ctx) {
156 
157  // Create a context for cpu
158  nbla::Context cpu_ctx{{"cpu:float"}, "CpuCachedArray", "0"};
159 
160  // Setup context
161  SingletonManager::get<GlobalContext>()->set_current_context(ctx);
162 
163  // Build network
164  int max_h = 256;
165  int batch_size = 64;
166  ParameterDirectory params;
167 
168  nbla::utils::load_parameters(params, "dcgan_param.protobuf");
169 
170  // # Fake path
171  auto z = make_shared<CgVariable>(Shape_t({batch_size, 100, 1, 1}), false);
172  auto fake = generator(z, max_h, false, params["gen"]);
173  fake->set_persistent(true);
174 
175  auto pred_fake = discriminator(fake, max_h, false, params["dis"]);
176  auto loss_gen = f::mean(
177  f::sigmoid_cross_entropy(pred_fake, f::constant(1, {batch_size, 1})),
178  {0, 1}, false);
179 
180  auto fake_dis = make_shared<CgVariable>(fake->variable(), true);
181  fake_dis->set_need_grad(true);
182  auto pred_fake_dis = discriminator(fake, max_h, false, params["dis"]);
183  pred_fake_dis->set_persistent(true);
184  auto loss_dis = f::mean(
185  f::sigmoid_cross_entropy(pred_fake_dis, f::constant(0, {batch_size, 1})),
186  {0, 1}, false);
187 
188  // # Real path
189  auto x = make_shared<CgVariable>(Shape_t({batch_size, 1, 28, 28}), false);
190  auto t = make_shared<CgVariable>(Shape_t({batch_size, 1}), false);
191  auto pred_real = discriminator(x, max_h, false, params["dis"]);
192  loss_dis = loss_dis + f::mean(f::sigmoid_cross_entropy(
193  pred_real, f::constant(1, {batch_size, 1})),
194  {0, 1}, false);
195 
196  // # Create Solver.
197  float learning_rate = 2.0e-4;
198  auto solver_gen = create_AdamSolver(ctx, learning_rate, 0.5, 0.999, 1.0e-8);
199  auto solver_dis = create_AdamSolver(ctx, learning_rate, 0.5, 0.999, 1.0e-8);
200  solver_gen->set_parameters(params["gen"].get_parameters());
201  solver_dis->set_parameters(params["dis"].get_parameters());
202 
203  // Create mnist data iterator
204  MnistDataIterator train_data_provider("train");
205 
206  //# Training loop.
207  FILE *fp;
208  fp = fopen("log.txt", "wt");
209  if (fp == NULL) {
210  fprintf(stderr, "Error in opening log file.");
211  return false;
212  }
213 
214  try {
215  int max_iter = 20000;
216  int n_val_iter = 10;
217  float weight_decay = 1.0e-4;
218  float mean_loss_gen = 0.;
219  float mean_loss_dis = 0.;
220  for (int iter = 0; iter < max_iter; iter++) {
221 
222  // Get batch and copy to input variables
223  train_data_provider.provide_data(cpu_ctx, batch_size, x, t);
224 
225  float_t *z_d =
226  z->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx, true);
227  for (int i = 0; i < z->variable()->size(); i++, z_d++)
228  *z_d = normal(engine);
229 
230  // # Generator update.
231  solver_gen->zero_grad();
232  loss_gen->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
233  loss_gen->variable()->grad()->fill(1.0);
234  loss_gen->backward(/*NdArrayPtr grad=*/nullptr, /*clear_buffer=*/true);
235  solver_gen->weight_decay(weight_decay);
236  solver_gen->update();
237 
238  // # Discriminator update.
239  solver_dis->zero_grad();
240  loss_dis->forward(/*clear_buffer=*/false, /*clear_no_need_grad=*/true);
241  loss_dis->variable()->grad()->fill(1.0);
242  loss_dis->backward(/*NdArrayPtr grad=*/nullptr, /*clear_buffer=*/true);
243  solver_dis->weight_decay(weight_decay);
244  solver_dis->update();
245 
246  // Get and print the average loss
247  float_t *loss_gen_d =
248  loss_gen->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx,
249  false);
250  mean_loss_gen += loss_gen_d[0];
251 
252  float_t *loss_dis_d =
253  loss_dis->variable()->cast_data_and_get_pointer<float_t>(cpu_ctx,
254  false);
255  mean_loss_dis += loss_dis_d[0];
256 
257  if ((iter + 1) % n_val_iter == 0) {
258  mean_loss_gen /= n_val_iter;
259  mean_loss_dis /= n_val_iter;
260  fprintf(fp, "iter: %d, loss_gen: %f, loss_dis: %f\n", iter + 0,
261  mean_loss_gen, mean_loss_dis);
262  fprintf(stdout, "iter: %d, loss_gen: %f, loss_dis: %f\n", iter + 0,
263  mean_loss_gen, mean_loss_dis);
264  mean_loss_gen = 0.;
265  mean_loss_dis = 0.;
266 
267  nbla::utils::save_parameters(params, "saved_dcgan_param.protobuf");
268  }
269 
270  // Get generated images"
271  if ((iter + 1) % max_iter == 0)
272  print_array(fake);
273  }
274  } catch (...) {
275  cout << "Exception in dcgan_training_with_static_graph.\n";
276  fclose(fp);
277  return false;
278  }
279  fclose(fp);
280  return true;
281 }
