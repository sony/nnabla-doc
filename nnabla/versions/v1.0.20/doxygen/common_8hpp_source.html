49 inline Size_t compute_size_by_shape(const Shape_t &shape, Size_t axis = 0) {
50  axis = std::max(static_cast<Size_t>(0), axis);
51  NBLA_CHECK(axis <= shape.size(), error_code::value,
52  "axis must be less than or equal to size of shape. "
53  "axis: %ld > size of shape: %ld.",
54  axis, shape.size());
55  Size_t ret = 1;
56  for (int i = axis; i < shape.size(); ++i) {
57  ret *= shape[i];
58  }
59  return ret;
60 }

64 inline Shape_t get_c_contiguous_strides(const Shape_t &shape) {
65  if (!shape.size()) {
66  return Shape_t(); // empty strides for 0-shaped array
67  }
68  Shape_t strides(shape.size(), 1);
69  for (int i = strides.size() - 2; i >= 0; --i) {
70  strides[i] *= strides[i + 1] * shape[i + 1];
71  }
72  return strides;
73 }
