24 template <typename SINGLETON> SINGLETON *SingletonManager::get() {
25  static std::mutex mtx_;
26  std::lock_guard<std::mutex> lock(mtx_);
27 
28  static SINGLETON *r = nullptr;
29  if (r)
30  return r;
31  SingletonManager &s = get_self();
32  // TODO: Enable debug print
33  // std::cout << "Creating a singleton \"" << typeid(SINGLETON).name() << "\""
34  // << std::endl;
35  r = new SINGLETON{};
36 
37  auto deleter = []() -> void {
38  // TODO: Enable debug print
39  // std::cout << "Deleting a singleton \"" << typeid(SINGLETON).name() <<
40  // "\""
41  // << std::endl;
42  delete r; // Static variable doesn't require capturing.
43  r = nullptr;
44  };
45  int id = s.count_;
46  s.singletons_.insert({id, {(uintptr_t)r, deleter}}); // Register deleter
47  s.adr2id_.insert({(uintptr_t)r, id}); // Register ID
48  s.count_ += 1;
49  return r;
50 }
