49 template <typename T> struct float_bits : public nbits<T> {
50  typedef typename nbits<T>::bit_t bit_t;
51  typedef typename std::make_signed<typename nbits<T>::bit_t>::type sbit_t;
52  static constexpr bit_t sign_mask = (bit_t)1 << (nbits<T>::n - 1);
53  static constexpr bit_t exp_mask = ((bit_t)1 << nbits<T>::nexp) - 1;
54  static constexpr bit_t inf_bits = exp_mask << nbits<T>::nsig;
55  static constexpr bit_t sig_mask = ((bit_t)1 << nbits<T>::nsig) - 1;
56  static constexpr sbit_t exp_bias = ((bit_t)1 << (nbits<T>::nexp - 1)) - 1;
57  static constexpr sbit_t exp_max = exp_bias;
58  static constexpr sbit_t exp_min = -exp_bias + 1;
59  static constexpr bit_t implied_bit = sig_mask + 1;
60  template <typename U> static constexpr bit_t diff_sig() {
61  static_assert(nbits<T>::n > nbits<U>::n, "T must be larger than U.");
62  return nbits<T>::nsig - nbits<U>::nsig;
63  }
64  static bit_t get_significand_bits(bit_t bits) { return bits & sig_mask; }
65  static sbit_t get_signed_exponent(bit_t bits) {
66  return (sbit_t)((bits >> nbits<T>::nsig) & exp_mask) - exp_bias;
67  }
68  template <typename U>
69  static typename nbits<U>::bit_t get_sign_bit_of(bit_t bits) {
70  static_assert(nbits<T>::n > nbits<U>::n, "T must be larger than U.");
71  return (bits >> (nbits<T>::n - nbits<U>::n)) & float_bits<U>::sign_mask;
72  }
73  template <typename U>
74  static bit_t get_sign_bit_from(typename nbits<U>::bit_t bits) {
75  static_assert(nbits<T>::n > nbits<U>::n, "T must be larger than U.");
76  return ((bit_t)(bits) << (nbits<T>::n - nbits<U>::n)) &
77  float_bits<T>::sign_mask;
78  }
79  template <typename U>
80  static typename float_bits<U>::bit_t signed_exp_to_bits(sbit_t signed_exp) {
81  return (signed_exp + float_bits<U>::exp_bias) << nbits<U>::nsig;
82  }
83  template <typename U> static void round_ties_to_even(bit_t &sig_bits) {
84  // Mask the last bit of U and the following bits.
85  constexpr bit_t mask = (1 << (diff_sig<U>() + 1)) - 1;
86  // Tied value (goes to even).
87  constexpr bit_t tie = 1 << (diff_sig<U>() - 1);
88  // tie 010...0 goes to 000...0, another tie 110...0 goes to (+1)000...0
89  if ((sig_bits & mask) != tie) {
90  sig_bits += tie;
91  }
92  }

105  template <typename U>
106  static typename nbits<U>::bit_t downconvert_to(bit_t bits) {
107  static_assert(nbits<T>::n > nbits<U>::n,
108  "Downconverting to bigger type is prohibited.");
109  typedef typename nbits<U>::bit_t tbit_t;
110 
111  // 1. Parse the float bits into sign, exponent and significand.
112  tbit_t tsign_bit = get_sign_bit_of<U>(bits);
113  sbit_t signed_exp = get_signed_exponent(bits);
114  bit_t sig_bits = get_significand_bits(bits);
115 
116  // 2. T value is NaN / inf.
117  if (signed_exp == exp_max + 1) {
118  if (sig_bits == 0) {
119  // inf
120  return tsign_bit | float_bits<U>::inf_bits;
121  } else {
122  // NaN: Propagate signaling NaN.
123  tbit_t tsig_bits = sig_bits >> diff_sig<U>();
124  if (tsig_bits == 0) {
125  tsig_bits += 1; // avoid inf
126  }
127  return tsign_bit | float_bits<U>::inf_bits | tsig_bits;
128  }
129  }
130 
131  // 3. If exponent overflows in target type, convert to signed inf.
132  if (signed_exp > float_bits<U>::exp_max) {
133  return tsign_bit | float_bits<U>::inf_bits;
134  }
135 
136  // 4. If exponent is lower than subnormal of U, return signed 0;
137  if (signed_exp < (float_bits<U>::exp_min - nbits<U>::nsig)) {
138  return tsign_bit;
139  }
140 
141  // 5. If exponent is in a range of exponent of U, convert to normal.
142  if (signed_exp >= float_bits<U>::exp_min) {
143  round_ties_to_even<U>(sig_bits);
144  // Note: Rounding may cause a bit spilling into exponent bits.
145  // `+` operator is used, which will produce correct results.
146  return tsign_bit | (signed_exp_to_bits<U>(signed_exp) +
147  (tbit_t)(sig_bits >> diff_sig<U>()));
148  }
149 
150  // 6. Rest is subnormal.
151  // 6-A. Add implied bit since T is normal.
152  sig_bits |= implied_bit;
153  // 6-B. Shift significand by exponent while canceling the effect of minimum
154  // exponent of target type.
155  int rshift = -signed_exp + float_bits<U>::exp_min;
156  sig_bits >>= rshift;
157  // 6-C. Round ties-to-even.
158  round_ties_to_even<U>(sig_bits);
159  return tsign_bit | (tbit_t)(sig_bits >> diff_sig<U>());
160  }

164  template <typename U> static U upconvert_to(bit_t bits) {
165  static_assert(nbits<U>::n > nbits<T>::n, "U must be larger than T.");
166  constexpr auto dsig = float_bits<U>::template diff_sig<T>();
167  typedef typename float_bits<U>::bit_t tbit_t;
168  // 0. Parse bits.
169  tbit_t tsign_bit = float_bits<U>::template get_sign_bit_from<T>(bits);
170  sbit_t signed_exp = get_signed_exponent(bits);
171  bit_t sig_bits = get_significand_bits(bits);
172 
173  // 1. NaN or inf.
174  if (signed_exp == exp_max + 1) {
175  // Just shifting significand.
176  return tsign_bit | float_bits<U>::inf_bits |
177  (((tbit_t)(sig_bits)) << dsig);
178  }
179 
180  // 2. 0 or subnormal.
181  if (signed_exp == exp_min - 1) {
182  // 1-A. 0
183  if (sig_bits == 0) {
184  return tsign_bit;
185  }
186  // 1-B. Subnormal
187  // Iterate left-shift until getting normalized expression
188  // `1.(significand)`.
189  sig_bits <<= 1;
190  // texp_bits starts from -exp_bias and decrements until get normalized
191  // expression.
192  tbit_t texp_bits = float_bits<U>::exp_bias - exp_bias;
193  while ((sig_bits & implied_bit) == 0) {
194  sig_bits <<= 1;
195  texp_bits--;
196  }
197  texp_bits <<= float_bits<U>::nsig;
198  tbit_t tsig_bits = ((tbit_t)(get_significand_bits(sig_bits))) << dsig;
199  return tsign_bit | texp_bits | tsig_bits;
200  }
201 
202  // 3. Normlized.
203  return tsign_bit | ((tbit_t)(signed_exp + float_bits<U>::exp_bias)
204  << float_bits<U>::nsig) |
205  ((tbit_t)(sig_bits) << dsig);
206  }
